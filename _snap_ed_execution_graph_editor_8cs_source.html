103  public override T CreateLink<T>(Graph graph, GraphPin output, GraphPin input, UISystem uiSystem)
104  {
105  if (input != null && input.Node != null)
106  {
107  input = input.Node.InputPin;
108  }
109 
110  // If we have a link in the opposite direction, then break that link
111  var sourceNode = output.Node;
112  var destNode = input.Node;
113  if (sourceNode != null && destNode != null)
114  {
115  var links = destNode.OutputPin.GetConntectedLinks();
116  foreach (var link in links)
117  {
118  if (link.Output.Node == input.Node && link.Input.Node == output.Node)
119  {
120  GraphOperations.DestroyLink(link, uiSystem.Undo);
121  }
122  }
123 
124  if (sourceNode is GrammarExecEntryNode)
125  {
126  // Destroy all outgoing links first
127  var outgoingLinks = output.GetConntectedLinks();
128  foreach (var link in outgoingLinks)
129  {
130  GraphOperations.DestroyLink(link, uiSystem.Undo);
131  }
132  }
133  }
134 
135  if (input.Node != null)
136  {
137  input = input.Node.InputPin;
138  if (input != null)
139  {
140  return base.CreateLink<T>(graph, output, input, uiSystem);
141  }
142  }
