155  protected GameObject BuildSpriteObject(SpriteDungeonThemeItem spriteData, Matrix4x4 transform, string NodeId) {
156  if (spriteData.sprite == null) return null;
157  var gameObj = new GameObject(spriteData.sprite.name);
158 
159  // Setup the sprite
160  var spriteRenderer = gameObj.AddComponent<SpriteRenderer>();
161  spriteRenderer.sprite = spriteData.sprite;
162  spriteRenderer.color = spriteData.color;
163  spriteRenderer.sortingOrder = spriteData.orderInLayer;
164 
165  if (spriteData.materialOverride != null) {
166  spriteRenderer.material = spriteData.materialOverride;
167  }
168  if (spriteData.sortingLayerName != null && spriteData.sortingLayerName.Length > 0) {
169  spriteRenderer.sortingLayerName = spriteData.sortingLayerName;
170  }
171 
172  // Setup the sprite collision
173  var collisionType = spriteData.collisionType;
174  if (collisionType != DungeonSpriteCollisionType.None) {
175  Vector2 baseScaleMultiplier = Vector2.one;
176  var sprite = spriteData.sprite;
177  var pixelsPerUnit = sprite.pixelsPerUnit;
178  baseScaleMultiplier.x = sprite.rect.width / pixelsPerUnit;
179  baseScaleMultiplier.y = sprite.rect.height / pixelsPerUnit;
180 
181  Collider2D collider = null;
182  if (collisionType == DungeonSpriteCollisionType.Box) {
183  var boxCollider = gameObj.AddComponent<BoxCollider2D>();
184  boxCollider.size = Vector3.Scale(spriteData.physicsSize, baseScaleMultiplier);
185  collider = boxCollider;
186  }
187  else if (collisionType == DungeonSpriteCollisionType.Circle) {
188  var circleCollider = gameObj.AddComponent<CircleCollider2D>();
189  circleCollider.radius = spriteData.physicsRadius * baseScaleMultiplier.x;
190  collider = circleCollider;
191  }
192  else if (collisionType == DungeonSpriteCollisionType.Polygon) {
193  collider = gameObj.AddComponent<PolygonCollider2D>();
194  }
195 
196  if (collider != null) {
197  collider.sharedMaterial = spriteData.physicsMaterial;
198  collider.offset = Vector3.Scale(spriteData.physicsOffset, baseScaleMultiplier);
199  }
200  }
201 
202 
203  // Set the transform
204  Matrix.DecomposeMatrix(ref transform, out _position, out _rotation, out _scale);
205  gameObj.transform.position = _position;
206  gameObj.transform.rotation = _rotation;
207  gameObj.transform.localScale = _scale;
208 
209  // Setup dungeon related parameters
210  if (itemParent != null) {
211  gameObj.transform.SetParent(itemParent.transform, true);
212  }
213 
214  var data = gameObj.AddComponent<DungeonSceneProviderData> ();
215  data.NodeId = NodeId;
216  data.dungeon = dungeon;
217  data.affectsNavigation = spriteData.affectsNavigation;
218 
219  return gameObj;
220  }
