25  public static Vector3 GetRuleNodeWorldPosition(SCRuleNode ruleNode, SCReferenceNode referenceNode, PropSocket marker, Vector3 gridSize, ref Matrix4x4 rotationFrame)
26  {
27  var scGraph = ruleNode.Graph as SpatialConstraintGraph;
28  var relativeToMarkerRotation = scGraph.asset.checkRelativeToMarkerRotation;
29 
30  var markerTransform = marker.Transform;
31 
32  Vector2 offset2D = (ruleNode.Bounds.center - referenceNode.Bounds.center) / SCBaseDomainNode.TileSize;
33  Vector3 offset = new Vector3(offset2D.x, 0, offset2D.y);
34  if (relativeToMarkerRotation)
35  {
36  offset = RotateVector(offset, markerTransform);
37  }
38  offset = RotateVector(offset, rotationFrame);
39 
40  // Find the world position of the marker and the location to search (relative to the spatial constraint node from the SC reference node)
41  var baseMarkerPosition = Matrix.GetTranslation(ref marker.Transform);
42  var ruleNodePosition = baseMarkerPosition + Vector3.Scale(offset, gridSize);
43  return ruleNodePosition;
44  }

68  bool ProcessSpatialConstraintFrame(SpatialConstraintProcessorContext context, SpatialConstraintRuleDomain domain, Matrix4x4 rotationFrame, out PropSocket[] outMarkersToRemove)
69  {
70  var nodes = context.constraintAsset.Graph.Nodes.Where(node => node is SCRuleNode);
71  outMarkersToRemove = new PropSocket[0];
72 
73  if (nodes.Count() == 0)
74  {
75  // No rules specified. Return true by default
76  return true;
77  }
78 
79  var nodeWorldPositions = new Dictionary<SCRuleNode, Vector3>();
80 
81  foreach (var node in nodes)
82  {
83  var ruleNode = node as SCRuleNode;
84 
85  var nodePosition = SpatialConstraintProcessorUtils.GetRuleNodeWorldPosition(ruleNode, domain.referenceNode,
86  context.marker, domain.gridSize, ref rotationFrame);
87  nodeWorldPositions.Add(ruleNode, nodePosition);
88 
89  var constraints = ruleNode.constraints.Where(c => c != null);
90  if (constraints.Count() == 0) continue;
91 
92  bool allRulesPassed = true;
93  bool atleastOneRulePassed = false;
94 
95  foreach (var constraint in constraints)
96  {
97  var ruleContext = new ConstraintRuleContext();
98  ruleContext.processorContext = context;
99  ruleContext.domain = domain;
100  ruleContext.ruleNode = ruleNode;
101  ruleContext.rotationFrame = rotationFrame;
102  ruleContext.ruleNodeWorldPosition = nodePosition;
103 
104 
105  bool success = constraint.Process(ruleContext);
106  if (constraint.inverseRule)
107  {
108  success = !success;
109  }
110 
111  allRulesPassed &= success;
112  atleastOneRulePassed |= success;
113  }
114 
115  if (ruleNode.constraintEvaluationMode == SCRuleNodeEvaluationMode.AllRulesMustPass && !allRulesPassed)
116  {
117  return false;
118  }
119  else if (ruleNode.constraintEvaluationMode == SCRuleNodeEvaluationMode.AtleastOneRuleShouldPass && !atleastOneRulePassed)
120  {
121  return false;
122  }
123  }
124 
125  // The spatial constraint setup has passed
126  // Process removal rules
127  var markersToRemove = new List<PropSocket>();
128  foreach (var node in nodes)
129  {
130  var ruleNode = node as SCRuleNode;
131  if (ruleNode.exclusionRuleMarkersToRemove.Length == 0)
132  {
133  continue;
134  }
135 
136  var radius = ruleNode.exclusionRuleSearchRadius;
137  float searchRadiusSq = radius * radius;
138  var nodePosition3D = nodeWorldPositions[ruleNode];
139  var searchPosition = SpatialConstraintProcessorUtils.GetPosition2D(nodePosition3D);
140  foreach (var markerToRemove in ruleNode.exclusionRuleMarkersToRemove)
141  {
142  var markerSearchSpace = context.levelMarkers.GetMarkersInSearchArea(searchPosition, radius);
143  foreach (var candidateMarker in markerSearchSpace)
144  {
145  var candidateMarkerName = candidateMarker.SocketType;
146  if (ruleNode.exclusionRuleMarkersToRemove.Contains(candidateMarkerName))
147  {
148  var candidateMarkerPosition = SpatialConstraintProcessorUtils.GetPosition2D(Matrix.GetTranslation(ref candidateMarker.Transform));
149  float distanceSq = (searchPosition - candidateMarkerPosition).sqrMagnitude;
150  if (distanceSq < searchRadiusSq)
151  {
152  markersToRemove.Add(candidateMarker);
153  }
154  }
155  }
156  }
157  }
158  outMarkersToRemove = markersToRemove.ToArray();
159 
160  return true;
161  }

163  public bool ProcessSpatialConstraint(SpatialConstraintProcessorContext context, out Matrix4x4 outOffset, out PropSocket[] outMarkersToRemove)
164  {
165  outOffset = Matrix4x4.identity;
166  var domain = GetDomain(context);
167  if (context.constraintAsset != null && context.constraintAsset.Graph != null)
168  {
169  Matrix4x4[] rotationFrames;
170  if (context.constraintAsset.rotateToFit)
171  {
172  rotationFrames = new Matrix4x4[]
173  {
174  Matrix4x4.identity,
175  Matrix4x4.Rotate(Quaternion.Euler(0, 90, 0)),
176  Matrix4x4.Rotate(Quaternion.Euler(0, 180, 0)),
177  Matrix4x4.Rotate(Quaternion.Euler(0, 270, 0)),
178  };
179  }
180  else
181  {
182  rotationFrames = new Matrix4x4[]
183  {
184  Matrix4x4.identity
185  };
186  }
187 
188  foreach (var rotationFrame in rotationFrames)
189  {
190  if (ProcessSpatialConstraintFrame(context, domain, rotationFrame, out outMarkersToRemove))
191  {
192  if (context.constraintAsset.applyFitRotation)
193  {
194  outOffset = rotationFrame;
195  }
196  return true;
197  }
198  }
199  }
200 
201  outMarkersToRemove = new PropSocket[0];
202  return false;
203  }
204  }
