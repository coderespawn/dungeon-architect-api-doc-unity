121  public void AddSpan(Span span)
122  {
123  if (span.Minimum > span.Maximum)
124  {
125  int tmp = span.Minimum;
126  span.Minimum = span.Maximum;
127  span.Maximum = tmp;
128  }
129 
130  //clamp the span to the cell's range of [0, maxHeight]
131  MathHelper.Clamp(ref span.Minimum, 0, height);
132  MathHelper.Clamp(ref span.Maximum, 0, height);
133 
134  lock (spans)
135  {
136  for (int i = 0; i < spans.Count; i++)
137  {
138  //Check whether the current span is below, or overlapping existing spans.
139  //If the span is completely above the current span the loop will continue.
140  Span cur = spans[i];
141  if (cur.Minimum > span.Maximum)
142  {
143  //The new span is below the current one and is not intersecting.
144  spans.Insert(i, span);
145  return;
146  }
147  else if (cur.Maximum >= span.Minimum)
148  {
149  //The new span is colliding with the current one, merge them together.
150  if (cur.Minimum < span.Minimum)
151  span.Minimum = cur.Minimum;
152 
153  if (cur.Maximum == span.Maximum)
154  {
155  //In the case that both spans end at the same voxel, the area gets merged. The new span's area
156  //has priority if both spans are walkable, so the only case where the area gets set is when
157  //the new area isn't walkable and the old one is.
158  if (!span.Area.IsWalkable && cur.Area.IsWalkable)
159  span.Area = cur.Area;
160  }
161  else if (cur.Maximum > span.Maximum)
162  {
163  span.Maximum = cur.Maximum;
164  span.Area = cur.Area;
165  }
166 
167  //Remove the current span and adjust i.
168  //We do this to avoid duplicating the current span.
169  spans.RemoveAt(i);
170  i--;
171  }
172  }
173 
174  //If the span is not inserted, it is the highest span and will be added to the end.
175  spans.Add(span);
176  }
177  }
178  }
