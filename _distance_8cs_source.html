28  internal static float PointToSegmentSquared(ref Vector3 pt, ref Vector3 p, ref Vector3 q)
29  {
30  //distance from P to Q
31  Vector3 pq = q - p;
32 
33  //disance from P to the lone point
34  float dx = pt.X - p.X;
35  float dy = pt.Y - p.Y;
36  float dz = pt.Z - p.Z;
37 
38  float segmentMagnitudeSquared = pq.LengthSquared();
39  float t = pq.X * dx + pq.Y * dy + pq.Z * dz;
40 
41  if (segmentMagnitudeSquared > 0)
42  t /= segmentMagnitudeSquared;
43 
44  //keep t between 0 and 1
45  if (t < 0)
46  t = 0;
47  else if (t > 1)
48  t = 1;
49 
50  dx = p.X + t * pq.X - pt.X;
51  dy = p.Y + t * pq.Y - pt.Y;
52  dz = p.Z + t * pq.Z - pt.Z;
53 
54  return dx * dx + dy * dy + dz * dz;
55  }

224  internal static bool PointToTriangle(Vector3 p, Vector3 a, Vector3 b, Vector3 c, out float height)
225  {
226  Vector3 v0 = c - a;
227  Vector3 v1 = b - a;
228  Vector3 v2 = p - a;
229 
230  float dot00, dot01, dot02, dot11, dot12;
231 
232  Vector3Extensions.Dot2D(ref v0, ref v0, out dot00);
233  Vector3Extensions.Dot2D(ref v0, ref v1, out dot01);
234  Vector3Extensions.Dot2D(ref v0, ref v2, out dot02);
235  Vector3Extensions.Dot2D(ref v1, ref v1, out dot11);
236  Vector3Extensions.Dot2D(ref v1, ref v2, out dot12);
237 
238  //compute barycentric coordinates
239  float invDenom = 1.0f / (dot00 * dot11 - dot01 * dot01);
240  float u = (dot11 * dot02 - dot01 * dot12) * invDenom;
241  float v = (dot00 * dot12 - dot01 * dot02) * invDenom;
242 
243  const float EPS = 1E-4f;
244 
245  //if point lies inside triangle, return interpolated y-coordinate
246  if (u >= -EPS && v >= -EPS && (u + v) <= 1 + EPS)
247  {
248  height = a.Y + v0.Y * u + v1.Y * v;
249  return true;
250  }
251 
252  height = float.MaxValue;
253  return false;
254  }
