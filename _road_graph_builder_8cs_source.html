198  {
199  var bakedNodes = new List<RoadGraphNode>();
200  int _edgeIdCounter = 0;
201 
202  foreach (var node in buildNodes)
203  {
204  var bakedNode = new RoadGraphNode();
205  bakedNode.nodeId = node.nodeId;
206  bakedNode.position = node.position;
207 
208  var edges = new List<RoadGraphEdge>();
209  for (int i = 0; i < node.connectedNodes.Count; i++)
210  {
211  var connectedNode = node.connectedNodes[i];
212  var edgeThickness = node.edgeThickness[i];
213  var edge = new RoadGraphEdge();
214  edge.edgeId = ++_edgeIdCounter;
215  edge.ownerNodeId = node.nodeId;
216  edge.otherNodeId = connectedNode.nodeId;
217  edge.thickness = edgeThickness;
218 
219  // Find the dot product with the x axis
220  Vector3 direction = (connectedNode.position - node.position).normalized;
221  var dot = Vector3.Dot(new Vector3(1, 0, 0), direction);
222  float angle = Mathf.Acos(dot);
223  if (direction.z < 0)
224  {
225  angle = 2 * Mathf.PI - angle;
226  }
227 
228  edge.angleToXAxis = angle;
229 
230  edges.Add(edge);
231  }
232 
233  bakedNode.adjacentEdges = edges.ToArray();
234  System.Array.Sort(bakedNode.adjacentEdges, SortEdgesByAngle);
235  bakedNodes.Add(bakedNode);
236  }
237 
238  var graph = new RoadGraph();
239  graph.nodes = bakedNodes.ToArray();
240  return graph;
241  }
