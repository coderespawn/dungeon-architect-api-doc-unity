74  public BulkReplacementEntry[] Process(DungeonThemeEditorWindow editor)
75  {
76  var graphEditor = editor.GraphEditor;
77  var targetFiles = GetTargetFiles();
78  var replacements = new List<BulkReplacementEntry>();
79  foreach (var node in graphEditor.Graph.Nodes)
80  {
81  if (node is GameObjectNode)
82  {
83  var goNode = node as GameObjectNode;
84  if (goNode.Template != null) {
85  string path = AssetDatabase.GetAssetPath(goNode.Template);
86  var pathInfo = new FileInfo(path);
87  if (pathInfo != null)
88  {
89  if (targetFiles.ContainsKey(pathInfo.Name))
90  {
91  // Found a target file. Replace
92  var replacementEntry = new BulkReplacementEntry();
93  replacementEntry.node = goNode;
94  replacementEntry.currentPrefabInfo = pathInfo;
95  replacementEntry.replacementPrefabInfo = targetFiles[pathInfo.Name];
96 
97  // Make sure the current and target are not the same prefabs, to avoid unnecessary replacements
98  if (replacementEntry.currentPrefabInfo.FullName != replacementEntry.replacementPrefabInfo.FullName)
99  {
100  replacements.Add(replacementEntry);
101  }
102  }
103  }
104  }
105  }
106  }
107 
108  return replacements.ToArray();
109  }
