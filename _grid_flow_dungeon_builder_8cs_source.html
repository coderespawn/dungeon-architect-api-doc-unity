99  Quaternion GetBaseTransform(FlowTilemap tilemap, int x, int y)
100  {
101  var cellTypesToTransform = new FlowTilemapCellType[]
102  {
103  FlowTilemapCellType.Wall,
104  FlowTilemapCellType.Door
105  };
106 
107  var cell = tilemap.Cells[x, y];
108  if (!cellTypesToTransform.Contains(cell.CellType))
109  {
110  return Quaternion.identity;
111  }
112 
113  var validL = IsCellOfType(tilemap, x - 1, y, cellTypesToTransform);
114  var validR = IsCellOfType(tilemap, x + 1, y, cellTypesToTransform);
115  var validB = IsCellOfType(tilemap, x, y - 1, cellTypesToTransform);
116  var validT = IsCellOfType(tilemap, x, y + 1, cellTypesToTransform);
117 
118  var angleY = 0;
119  if (validL && validR)
120  {
121  angleY = validT ? 180 : 0;
122  }
123  else if (validT && validB)
124  {
125  angleY = validR ? 270 : 90;
126  }
127  else if (validL && validT) angleY = 180;
128  else if (validL && validB) angleY = 90;
129  else if (validR && validT) angleY = 270;
130  else if (validR && validB) angleY = 0;
131 
132  return Quaternion.Euler(0, angleY, 0);
133  }

169  void EmitLevelMarkers()
170  {
171  if (gridFlowConfig == null || gridFlowModel == null)
172  {
173  Debug.LogError("GridFlowBuilder: Invalid state");
174  return;
175  }
176 
177  var items = gridFlowModel.layoutGraph.GetAllItems();
178  var itemMap = new Dictionary<DungeonUID, FlowItem>();
179  foreach (var item in items)
180  {
181  itemMap[item.itemId] = item;
182  }
183 
184  var tilemap = gridFlowModel.tilemap;
185  if (tilemap == null)
186  {
187  return;
188  }
189 
190  var basePosition = transform.position;
191  var gridSize = gridFlowConfig.gridSize;
192  // Emit the cell markers
193  for (int x = 0; x < tilemap.Width; x++)
194  {
195  for (int y = 0; y < tilemap.Height; y++)
196  {
197  var position = basePosition + Vector3.Scale(new Vector3(x + 0.5f, 0, y + 0.5f), gridSize);
198  var baseRotation = GetBaseTransform(tilemap, x, y);
199  var markerTransform = Matrix4x4.TRS(position, baseRotation, Vector3.one);
200  var cell = tilemap.Cells[x, y];
201  int cellId = tilemap.Width * y + x;
202 
203  if (cell.Item != DungeonUID.Empty && itemMap.ContainsKey(cell.Item))
204  {
205  var item = itemMap[cell.Item];
206  if (item.markerName != null && item.markerName.Length > 0 && item.type != FlowGraphItemType.Lock)
207  {
208  // Emit this item
209  var itemData = new FlowItemMetadata();
210  itemData.itemId = item.itemId;
211  itemData.itemType = item.type;
212  itemData.referencedItems = new List<DungeonUID>(item.referencedItemIds).ToArray();
213 
214  EmitMarker(item.markerName, markerTransform, new IntVector(x, 0, y), cellId, itemData);
215  }
216  }
217 
218  bool removeElevationMarker = false;
219  if (cell.Overlay != null && cell.Overlay.markerName != null)
220  {
221  var heightOffset = 0.0f;
222  if (cell.Overlay.mergeConfig != null)
223  {
224  heightOffset += cell.LayoutCell
225  ? cell.Overlay.mergeConfig.markerHeightOffsetForLayoutTiles
226  : cell.Overlay.mergeConfig.markerHeightOffsetForNonLayoutTiles;
227  }
228 
229  var height = cell.Height;
230  var overlayPosition = basePosition + Vector3.Scale(new Vector3(x + 0.5f, height + heightOffset, y + 0.5f), gridSize);
231  var overlayMarkerTransform = Matrix4x4.TRS(overlayPosition, Quaternion.identity, Vector3.one);
232  EmitMarker(cell.Overlay.markerName, overlayMarkerTransform, new IntVector(x, 0, y), cellId);
233 
234  if (cell.Overlay.mergeConfig != null)
235  {
236  removeElevationMarker = cell.Overlay.mergeConfig.removeElevationMarker;
237  }
238  }
239 
240  switch (cell.CellType)
241  {
242  case FlowTilemapCellType.Floor:
243  EmitMarker(GridFlowDungeonConstants.MarkerGround, markerTransform, new IntVector(x, 0, y), cellId);
244  break;
245 
246  case FlowTilemapCellType.Wall:
247  EmitMarker(GridFlowDungeonConstants.MarkerWall, markerTransform, new IntVector(x, 0, y), cellId);
248  EmitMarker(GridFlowDungeonConstants.MarkerGround, markerTransform, new IntVector(x, 0, y), cellId);
249  break;
250 
251  case FlowTilemapCellType.Door:
252  {
253  var doorMarker = GridFlowDungeonConstants.MarkerDoor;
254  var doorData = cell.Userdata as FlowTilemapCellDoorInfo;
255  if (doorData != null && doorData.oneWay)
256  {
257  // One way door
258  doorMarker = GridFlowDungeonConstants.MarkerDoorOneWay;
259 
260  // Apply the correct one-way direction
261  var flipDirection = (doorData.nodeA.x > doorData.nodeB.x) || (doorData.nodeA.y > doorData.nodeB.y);
262  if (!flipDirection)
263  {
264  var doorRotation = baseRotation * Quaternion.Euler(0, 180, 0);
265  markerTransform = Matrix4x4.TRS(position, doorRotation, Vector3.one);
266  }
267  }
268 
269  FlowItemMetadata lockItemData = null;
270  if (cell.Item != System.Guid.Empty && itemMap.ContainsKey(cell.Item))
271  {
272  var item = itemMap[cell.Item];
273  CreateLockItemMetadata(item, ref doorMarker, out lockItemData);
274  }
275 
276  EmitMarker(doorMarker, markerTransform, new IntVector(x, 0, y), cellId, lockItemData);
277  EmitMarker(GridFlowDungeonConstants.MarkerGround, markerTransform, new IntVector(x, 0, y), cellId);
278  }
279  break;
280 
281  case FlowTilemapCellType.Custom:
282  if (cell.CustomCellInfo != null && !removeElevationMarker)
283  {
284  var markerName = cell.CustomCellInfo.name;
285  var height = cell.Height;
286  var customPosition = basePosition + Vector3.Scale(new Vector3(x + 0.5f, height, y + 0.5f), gridSize);
287  var customMarkerTransform = Matrix4x4.TRS(customPosition, Quaternion.identity, Vector3.one);
288 
289  EmitMarker(markerName, customMarkerTransform, new IntVector(x, 0, y), cellId);
290  }
291  break;
292 
293  }
294  }
295  }

297  // Emit the edge markers
298  {
299 
300  var walkableCellTypes = new FlowTilemapCellType[]
301  {
302  FlowTilemapCellType.Floor,
303  FlowTilemapCellType.Wall,
304  FlowTilemapCellType.Door
305  };
306 
307  var wallSeparators = new HashSet<IntVector2>();
308  var fenceSeparators = new HashSet<IntVector2>();
309 
310  foreach (var edge in tilemap.Edges)
311  {
312  var coord = edge.EdgeCoord;
313  var isGroundTile = IsCellOfType(tilemap, coord.x, coord.y, walkableCellTypes);
314 
315  if (edge.EdgeType == FlowTilemapEdgeType.Empty) continue;
316 
317  Vector3 position;
318  float angle;
319  if (edge.HorizontalEdge)
320  {
321  angle = isGroundTile ? 0 : 180;
322  position = basePosition + Vector3.Scale(new Vector3(coord.x + 0.5f, 0, coord.y), gridSize);
323  }
324  else
325  {
326  angle = isGroundTile ? 90 : 270;
327  position = basePosition + Vector3.Scale(new Vector3(coord.x, 0, coord.y + 0.5f), gridSize);
328  }
329 
330  var baseRotation = Quaternion.Euler(0, angle, 0);
331  var markerTransform = Matrix4x4.TRS(position, baseRotation, Vector3.one);
332 
333  bool supportedMarker = true;
334  var markerName = "";
335  FlowItemMetadata itemMetadata = null;
336  if (edge.EdgeType == FlowTilemapEdgeType.Wall)
337  {
338  markerName = GridFlowDungeonConstants.MarkerWall;
339 
340  wallSeparators.Add(coord);
341  if (edge.HorizontalEdge)
342  {
343  wallSeparators.Add(coord + new IntVector2(1, 0));
344  }
345  else
346  {
347  wallSeparators.Add(coord + new IntVector2(0, 1));
348  }
349  }
350  else if (edge.EdgeType == FlowTilemapEdgeType.Fence)
351  {
352  markerName = GridFlowDungeonConstants.MarkerFence;
353 
354  fenceSeparators.Add(coord);
355  if (edge.HorizontalEdge)
356  {
357  fenceSeparators.Add(coord + new IntVector2(1, 0));
358  }
359  else
360  {
361  fenceSeparators.Add(coord + new IntVector2(0, 1));
362  }
363  }
364  else if (edge.EdgeType == FlowTilemapEdgeType.Door)
365  {
366  markerName = GridFlowDungeonConstants.MarkerDoor;
367  var doorData = edge.Userdata as FlowTilemapCellDoorInfo;
368  if (doorData != null && doorData.oneWay)
369  {
370  // One way door
371  markerName = GridFlowDungeonConstants.MarkerDoorOneWay;
372 
373  // Apply the correct one-way direction
374  var flipDirection = (doorData.nodeA.x > doorData.nodeB.x) || (doorData.nodeA.y > doorData.nodeB.y);
375  if (!flipDirection)
376  {
377  var doorRotation = baseRotation * Quaternion.Euler(0, 180, 0);
378  markerTransform = Matrix4x4.TRS(position, doorRotation, Vector3.one);
379  }
380  }
381 
382  if (edge.Item != System.Guid.Empty && itemMap.ContainsKey(edge.Item))
383  {
384  var item = itemMap[edge.Item];
385  CreateLockItemMetadata(item, ref markerName, out itemMetadata);
386  }
387  }
388  else
389  {
390  supportedMarker = false;
391  }
392 
393  if (supportedMarker)
394  {
395  EmitMarker(markerName, markerTransform, new IntVector(coord.x, 0, coord.y), 0, itemMetadata);
396  }
397  }
398 
399  // Emit wall separator markers
400  foreach (var gridCoord in wallSeparators)
401  {
402  var position = basePosition + Vector3.Scale(new Vector3(gridCoord.x, 0, gridCoord.y), gridSize);
403  var markerTransform = Matrix4x4.TRS(position, Quaternion.identity, Vector3.one);
404  EmitMarker(GridFlowDungeonConstants.MarkerWallSeparator, markerTransform, new IntVector(gridCoord.x, 0, gridCoord.y), 0);
405  }
406 
407  // Emit fence separator markers
408  foreach (var gridCoord in fenceSeparators)
409  {
410  if (wallSeparators.Contains(gridCoord)) continue;
411 
412  var position = basePosition + Vector3.Scale(new Vector3(gridCoord.x, 0, gridCoord.y), gridSize);
413  var markerTransform = Matrix4x4.TRS(position, Quaternion.identity, Vector3.one);
414  EmitMarker(GridFlowDungeonConstants.MarkerFenceSeparator, markerTransform, new IntVector(gridCoord.x, 0, gridCoord.y), 0);
415  }
