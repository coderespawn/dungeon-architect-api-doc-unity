121  var visited = new bool[MazeConfig.mazeWidth, MazeConfig.mazeHeight];
122  for (int x = 0; x < w; x++)
123  {
124  for (int y = 0; y < h; y++)
125  {
126  //bool boundary = (x == 0 || y == 0 || x == w - 1 || y == h - 1);
127  visited[x, y] = false;

129  }
130 
131  var stack = new Stack<IntVector2>();
132  {
133  var startPoint = new IntVector2(0, 2 + random.Next() % (MazeConfig.mazeHeight - 4));
134  visited[startPoint.x, startPoint.y] = true;
135  stack.Push(startPoint);
136  }
137 
138  while (stack.Count > 0)
139  {
140  var currentPoint = stack.Peek();
141  IntVector2 nextPoint;
142  if (GetNextNeighbor(currentPoint, out nextPoint, visited))
143  {
144  visited[nextPoint.x, nextPoint.y] = true;
145  stack.Push(nextPoint);
146  }
147  else
148  {
149  stack.Pop();
150  }
151  }

156  {
157  for (int y = 0; y < h; y++)
