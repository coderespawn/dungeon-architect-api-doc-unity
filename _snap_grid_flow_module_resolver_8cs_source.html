23  private static int Solve(string[] markerNames, Dictionary<PlaceableMarker, int> availableMarkers) {
24  int numFailed;
25  if (availableMarkers.Count > 0) {
26  numFailed = SolveImpl(markerNames, 0, availableMarkers);
27  Debug.Assert(numFailed >= 0);
28  }
29  else
30  {
31  numFailed = markerNames.Length;
32  }
33 
34  const int FAIL_WEIGHT = 1000000;
35  return numFailed * FAIL_WEIGHT;
36  }

38  // Returns the no. of items failed in the processed sub tree
39  private static int SolveImpl(string[] markerNames, int index, Dictionary<PlaceableMarker, int> availableMarkers) {
40  if (index == markerNames.Length) {
41  return 0;
42  }
43 
44  Debug.Assert(index >= 0 || index < markerNames.Length);
45 
46  int bestFrameFailCount = markerNames.Length;
47  var markerName = markerNames[index];
48  var keys = availableMarkers.Keys.ToArray();
49  foreach (var key in keys) {
50  var availableMarkerAsset = key;
51  int count = availableMarkers[key];
52 
53  bool canAttachHere = count > 0 && availableMarkerAsset.supportedMarkers.Contains(markerName);
54  int frameFailCount = canAttachHere ? 0 : 1;
55  if (canAttachHere) {
56  count--;
57  }
58  frameFailCount += SolveImpl(markerNames, index + 1, availableMarkers);
59  if (canAttachHere) {
60  count++;
61  }
62 
63  availableMarkers[availableMarkerAsset] = count;
64 
65  if (frameFailCount < bestFrameFailCount) {
66  bestFrameFailCount = frameFailCount;
67  }
68 
69  if (availableMarkerAsset.supportedMarkers.Length == 1 && availableMarkerAsset.supportedMarkers[0] == markerName) {
70  // Faster bailout
71  break;
72  }
73  }
74 
75  return bestFrameFailCount;
76  }

151  if (markerName.Length > 0) {
152  desiredNodeMarkers.Add(markerName);
153  }

169  foreach (var asmIdx in shuffledAsmIndices) {
170  var moduleAssembly = moduleInfo.RotatedAssemblies[asmIdx];
171  SgfModuleAssemblySideCell[] doorIndices;

176

178  candidate.ModuleRotation = Quaternion.AngleAxis(asmIdx * -90, Vector3.up);
179  candidate.AssemblyIndex = asmIdx;
180  candidate.DoorIndices = doorIndices;
181 
182  var gap = 1000;
183  float connectionWeight = bChooseModulesWithMinDoors ? (moduleInfo.Connections.Length * gap) : 0;
184  float moduleWeight = random.Range(1 - moduleEntryWeight, 1.0f) * (gap - 1);
185 
186  candidate.Priority = Mathf.RoundToInt(itemFitness + connectionWeight + moduleWeight);
187  }
188  }
189  }

193  return false;
194  }

198  {

201  for (int idx = 0; idx < fitCandidates.Count; idx++) {

203  if (bestFitPriority > candidate.Priority) {
204  bestFitIndices.Clear();
205  bestFitIndices.Add(idx);
206  bestFitPriority = candidate.Priority;
207  }
208  else if (bestFitPriority == candidate.Priority) {
209  bestFitIndices.Add(idx);
210  }
211  }

303  bool foundSrcCell = false;

305

310  foreach (var sourceDoorCell in sourceDoorCells)
311  {
312  if (sourceDoorCell.linkId == graphLink.linkId)
313  {
314  srcCell = sourceDoorCell;
315  foundSrcCell = true;
316  break;
317  }
318  }
319  }
320

322  {
323  var destDoorCells = activeModuleDoorIndices[destId];
324  foreach (var destDoorCell in destDoorCells)
325  {
326  if (destDoorCell.linkId == graphLink.linkId)
327  {
328  dstCell = destDoorCell;
329  foundDstCell = true;
330  break;
331  }
332  }
333  }
334  }
335

337  outModuleNodes = new SgfModuleNode[]{};
338  return false;
339  }
340 
341  if (!moduleNodesById.ContainsKey(sourceId) || !moduleNodesById.ContainsKey(destId)) {
342  outModuleNodes = new SgfModuleNode[]{};
343  return false;
344  }
345 
346  var srcModule = moduleNodesById[sourceId];
347  var dstModule = moduleNodesById[destId];
348  var srcDoor = srcModule.Doors[srcCell.connectionIdx];
349  var dstDoor = dstModule.Doors[dstCell.connectionIdx];
