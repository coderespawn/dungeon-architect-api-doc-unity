179  public static BBox3 GetBoundingBox(this IEnumerable<Triangle3> tris, float padding)
180  {
181  BBox3 bounds = new BBox3();
182  Vector3 va, vb, vc;
183  foreach (Triangle3 tri in tris)
184  {
185  va = tri.A;
186  vb = tri.B;
187  vc = tri.C;
188  ApplyVertexToBounds(ref va, ref bounds);
189  ApplyVertexToBounds(ref vb, ref bounds);
190  ApplyVertexToBounds(ref vc, ref bounds);
191  }
192 
193  //pad the bounding box a bit to make sure outer triangles are fully contained.
194  ApplyPaddingToBounds(padding, ref bounds);
195 
196  return bounds;
197  }

224  private static void ApplyVertexToBounds(ref Vector3 v, ref BBox3 b)
225  {
226  if (v.X < b.Min.X)
227  b.Min.X = v.X;
228  if (v.Y < b.Min.Y)
229  b.Min.Y = v.Y;
230  if (v.Z < b.Min.Z)
231  b.Min.Z = v.Z;
232  if (v.X > b.Max.X)
233  b.Max.X = v.X;
234  if (v.Y > b.Max.Y)
235  b.Max.Y = v.Y;
236  if (v.Z > b.Max.Z)
237  b.Max.Z = v.Z;
238  }
