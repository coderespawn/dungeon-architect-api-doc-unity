231 
232 
233  if (node != null)
234  {
235  // Check if the menu was created by dragging out a link
236  if (e.sourcePin != null)
237  {
238  GraphPin targetPin =
239  e.sourcePin.PinType == GraphPinType.Input ?
240  node.OutputPins[0] :
241  node.InputPins[0];
242 
243  // Align the target pin with the mouse position where the link was dragged and released
244  node.Position = e.mouseWorldPosition - targetPin.Position;
245 
246  GraphPin inputPin, outputPin;

286  void UpdateSpatialConstraintEditorWindow(GraphNode[] nodes)
287  {
288  SpatialConstraintAsset spatialAsset = null;
289  SpatialConstraintsEditorAssignmentState assetState;
290  VisualNode node = null;
291  if (nodes.Length == 1 && nodes[0] is VisualNode)
292  {
293  node = nodes[0] as VisualNode;
294  }
295 
296  if (node != null)
297  {
298  if (node.spatialConstraint == null)
299  {
300  Debug.LogError("Spatial Constraint asset not assigned. Please reload the theme editor");
301  return;
302  }
303 
304  if (node.useSpatialConstraint)
305  {
306  spatialAsset = node.spatialConstraint;
307  assetState = SpatialConstraintsEditorAssignmentState.Assigned;
308  }
309  else
310  {
311  assetState = SpatialConstraintsEditorAssignmentState.ConstraintsDisabled;
312  }
313  }
314  else
315  {
316  assetState = SpatialConstraintsEditorAssignmentState.NotAssigned;
317  }

319  SetSpatialConstraintEditorAsset(spatialAsset, assetState);

489  class NodeDeletionOrderComparer : IComparer<GraphNode>
490  {
491  int GetWeight(GraphNode node)
492  {
493  if (node is MarkerEmitterNode) return 0;
494  if (node is VisualNode) return 1;
495  if (node is MarkerNode) return 2;
496  return 3;
497  }
498 
499  public int Compare(GraphNode x, GraphNode y)
500  {
501  int wx = GetWeight(x);
502  int wy = GetWeight(y);
503  if (wx == wy) return 0;
504  return (wx < wy) ? -1 : 1;
505  }
506  }

511  class GraphPinHierarchyComparer : IComparer<GraphPin>
512  {
513  int GetWeight(GraphNode node)
514  {
515  if (node is MarkerNode) return 1;
516  else if (node is VisualNode) return 2;
517  else if (node is MarkerEmitterNode) return 3;
518  else return 4;
519  }
520 
521  public int Compare(GraphPin x, GraphPin y)
522  {
523  if (x == null || y == null) return 0;
524  var wx = GetWeight(x.Node);
525  var wy = GetWeight(y.Node);
526 
527  if (wx == wy) return 0;
528  return wx < wy ? -1 : 1;
529  }
