239  internal static int ClipPolygonToPlane(Vector3[] inVertices, Vector3[] outVertices, float[] distances, int numVerts, float planeX, float planeZ, float planeD)
240  {
241  for (int i = 0; i < numVerts; i++)
242  distances[i] = planeX * inVertices[i].X + planeZ * inVertices[i].Z + planeD;
243 
244  int m = 0;
245  Vector3 temp;
246  for (int i = 0, j = numVerts - 1; i < numVerts; j = i, i++)
247  {
248  bool inj = distances[j] >= 0;
249  bool ini = distances[i] >= 0;
250 
251  if (inj != ini)
252  {
253  float s = distances[j] / (distances[j] - distances[i]);
254 
255  Vector3.Subtract(ref inVertices[i], ref inVertices[j], out temp);
256  Vector3.Multiply(ref temp, s, out temp);

259  }
