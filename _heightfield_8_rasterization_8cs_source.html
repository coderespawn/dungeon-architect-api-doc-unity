837  public void RasterizeTriangle(ref Vector3 a, ref Vector3 b, ref Vector3 c, Area area)
838  {
839  //distances buffer for ClipPolygonToBounds
840  float[] distances = new float[12];
841 
842  float invCellSize = 1f / cellSize;
843  float invCellHeight = 1f / cellHeight;
844  float boundHeight = bounds.Max.Y - bounds.Min.Y;
845 
846  //calculate the triangle's bounding box
847  BBox3 bbox;
848  Triangle3.GetBoundingBox(ref a, ref b, ref c, out bbox);
849 
850  //make sure that the triangle is at least in one cell.
851  if (!BBox3.Overlapping(ref bbox, ref bounds))
852  return;
853 
854  //figure out which rows.
855  int z0 = (int)((bbox.Min.Z - bounds.Min.Z) * invCellSize);
856  int z1 = (int)((bbox.Max.Z - bounds.Min.Z) * invCellSize);
857 
858  //clamp to the field boundaries.
859  MathHelper.Clamp(ref z0, 0, length - 1);
860  MathHelper.Clamp(ref z1, 0, length - 1);
861 
862  Vector3[] inVerts = new Vector3[7], outVerts = new Vector3[7], inRowVerts = new Vector3[7];
863 
864  for (int z = z0; z <= z1; z++)
865  {
866  //copy the original vertices to the array.
867  inVerts[0] = a;
868  inVerts[1] = b;
869  inVerts[2] = c;
870 
871  //clip the triangle to the row
872  int nvrow = 3;
873  float cz = bounds.Min.Z + z * cellSize;
874  nvrow = MathHelper.ClipPolygonToPlane(inVerts, outVerts, distances, nvrow, 0, 1, -cz);
875  if (nvrow < 3)
876  continue;
877  nvrow = MathHelper.ClipPolygonToPlane(outVerts, inRowVerts, distances, nvrow, 0, -1, cz + cellSize);
878  if (nvrow < 3)
879  continue;
880 
881  float minX = inRowVerts[0].X, maxX = minX;
882  for (int i = 1; i < nvrow; i++)
883  {
884  float vx = inRowVerts[i].X;
885  if (minX > vx)
886  minX = vx;
887  if (maxX < vx)
888  maxX = vx;
889  }
890 
891  int x0 = (int)((minX - bounds.Min.X) * invCellSize);
892  int x1 = (int)((maxX - bounds.Min.X) * invCellSize);
893 
894  MathHelper.Clamp(ref x0, 0, width - 1);
895  MathHelper.Clamp(ref x1, 0, width - 1);
896 
897  for (int x = x0; x <= x1; x++)
898  {
899  //clip the triangle to the column
900  int nv = nvrow;
