172  public override void Tick()
173  {
174  if (buildQueue == null || buildQueue.Count == 0)
175  {
176  return;

178 
179  var stopwatch = System.Diagnostics.Stopwatch.StartNew();
180  while (buildQueue.Count > 0 && stopwatch.ElapsedMilliseconds < maxMilliPerFrame)
181  {

187  spawnList.Add(spawnInfo);

193
