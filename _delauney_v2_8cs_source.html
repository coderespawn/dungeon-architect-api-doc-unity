179  static bool CircumCircle(float xp, float yp, float x1, float y1, float x2,
180  float y2, float x3, float y3, ref float xc, ref float yc, ref float r)
181  {
182  float m1, m2, mx1, mx2, my1, my2;
183  float dx, dy, rsqr, drsqr;
184 
185  /* Check for coincident points */
186  if (Mathf.Abs(y1 - y2) < EPSILON && Mathf.Abs(y2 - y3) < EPSILON)
187  {
188  return false;
189  }
190 
191  if (Mathf.Abs(y2 - y1) < EPSILON)
192  {
193  m2 = -(x3 - x2) / (y3 - y2);
194  mx2 = (x2 + x3) / 2.0f;
195  my2 = (y2 + y3) / 2.0f;
196  xc = (x2 + x1) / 2.0f;
197  yc = m2 * (xc - mx2) + my2;
198  }
199  else if (Mathf.Abs(y3 - y2) < EPSILON)
200  {
201  m1 = -(x2 - x1) / (y2 - y1);
202  mx1 = (x1 + x2) / 2.0f;
203  my1 = (y1 + y2) / 2.0f;
204  xc = (x3 + x2) / 2.0f;
205  yc = m1 * (xc - mx1) + my1;
206  }
207  else
208  {
209  m1 = -(x2 - x1) / (y2 - y1);
210  m2 = -(x3 - x2) / (y3 - y2);
211  mx1 = (x1 + x2) / 2.0f;
212  mx2 = (x2 + x3) / 2.0f;
213  my1 = (y1 + y2) / 2.0f;
214  my2 = (y2 + y3) / 2.0f;
215  xc = (m1 * mx1 - m2 * mx2 + my2 - my1) / (m1 - m2);
216  yc = m1 * (xc - mx1) + my1;
217  }
218  dx = x2 - xc;
219  dy = y2 - yc;
220  rsqr = dx * dx + dy * dy;
221  r = Mathf.Sqrt(rsqr);
222  dx = xp - xc;
223  dy = yp - yc;
224  drsqr = dx * dx + dy * dy;
225  return (drsqr <= rsqr) ? true : false;
226  }
