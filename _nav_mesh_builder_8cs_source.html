45  public NavMeshBuilder(PolyMesh polyMesh, PolyMeshDetail polyMeshDetail, OffMeshConnection[] offMeshCons, NavMeshGenerationSettings settings)
46  {
47  if (settings.VertsPerPoly > PathfindingCommon.VERTS_PER_POLYGON)
48  throw new InvalidOperationException("The number of vertices per polygon is above SharpNav's limit");
49  if (polyMesh.VertCount == 0)
50  throw new InvalidOperationException("The provided PolyMesh has no vertices.");
51  if (polyMesh.PolyCount == 0)
52  throw new InvalidOperationException("The provided PolyMesh has not polys.");
53 
54  int nvp = settings.VertsPerPoly;
55 
56  //classify off-mesh connection points
57  BoundarySide[] offMeshSides = new BoundarySide[offMeshCons.Length * 2];
58  int storedOffMeshConCount = 0;
59  int offMeshConLinkCount = 0;
60 
61  if (offMeshCons.Length > 0)
62  {
63  //find height bounds
64  float hmin = float.MaxValue;
65  float hmax = -float.MaxValue;
66 
67  if (polyMeshDetail != null)
68  {
69  for (int i = 0; i < polyMeshDetail.VertCount; i++)
70  {
71  float h = polyMeshDetail.Verts[i].Y;
72  hmin = Math.Min(hmin, h);
73  hmax = Math.Max(hmax, h);
74  }
75  }
76  else
77  {
78  for (int i = 0; i < polyMesh.VertCount; i++)
79  {
80  PolyVertex iv = polyMesh.Verts[i];
81  float h = polyMesh.Bounds.Min.Y + iv.Y * settings.CellHeight;
82  hmin = Math.Min(hmin, h);
83  hmax = Math.Max(hmax, h);
84  }
85  }
86 
87  hmin -= settings.MaxClimb;
88  hmax += settings.MaxClimb;
89  BBox3 bounds = polyMesh.Bounds;
90  bounds.Min.Y = hmin;
91  bounds.Max.Y = hmax;
92 
93  for (int i = 0; i < offMeshCons.Length; i++)
94  {
95  Vector3 p0 = offMeshCons[i].Pos0;
96  Vector3 p1 = offMeshCons[i].Pos1;
97 
98  offMeshSides[i * 2 + 0] = BoundarySideExtensions.FromPoint(p0, bounds);
99  offMeshSides[i * 2 + 1] = BoundarySideExtensions.FromPoint(p1, bounds);
100 
101  //off-mesh start position isn't touching mesh
102  if (offMeshSides[i * 2 + 0] == BoundarySide.Internal)
103  {
104  if (p0.Y < bounds.Min.Y || p0.Y > bounds.Max.Y)
105  offMeshSides[i * 2 + 0] = 0;
106  }
107 
108  //count number of links to allocate
109  if (offMeshSides[i * 2 + 0] == BoundarySide.Internal)
110  offMeshConLinkCount++;
111  if (offMeshSides[i * 2 + 1] == BoundarySide.Internal)
112  offMeshConLinkCount++;
113 
114  if (offMeshSides[i * 2 + 0] == BoundarySide.Internal)
115  storedOffMeshConCount++;
116  }
117  }
118 
119  //off-mesh connections stored as polygons, adjust values
120  int totPolyCount = polyMesh.PolyCount + storedOffMeshConCount;
121  int totVertCount = polyMesh.VertCount + storedOffMeshConCount * 2;
122 
123  //find portal edges
124  int edgeCount = 0;
125  int portalCount = 0;
126  for (int i = 0; i < polyMesh.PolyCount; i++)
127  {
128  PolyMesh.Polygon p = polyMesh.Polys[i];
129  for (int j = 0; j < nvp; j++)
130  {
131  if (p.Vertices[j] == PolyMesh.NullId)
132  break;
133 
134  edgeCount++;
135 
136  if (PolyMesh.IsBoundaryEdge(p.NeighborEdges[j]))
137  {
138  int dir = p.NeighborEdges[j] % 16;
139  if (dir != 15)
140  portalCount++;
141  }
142  }
143  }
144 
145  int maxLinkCount = edgeCount + portalCount * 2 + offMeshConLinkCount * 2;
146 
147  //find unique detail vertices
148  int uniqueDetailVertCount = 0;
149  int detailTriCount = 0;
150  if (polyMeshDetail != null)
151  {
152  detailTriCount = polyMeshDetail.TrisCount;
153  for (int i = 0; i < polyMesh.PolyCount; i++)
154  {
155  int numDetailVerts = polyMeshDetail.Meshes[i].VertexCount;
156  int numPolyVerts = polyMesh.Polys[i].VertexCount;
157  uniqueDetailVertCount += numDetailVerts - numPolyVerts;
158  }
159  }
160  else
161  {
162  uniqueDetailVertCount = 0;
163  detailTriCount = 0;
164  for (int i = 0; i < polyMesh.PolyCount; i++)
165  {
166  int numPolyVerts = polyMesh.Polys[i].VertexCount;
167  uniqueDetailVertCount += numPolyVerts - 2;
168  }
169  }
170 
171  //allocate data
172  header = new PathfindingCommon.NavMeshInfo();
173  navVerts = new Vector3[totVertCount];
174  navPolys = new Poly[totPolyCount];
175  navDMeshes = new PolyMeshDetail.MeshData[polyMesh.PolyCount];
176  navDVerts = new Vector3[uniqueDetailVertCount];
177  navDTris = new PolyMeshDetail.TriangleData[detailTriCount];
178  offMeshConnections = new OffMeshConnection[storedOffMeshConCount];
179 
180  //store header
181  //HACK TiledNavMesh should figure out the X/Y/layer instead of the user maybe?
182  header.X = 0;
183  header.Y = 0;
184  header.Layer = 0;
185  header.PolyCount = totPolyCount;
186  header.VertCount = totVertCount;
187  header.MaxLinkCount = maxLinkCount;
188  header.Bounds = polyMesh.Bounds;
189  header.DetailMeshCount = polyMesh.PolyCount;
190  header.DetailVertCount = uniqueDetailVertCount;
191  header.DetailTriCount = detailTriCount;
192  header.OffMeshBase = polyMesh.PolyCount;
193  header.WalkableHeight = settings.AgentHeight;
194  header.WalkableRadius = settings.AgentRadius;
195  header.WalkableClimb = settings.MaxClimb;
196  header.OffMeshConCount = storedOffMeshConCount;
197  header.BvNodeCount = settings.BuildBoundingVolumeTree ? polyMesh.PolyCount * 2 : 0;
198  header.BvQuantFactor = 1f / settings.CellSize;
199 
200  int offMeshVertsBase = polyMesh.VertCount;
201  int offMeshPolyBase = polyMesh.PolyCount;
202 
203  //store vertices
204  for (int i = 0; i < polyMesh.VertCount; i++)
205  {
206  PolyVertex iv = polyMesh.Verts[i];
207  navVerts[i].X = polyMesh.Bounds.Min.X + iv.X * settings.CellSize;
208  navVerts[i].Y = polyMesh.Bounds.Min.Y + iv.Y * settings.CellHeight;
209  navVerts[i].Z = polyMesh.Bounds.Min.Z + iv.Z * settings.CellSize;
210  }
211 
212  //off-mesh link vertices
213  int n = 0;
214  for (int i = 0; i < offMeshCons.Length; i++)
215  {
216  //only store connections which start from this tile
217  if (offMeshSides[i * 2 + 0] == BoundarySide.Internal)
218  {
219  navVerts[offMeshVertsBase + (n * 2 + 0)] = offMeshCons[i].Pos0;
220  navVerts[offMeshVertsBase + (n * 2 + 1)] = offMeshCons[i].Pos1;
221  n++;
222  }
223  }
224 
225  //store polygons
226  for (int i = 0; i < polyMesh.PolyCount; i++)
227  {
228  navPolys[i] = new Poly();
229  navPolys[i].VertCount = 0;
230  navPolys[i].Flags = polyMesh.Polys[i].Flags;
231  navPolys[i].Area = polyMesh.Polys[i].Area;
232  navPolys[i].PolyType = PolygonType.Ground;
233  navPolys[i].Verts = new int[nvp];
234  navPolys[i].Neis = new int[nvp];
235  for (int j = 0; j < nvp; j++)
236  {
237  if (polyMesh.Polys[i].Vertices[j] == PolyMesh.NullId)
238  break;
239 
240  navPolys[i].Verts[j] = polyMesh.Polys[i].Vertices[j];
241  if (PolyMesh.IsBoundaryEdge(polyMesh.Polys[i].NeighborEdges[j]))
242  {
243  //border or portal edge
244  int dir = polyMesh.Polys[i].NeighborEdges[j] % 16;
245  if (dir == 0xf) //border
246  navPolys[i].Neis[j] = 0;
247  else if (dir == 0) //portal x-
248  navPolys[i].Neis[j] = Link.External | 4;
249  else if (dir == 1) //portal z+
250  navPolys[i].Neis[j] = Link.External | 2;
251  else if (dir == 2) //portal x+
252  navPolys[i].Neis[j] = Link.External | 0;
253  else if (dir == 3) //portal z-
254  navPolys[i].Neis[j] = Link.External | 6;
255  }
256  else
257  {
258  //normal connection
259  navPolys[i].Neis[j] = polyMesh.Polys[i].NeighborEdges[j] + 1;
260  }
261 
262  navPolys[i].VertCount++;
263  }
264  }
265 
266  //off-mesh connection vertices
267  n = 0;
268  for (int i = 0; i < offMeshCons.Length; i++)
269  {
270  //only store connections which start from this tile
271  if (offMeshSides[i * 2 + 0] == BoundarySide.Internal)
272  {
273  navPolys[offMeshPolyBase + n].VertCount = 2;
274  navPolys[offMeshPolyBase + n].Verts = new int[nvp];
275  navPolys[offMeshPolyBase + n].Verts[0] = offMeshVertsBase + (n * 2 + 0);
276  navPolys[offMeshPolyBase + n].Verts[1] = offMeshVertsBase + (n * 2 + 1);
277  navPolys[offMeshPolyBase + n].Flags = (int)offMeshCons[i].Flags;
278  navPolys[offMeshPolyBase + n].Area = polyMesh.Polys[offMeshCons[i].Poly].Area; //HACK is this correct?
279  navPolys[offMeshPolyBase + n].PolyType = PolygonType.OffMeshConnection;
280  n++;
281  }
282  }
283 
284  //store detail meshes and vertices
285  if (polyMeshDetail != null)
286  {
287  int vbase = 0;
288  List<Vector3> storedDetailVerts = new List<Vector3>();
289  for (int i = 0; i < polyMesh.PolyCount; i++)
290  {
291  int vb = polyMeshDetail.Meshes[i].VertexIndex;
292  int numDetailVerts = polyMeshDetail.Meshes[i].VertexCount;
293  int numPolyVerts = navPolys[i].VertCount;
294  navDMeshes[i].VertexIndex = vbase;
295  navDMeshes[i].VertexCount = numDetailVerts - numPolyVerts;
296  navDMeshes[i].TriangleIndex = polyMeshDetail.Meshes[i].TriangleIndex;
297  navDMeshes[i].TriangleCount = polyMeshDetail.Meshes[i].TriangleCount;
298 
299  //Copy detail vertices
300  //first 'nv' verts are equal to nav poly verts
301  //the rest are detail verts
302  for (int j = 0; j < navDMeshes[i].VertexCount; j++)
303  {
304  storedDetailVerts.Add(polyMeshDetail.Verts[vb + numPolyVerts + j]);
305  }
306 
307  vbase += numDetailVerts - numPolyVerts;
308  }
309 
310  navDVerts = storedDetailVerts.ToArray();
311 
312  //store triangles
313  for (int j = 0; j < polyMeshDetail.TrisCount; j++)
314  navDTris[j] = polyMeshDetail.Tris[j];
315  }
316  else
317  {
318  //create dummy detail mesh by triangulating polys
319  int tbase = 0;
320  for (int i = 0; i < polyMesh.PolyCount; i++)
321  {
322  int numPolyVerts = navPolys[i].VertCount;
323  navDMeshes[i].VertexIndex = 0;
324  navDMeshes[i].VertexCount = 0;
325  navDMeshes[i].TriangleIndex = tbase;
326  navDMeshes[i].TriangleCount = numPolyVerts - 2;
327 
328  //triangulate polygon
329  for (int j = 2; j < numPolyVerts; j++)
330  {
331  navDTris[tbase].VertexHash0 = 0;
332  navDTris[tbase].VertexHash1 = j - 1;
333  navDTris[tbase].VertexHash2 = j;
334 
335  //bit for each edge that belongs to the poly boundary
336  navDTris[tbase].Flags = 1 << 2;
337  if (j == 2)
338  navDTris[tbase].Flags |= 1 << 0;
339  if (j == numPolyVerts - 1)
340  navDTris[tbase].Flags |= 1 << 4;
341 
342  tbase++;
343  }
344  }
345  }
346 
347  //store and create BV tree
348  if (settings.BuildBoundingVolumeTree)
349  {
350  //build tree
351  navBvTree = new BVTree(polyMesh.Verts, polyMesh.Polys, nvp, settings.CellSize, settings.CellHeight);
352  }
353 
354  //store off-mesh connections
355  n = 0;
356  for (int i = 0; i < offMeshConnections.Length; i++)
357  {
358  //only store connections which start from this tile
359  if (offMeshSides[i * 2 + 0] == BoundarySide.Internal)
360  {
361  offMeshConnections[n].Poly = offMeshPolyBase + n;
362 
363  //copy connection end points
364  offMeshConnections[n].Pos0 = offMeshCons[i].Pos0;
365  offMeshConnections[n].Pos1 = offMeshCons[i].Pos1;
366 
367  offMeshConnections[n].Radius = offMeshCons[i].Radius;
368  offMeshConnections[n].Flags = offMeshCons[i].Flags;
369  offMeshConnections[n].Side = offMeshSides[i * 2 + 1];
370  offMeshConnections[n].Tag = offMeshCons[i].Tag;
371 
372  n++;
373  }
374  }
375  }
