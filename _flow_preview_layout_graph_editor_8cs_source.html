204  void DrawReferenceLink(UIRenderer renderer, Rect start, Rect end)
205  {
206  var centerA = start.center;
207  var centerB = end.center;
208  var radiusA = Mathf.Max(start.size.x, start.size.y) * 0.5f;
209  var radiusB = Mathf.Max(end.size.x, end.size.y) * 0.5f;
210  var vecAtoB = (centerB - centerA);
211  var lenAtoB = vecAtoB.magnitude;
212  var dirAtoB = vecAtoB / lenAtoB;
213 
214  var startPos = centerA + dirAtoB * radiusA;
215  var endPos = centerA + dirAtoB * (lenAtoB - radiusB);
216  StraightLineGraphLinkRenderer.DrawLine(renderer, startPos, endPos, camera, Color.red, 2);
217  }
