600  public bool MergeWithRegion(Region otherRegion)
601  {
602  RegionId thisId = id;
603  RegionId otherId = otherRegion.id;
604 
605  // Duplicate current neighbourhood.
606  List<RegionId> thisConnected = new List<RegionId>();
607  for (int i = 0; i < connections.Count; ++i)
608  thisConnected.Add(connections[i]);
609  List<RegionId> otherConnected = otherRegion.connections;
610 
611  // Find insertion point on this region
612  int insertInThis = -1;
613  for (int i = 0; i < thisConnected.Count; ++i)
614  {
615  if (thisConnected[i] == otherId)
616  {
617  insertInThis = i;
618  break;
619  }
620  }
621 
622  if (insertInThis == -1)
623  return false;
624 
625  // Find insertion point on the other region
626  int insertInOther = -1;
627  for (int i = 0; i < otherConnected.Count; ++i)
628  {
629  if (otherConnected[i] == thisId)
630  {
631  insertInOther = i;
632  break;
633  }
634  }
635 
636  if (insertInOther == -1)
637  return false;
638 
639  // Merge neighbours.
640  connections = new List<RegionId>();
641  for (int i = 0, ni = thisConnected.Count; i < ni - 1; ++i)
642  connections.Add(thisConnected[(insertInThis + 1 + i) % ni]);
643 
644  for (int i = 0, ni = otherConnected.Count; i < ni - 1; ++i)
645  connections.Add(otherConnected[(insertInOther + 1 + i) % ni]);
646 
647  RemoveAdjacentNeighbours();
648 
649  for (int j = 0; j < otherRegion.floors.Count; ++j)
650  AddUniqueFloorRegion(otherRegion.floors[j]);
651  spanCount += otherRegion.spanCount;
652  otherRegion.spanCount = 0;
653  otherRegion.connections.Clear();
654 
655  return true;
656  }
