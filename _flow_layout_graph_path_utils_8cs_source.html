80  public static void FinalizePath(FlowLayoutStaticGrowthState staticState, FlowLayoutGrowthState state)
81  {
82  var path = state.Path;
83 
84  if (path.Count == 0) {
85  return;
86  }
87 
88  // Create merged node groups
89  foreach (var groupInfo in state.NodeGroups) {
90  CreateMergedCompositeNode(staticState.GraphQuery, groupInfo);
91  }
92 
93  FlowLayoutGraph graph = staticState.GraphQuery.GetGraph();
94  var childToParentMap = new Dictionary<DungeonUID, DungeonUID>(); // [ChildNodeId -> ParentNodeId]
95  foreach (var parentNode in graph.Nodes) {
96  if (parentNode.MergedCompositeNodes.Count > 1) {
97  foreach (var childNode in parentNode.MergedCompositeNodes) {
98  childToParentMap[childNode.nodeId] = parentNode.nodeId;
99  }
100  }
101  }
102 
103  var pathLength = path.Count;
104  for (int i = 0; i < pathLength; i++) {
105  var pathItem = path[i];
106  var origNodeId = pathItem.NodeId;
107  var origPrevNodeId = pathItem.PreviousNodeId;
108  if (childToParentMap.ContainsKey(pathItem.NodeId))
109  {
110  pathItem.NodeId = childToParentMap[pathItem.NodeId];
111  }
112 
113  if (childToParentMap.ContainsKey(pathItem.PreviousNodeId))
114  {
115  pathItem.PreviousNodeId = childToParentMap[pathItem.PreviousNodeId];
116  }
117 
118  FlowLayoutGraphNode pathNode = staticState.GraphQuery.GetNode(pathItem.NodeId);
119  if (pathNode == null) continue;
120  pathNode.active = true;
121  pathNode.color = staticState.NodeColor;
122  pathNode.pathIndex = i;
123  pathNode.pathLength = pathLength;
124 
125  string pathName;
126  if (i == 0 && staticState.StartNodePathNameOverride.Length > 0) {
127  pathName = staticState.StartNodePathNameOverride;
128  }
129  else if (i == path.Count - 1 && staticState.EndNodePathNameOverride.Length > 0) {
130  pathName = staticState.EndNodePathNameOverride;
131  }
132  else {
133  pathName = staticState.PathName;
134  }
135  pathNode.pathName = pathName;
136 
137 
138  // Link the path nodes
139  if (i > 0) {
140  var linkSrc = pathItem.PreviousNodeId;
141  var linkDst = pathItem.NodeId;
142  var linkSrcSub = origPrevNodeId;
143  var linkDstSub = origNodeId;
144 
145  var possibleLinks = staticState.Graph.GetLinks(linkSrc, linkDst, true);
146  foreach (var possibleLink in possibleLinks) {
147  if (possibleLink == null) continue;
148  if (possibleLink.source == linkSrc && possibleLink.destination == linkDst) {
149  bool bValid = (!possibleLink.sourceSubNode.IsValid() || possibleLink.sourceSubNode == linkSrcSub);
150  bValid &= (!possibleLink.destinationSubNode.IsValid() || possibleLink.destinationSubNode == linkDstSub);
151 
152  // Found the correct link
153  if (bValid) {
154  possibleLink.state.type = FlowLayoutGraphLinkType.Connected;
155  break;
156  }
157  }
158  else if (possibleLink.source == linkDst && possibleLink.destination == linkSrc) {
159  bool bValid = (!possibleLink.sourceSubNode.IsValid() || possibleLink.sourceSubNode == linkDstSub);
160  bValid &= (!possibleLink.destinationSubNode.IsValid() || possibleLink.destinationSubNode == linkSrcSub);
161 
162  // Found the correct link
163  if (bValid) {
164  possibleLink.state.type = FlowLayoutGraphLinkType.Connected;
165  possibleLink.ReverseDirection();
166  break;
167  }
168  }
169  }
170  }
171  }
172 
173  // Setup the start / end links
174  if (staticState.HeadNode != null) {
175  var linkSrc = staticState.HeadNode.nodeId;
176  var linkDst = path[0].NodeId;
177  var headLink = staticState.Graph.GetLink(linkSrc, linkDst, true);
178  if (headLink != null) {
179  headLink.state.type = FlowLayoutGraphLinkType.Connected;
180  if (headLink.source == linkDst && headLink.destination == linkSrc) {
181  headLink.ReverseDirection();
182  }
183  }
184  }
185 
186  // Find the end node, if any so that it can merge back to the specified branch (specified in variable EndOnPath)
187  if (state.TailNode != null) {
188  var linkSrc = path[path.Count - 1].NodeId;
189  var linkDst = state.TailNode.nodeId;
190  var tailLink = staticState.Graph.GetLink(linkSrc, linkDst, true);
191  if (tailLink != null) {
192  tailLink.state.type = FlowLayoutGraphLinkType.Connected;
193  if (tailLink.source == linkDst && tailLink.destination == linkSrc) {
194  tailLink.ReverseDirection();
195  }
196  }
197  }

200  static FlowLayoutGraphNode CreateMergedCompositeNode(FlowLayoutGraphQuery graphQuery, FlowLayoutGraphNodeGroup nodeGroup)
201  {
202  if (nodeGroup.GroupNodes.Count <= 1) {
203  return null;
204  }
205 
206  var graph = graphQuery.GetGraph();
207  var subNodes = new HashSet<FlowLayoutGraphNode>();
208  var subNodeIds = new HashSet<DungeonUID>();
209  var subItems = new HashSet<FlowItem>();
210 
211  var previewLocation = Vector3.zero;
212  var coord = Vector3.zero;
213  foreach (var subNodeId in nodeGroup.GroupNodes) {
214  FlowLayoutGraphNode subNode = graph.GetNode(subNodeId);
215  if (subNode == null) continue;
216  subNodes.Add(subNode);
217  subNodeIds.Add(subNodeId);
218  foreach (var item in subNode.items)
219  {
220  subItems.Add(item);
221  }
222  coord += subNode.coord;
223  previewLocation += subNode.position;
224  }
225  var numSubNodes = subNodes.Count;
226  if (numSubNodes > 0) {
227  coord /= numSubNodes;
228  previewLocation /= numSubNodes;
229 
230  FlowLayoutGraphNode newNode = graph.CreateNode();
231  newNode.active = true;
232  newNode.items = new List<FlowItem>(subItems);
233  newNode.coord = coord;
234  newNode.position = previewLocation;
235  newNode.MergedCompositeNodes = new List<FlowLayoutGraphNode>(subNodes);
236 
237  // Remove all the sub nodes from the graph
238  foreach (FlowLayoutGraphNode subNode in subNodes) {
239  graph.Nodes.Remove(subNode);
240  }
241 
242  foreach (FlowLayoutGraphLink link in graph.Links) {
243  if (subNodeIds.Contains(link.source)) {
244  link.sourceSubNode = link.source;
245  link.source = newNode.nodeId;
246  }
247  if (subNodeIds.Contains(link.destination)) {
248  link.destinationSubNode = link.destination;
249  link.destination = newNode.nodeId;
250  }
251  }
252 
253  var filteredLinks = new List<FlowLayoutGraphLink>();
254  foreach (var link in graph.Links)
255  {
256  if (link.source != link.destination)
257  {
258  filteredLinks.Add(link);
259  }
260  }
261 
262  graph.Links = filteredLinks;
263  graphQuery.Rebuild();
264  return newNode;
265  }
266 
267  return null;
268  }

393  private void ExecuteImpl(int startIdx, int endIdx)
394  {
395  bool running = true;
396  while (running && !timeout && !foundResult)
397  {
398  running = false;
399  for (int i = startIdx; i <= endIdx; i++)
400  {
401  var growthSystem = growthSystems[i];
402  if (growthSystem.Running)
403  {
404  growthSystem.ExecuteStep(FlowLayoutPathStackGrowthTask.Execute);
405 
406  running |= growthSystem.Running;
407  if (growthSystem.FoundResult)
408  {
409  foundResult = true;
410  result = growthSystem.Result;
411  break;
412  }
413 
414  frameCounter++;
415  if (frameCounter >= maxFramesToProcess)
416  {
417  timeout = true;
418  break;
419  }
420  }
421  }
422  }

436  public static void Execute(FlowLayoutPathStackFrame inFrameState, FlowLayoutStaticGrowthState staticState,
437  StackSystem<FlowLayoutPathStackFrame, FlowLayoutStaticGrowthState, FFlowLayoutPathingSystemResult> stackSystem)
438  {
439  Debug.Assert(staticState.MinPathSize > 0 && staticState.MaxPathSize > 0);
440  Debug.Assert(staticState.GraphQuery != null);
441 
442  var state = inFrameState.State;
443  var currentNode = inFrameState.CurrentNode;
444  var incomingNode = inFrameState.IncomingNode;
445 
446  var pathIndex = state.Path.Count;
447  var pathLength = Mathf.Clamp(pathIndex + 1, staticState.MinPathSize, staticState.MaxPathSize);
448  if (pathIndex == 0 && staticState.HeadNode != null) {
449  // Check if we can connect from the head node to this node
450  if (!staticState.GraphConstraint.IsValid(staticState.GraphQuery, staticState.HeadNode, new FlowLayoutGraphNode[]{currentNode})) {
451  return;
452  }
453  }
454 
455  if (staticState.NodeCreationConstraint != null) {
456  if (!staticState.NodeCreationConstraint.CanCreateNodeAt(currentNode, pathLength, pathIndex)) {
457  return;
458  }
459  }
460 
461  bool bFirstNodeInPath = (pathIndex == 0);
462 
463  var baseIncomingConstraintLinks = new List<FFAGConstraintsLink>();
464  if (bFirstNodeInPath && staticState.HeadNode != null) {
465  var headSubNode = staticState.HeadNode;
466  if (staticState.HeadNode.MergedCompositeNodes.Count > 1) {
467  foreach (var graphLink in staticState.Graph.Links) {
468  if (graphLink.state.type != FlowLayoutGraphLinkType.Unconnected) continue;
469  if (graphLink.source == currentNode.nodeId && graphLink.destination == staticState.HeadNode.nodeId) {
470  headSubNode = staticState.GraphQuery.GetSubNode(graphLink.destinationSubNode);
471  Debug.Assert(headSubNode != null);
472  break;
473  }
474  else if (graphLink.source == staticState.HeadNode.nodeId && graphLink.destination == currentNode.nodeId) {
475  headSubNode = staticState.GraphQuery.GetSubNode(graphLink.sourceSubNode);
476  Debug.Assert(headSubNode != null);
477  break;
478  }
479  }
480  }
481  baseIncomingConstraintLinks.Add(new FFAGConstraintsLink(currentNode, headSubNode));
482  }
483  if (incomingNode != null) {
484  baseIncomingConstraintLinks.Add(new FFAGConstraintsLink(currentNode, incomingNode));
485  }
486 
487  Debug.Assert(staticState.NodeGroupGenerator != null);
488  FlowLayoutPathNodeGroup[] possibleNodeGroupsArray = staticState.NodeGroupGenerator.Generate(staticState.GraphQuery, currentNode, staticState.Random, state.Visited);
489  var possibleNodeGroups = new List<FlowLayoutPathNodeGroup>(possibleNodeGroupsArray);
490 
491  MathUtils.Shuffle(possibleNodeGroups, staticState.Random);
492 
493  {
494  float selectionThreshold = staticState.Random.NextFloat();
495  var hiPriorityGroups = new List<FlowLayoutPathNodeGroup>();
496  for (int i = 0; i < possibleNodeGroups.Count; i++) {
497  if (possibleNodeGroups[i].Weight > selectionThreshold) {
498  hiPriorityGroups.Add(possibleNodeGroups[i]);
499  possibleNodeGroups.RemoveAt(i);
500  i--;
501  }
502  }
503  possibleNodeGroups.AddRange(hiPriorityGroups);
504  }
505 
506  foreach (var growthNodeGroup in possibleNodeGroups) {
507  // Check if we can use this newly created group node by connecting in to it
508  if (!staticState.GraphConstraint.IsValid(staticState.GraphQuery, growthNodeGroup, pathIndex, pathLength, baseIncomingConstraintLinks.ToArray())) continue;
509 
510  FlowLayoutGrowthState nextState = state.Clone();
511 
512  // Update the frame path and visited state
513  foreach (var groupNode in growthNodeGroup.GroupNodes)
514  {
515  nextState.Visited.Add(groupNode);
516  }
517 
518  var pathFrame = new FlowLayoutGrowthStatePathItem();
519  pathFrame.NodeId = currentNode.nodeId;
520  pathFrame.PreviousNodeId = incomingNode != null ? incomingNode.nodeId : DungeonUID.Empty;
521  nextState.Path.Add(pathFrame);
522 
523  // Add path node group info
524  if (growthNodeGroup.IsGroup) {
525  var nodeGroup = new FlowLayoutGraphNodeGroup();
526  nodeGroup.GroupId = DungeonUID.NewUID();
527  nodeGroup.GroupNodes = growthNodeGroup.GroupNodes;
528  nextState.NodeGroups.Add(nodeGroup);
529  }
530 
531  // Check if we reached the desired path size
532  if (nextState.Path.Count >= staticState.MinPathSize) {
533  // Check if we are near the sink node, if any
534  if (staticState.SinkNodes.Count == 0) {
535  // No sink nodes defined.
536  var result = new FFlowLayoutPathingSystemResult(nextState, staticState);
537  stackSystem.FinalizeResult(result);
538  return;
539  }
540 
541  {
542  var sinkNodeIndices = MathUtils.GetShuffledIndices(staticState.SinkNodes.Count, staticState.Random);
543  var groupEdgeNodeIndices = MathUtils.GetShuffledIndices(growthNodeGroup.GroupEdgeNodes.Count, staticState.Random);
544  foreach (var groupEdgeNodeIndex in groupEdgeNodeIndices) {
545  var groupEdgeNodeId = growthNodeGroup.GroupEdgeNodes[groupEdgeNodeIndex];
546  var connectedNodeIds = staticState.Graph.GetConnectedNodes(groupEdgeNodeId);
547  var connectedNodeIndices = MathUtils.GetShuffledIndices(connectedNodeIds.Length, staticState.Random);
548  foreach (var connectedNodeIndex in connectedNodeIndices) {
549  var connectedNodeId = connectedNodeIds[connectedNodeIndex];
550  var connectedNode = staticState.GraphQuery.GetNode(connectedNodeId);
551  foreach (var sinkNodeIndex in sinkNodeIndices) {
552  var sinkNode = staticState.SinkNodes[sinkNodeIndex];
553  if (sinkNode == null) continue;
554 
555  if (nextState.Path.Count == 1 && sinkNode == staticState.HeadNode) {
556  // If the path node size is 1, we don't want to connect back to the head node
557  continue;
558  }
559 
560  if (connectedNode == sinkNode) {
561  var groupEdgeNode = staticState.GraphQuery.GetNode(groupEdgeNodeId);
562  // TODO: Iterate through the edge nodes and check if we can connect to the tail node
563  var incomingConstraintLinks = new List<FFAGConstraintsLink>(baseIncomingConstraintLinks);
564  var connectedSubNode = connectedNode;
565  if (connectedNode.MergedCompositeNodes.Count > 1) {
566  foreach (var graphLink in staticState.Graph.Links) {
567  if (graphLink.state.type != FlowLayoutGraphLinkType.Unconnected) continue;
568  if (graphLink.source == groupEdgeNodeId && graphLink.destination == connectedNodeId) {
569  connectedSubNode = staticState.GraphQuery.GetSubNode(graphLink.destinationSubNode);
570  Debug.Assert(connectedSubNode != null);
571  break;
572  }
573  else if (graphLink.source == connectedNodeId && graphLink.destination == groupEdgeNodeId) {
574  connectedSubNode = staticState.GraphQuery.GetSubNode(graphLink.sourceSubNode);
575  Debug.Assert(connectedSubNode != null);
576  break;
577  }
578  }
579  }
580  incomingConstraintLinks.Add(new FFAGConstraintsLink(groupEdgeNode, connectedSubNode));
581  if (!staticState.GraphConstraint.IsValid(
582  staticState.GraphQuery, growthNodeGroup, pathIndex, pathLength, incomingConstraintLinks.ToArray()))
583  {
584  continue;
585  }
586 
587  var sinkIncomingNodes = new List<FlowLayoutGraphNode>() { groupEdgeNode };
588  if (sinkNode == staticState.HeadNode) {
589  // The sink and the head are the same. Add the first node to the connected list
590  var firstNodeInPath = staticState.GraphQuery.GetNode(nextState.Path[0].NodeId);
591  if (firstNodeInPath != null) {
592  sinkIncomingNodes.Add(firstNodeInPath);
593  }
594  }
595 
596  if (!staticState.GraphConstraint.IsValid(staticState.GraphQuery, sinkNode, sinkIncomingNodes.ToArray()))
597  continue;
598 
599  nextState.TailNode = sinkNode;
600  stackSystem.FinalizeResult(new FFlowLayoutPathingSystemResult(nextState, staticState));
601  return;
602  }
603  }
604  }
605  }
606  }
607 
608  if (nextState.Path.Count == staticState.MaxPathSize) {
609  // no sink nodes nearby and we've reached the max path size
610  return;
611  }
612  }
613 
614 
615  // Try to grow from each outgoing node
616  {
617  var groupEdgeNodeIndices = MathUtils.GetShuffledIndices(growthNodeGroup.GroupEdgeNodes.Count, staticState.Random);
618  foreach (var groupEdgeNodeIndex in groupEdgeNodeIndices) {
619  var groupEdgeNodeId = growthNodeGroup.GroupEdgeNodes[groupEdgeNodeIndex];
620  var connectedNodeIds = staticState.Graph.GetConnectedNodes(groupEdgeNodeId);
621  var connectedNodeIndices = MathUtils.GetShuffledIndices(connectedNodeIds.Length, staticState.Random);
622  foreach (var connectedNodeIndex in connectedNodeIndices) {
623  var connectedNodeId = connectedNodeIds[connectedNodeIndex];
624  if (nextState.Visited.Contains(connectedNodeId)) continue;
625 
626  var connectedNode = staticState.GraphQuery.GetNode(connectedNodeId);
627  if (connectedNode == null) continue;
628  if (connectedNode.active) continue;
629  var groupEdgeNode = staticState.GraphQuery.GetNode(groupEdgeNodeId);
630 
631  var incomingConstraintLinks = new List<FFAGConstraintsLink>(baseIncomingConstraintLinks);
632  incomingConstraintLinks.Add(new FFAGConstraintsLink(groupEdgeNode, connectedNode));
633  if (!staticState.GraphConstraint.IsValid(staticState.GraphQuery, growthNodeGroup, pathIndex, pathLength, incomingConstraintLinks.ToArray())) {
634  continue;
635  }
636 
637  var nextFrame = new FlowLayoutPathStackFrame();
638  nextFrame.State = nextState;
639  nextFrame.CurrentNode = connectedNode;
640  nextFrame.IncomingNode = groupEdgeNode;
641  stackSystem.PushFrame(nextFrame);
642  }
643  }
644  }
645  }
646  }
