36  public void ApplyTheme(LevelMarkerList markers, List<DungeonThemeData> Themes)
37  {
38  var instanceCache = new InstanceCache();
39  var constraintProcessor = context.spatialConstraintProcessor;
40 
41  if (random == null)
42  {
43  random = new PMRandom(context.config.Seed);
44  }
45 
46  PropBySocketTypeByTheme_t PropBySocketTypeByTheme = new PropBySocketTypeByTheme_t();
47  foreach (DungeonThemeData Theme in Themes)
48  {
49  CreatePropLookup(Theme, PropBySocketTypeByTheme);
50  }
51 
52  // Collect all the theme override volumes and prepare their theme lookup
53  var overrideVolumes = new List<ThemeOverrideVolume>();
54  Dictionary<Graph, DungeonThemeData> GraphToThemeMapping = new Dictionary<Graph, DungeonThemeData>();
55 
56  // Process the theme override volumes
57  var themeOverrides = context.themeOverrideVolumes;
58  foreach (var volume in themeOverrides)
59  {
60  overrideVolumes.Add(volume);
61  var graph = volume.overrideTheme;
62  if (graph != null && !GraphToThemeMapping.ContainsKey(graph))
63  {
64  DungeonThemeData theme = new DungeonThemeData();
65  theme.BuildFromGraph(volume.overrideTheme);
66  GraphToThemeMapping.Add(volume.overrideTheme, theme);
67 
68  CreatePropLookup(theme, PropBySocketTypeByTheme);
69  }
70  }
71 
72  var srandom = new PMRandom(context.config.Seed);
73 
74  var nodesExecutionContext = new NodeListExecutionContext();
75  nodesExecutionContext.instanceCache = instanceCache;
76  nodesExecutionContext.constraintProcessor = constraintProcessor;
77  nodesExecutionContext.srandom = srandom;
78  nodesExecutionContext.SceneProvider = context.sceneProvider;
79  nodesExecutionContext.objectInstantiator = context.objectInstantiator;
80 
81  var spawnDataList = new List<DungeonNodeSpawnData>();
82 
83  var delayedExecutionList = new Queue<NodeListExecutionData>();
84  // Fill up the markers with the defined mesh data
85  for (int i = 0; i < markers.Count; i++)
86  {
87  PropSocket socket = markers[i];
88  if (!socket.markForDeletion)
89  {
90  DungeonThemeData themeToUse = GetBestMatchedTheme(Themes, socket, PropBySocketTypeByTheme); // PropAsset;
91  DungeonThemeData fallbackThemeToUse = themeToUse;
92 
93  // Check if this socket resides within a override volume
94  {
95  var socketPosition = Matrix.GetTranslation(ref socket.Transform);
96  foreach (var volume in overrideVolumes)
97  {
98  if (volume.GetBounds().Contains(socketPosition))
99  {
100  var graph = volume.overrideTheme;
101  if (graph != null && GraphToThemeMapping.ContainsKey(graph))
102  {
103  themeToUse = GraphToThemeMapping[volume.overrideTheme];
104  if (!volume.useBaseThemeForMissingMarkers)
105  {
106  fallbackThemeToUse = themeToUse;
107  }
108  break;
109  }
110  }
111  }
112  }
113 
114  if (themeToUse != null)
115  {
116  PropBySocketType_t PropBySocketType = PropBySocketTypeByTheme[themeToUse];
117  if (!PropBySocketType.ContainsKey(socket.SocketType) && fallbackThemeToUse != null && fallbackThemeToUse != themeToUse)
118  {
119  PropBySocketType = PropBySocketTypeByTheme[fallbackThemeToUse];
120  }
121 
122  if (PropBySocketType.ContainsKey(socket.SocketType))
123  {
124  var data = new NodeListExecutionData();
125  data.socket = socket;
126  data.nodeDataList = PropBySocketType[socket.SocketType];
127 
128  if (ShouldDelayExecution(data.nodeDataList))
129  {
130  delayedExecutionList.Enqueue(data);
131  }
132  else
133  {
134  ExecuteNodesUnderMarker(data, nodesExecutionContext, markers, spawnDataList);
135  }
136  }
137  }
138  }
139 
140  // We execute the delayed node list (that have spatial constraints) at the very end of the list
141  // Each execution of the delayed node however, can add more items to this list
142  bool isLastIndex = (i == markers.Count - 1);
143  while (isLastIndex && delayedExecutionList.Count > 0)
144  {
145  var data = delayedExecutionList.Dequeue();
146  if (!data.socket.markForDeletion)
147  {
148  ExecuteNodesUnderMarker(data, nodesExecutionContext, markers, spawnDataList);
149  }
150 
151  isLastIndex = (i == markers.Count - 1);
152  }
153  }
154 
155  RecursivelyTagMarkersForDeletion(markers);
156 
157  if (context.objectSpawner != null)
158  {
159  context.objectSpawner.Destroy();
160  }
161 
162 
163  context.objectSpawner.Spawn(spawnDataList.ToArray(), nodesExecutionContext.SceneProvider,
164  random, nodesExecutionContext.objectInstantiator, context.spawnListeners);
165  }

167  // Picks a theme from the list that has a definition for the defined socket
168  DungeonThemeData GetBestMatchedTheme(List<DungeonThemeData> Themes, PropSocket socket, PropBySocketTypeByTheme_t PropBySocketTypeByTheme)
169  {
170  var ValidThemes = new List<DungeonThemeData>();
171  foreach (DungeonThemeData Theme in Themes)
172  {
173  if (PropBySocketTypeByTheme.ContainsKey(Theme))
174  {
175  PropBySocketType_t PropBySocketType = PropBySocketTypeByTheme[Theme];
176  if (PropBySocketType.Count > 0)
177  {
178  if (PropBySocketType.ContainsKey(socket.SocketType) && PropBySocketType[socket.SocketType].Count > 0)
179  {
180  ValidThemes.Add(Theme);
181  }
182  }
183  }
184  }
185  if (ValidThemes.Count == 0)
186  {
187  return null;
188  }
189 
190  int index = Mathf.FloorToInt(random.GetNextUniformFloat() * ValidThemes.Count) % ValidThemes.Count;
191  return ValidThemes[index];
192  }

289  void ExecuteNodesUnderMarker(NodeListExecutionData data, NodeListExecutionContext nodeContext, LevelMarkerList markers, List<DungeonNodeSpawnData> spawnDataList)
290  {
291  var marker = data.socket;
292  var nodeDataList = data.nodeDataList;
293  foreach (DungeonThemeItem nodeData in nodeDataList)
294  {
295  bool insertMesh = false;
296  Matrix4x4 transform = marker.Transform * nodeData.Offset;
297 
298  if (nodeData.UseSelectionRule && nodeData.SelectorRuleClassName != null)
299  {
300  var selectorRule = nodeContext.instanceCache.GetInstance(nodeData.SelectorRuleClassName) as SelectorRule;
301  if (selectorRule != null)
302  {
303  // Run the selection rule logic to determine if we need to insert this mesh in the scene
304  insertMesh = selectorRule.CanSelect(marker, transform, context.model, random.UniformRandom);
305  }
306  }
307  else
308  {
309  // Perform probability based selection logic
310  float probability = nodeContext.srandom.GetNextUniformFloat();
311  insertMesh = (probability < nodeData.Affinity);
312  }
313 
314  if (insertMesh && nodeData.useSpatialConstraint && nodeData.spatialConstraint != null)
315  {
316  Matrix4x4 spatialOffset;
317  PropSocket[] markersToRemove;
318  if (!ProcessSpatialConstraint(nodeContext.constraintProcessor, nodeData.spatialConstraint, marker, markers, out spatialOffset, out markersToRemove))
319  {
320  // Fails spatial constraint
321  insertMesh = false;
322  }
323  else
324  {
325  // Apply the offset
326  var markerOffset = marker.Transform;
327  if (nodeData.spatialConstraint != null && !nodeData.spatialConstraint.applyMarkerRotation)
328  {
329  var markerPosition = Matrix.GetTranslation(ref markerOffset);
330  var markerScale = Matrix.GetScale(ref markerOffset);
331  markerOffset = Matrix4x4.TRS(markerPosition, Quaternion.identity, markerScale);
332  }
333  transform = markerOffset * spatialOffset * nodeData.Offset;
334 
335  foreach (var markerToRemove in markersToRemove)
336  {
337  markerToRemove.markForDeletion = true;
338  }
339  }
340  }
341 
342  if (insertMesh)
343  {
344  // Attach this prop to the socket
345  // Apply transformation logic, if specified
346  if (nodeData.UseTransformRule && nodeData.TransformRuleClassName != null && nodeData.TransformRuleClassName.Length > 0)
347  {
348  var transformer = nodeContext.instanceCache.GetInstance(nodeData.TransformRuleClassName) as TransformationRule;
349  if (transformer != null)
350  {
351  Vector3 _position, _scale;
352  Quaternion _rotation;
353  transformer.GetTransform(marker, context.model, transform, random.UniformRandom, out _position, out _rotation, out _scale);
354  var offset = Matrix4x4.TRS(_position, _rotation, _scale);
355  transform = transform * offset;
356  }
357  }
358 
359  // Create a spawn request
360  var spawnData = new DungeonNodeSpawnData();
361  spawnData.nodeData = nodeData;
362  spawnData.transform = transform;
363  spawnData.socket = marker;
364  spawnDataList.Add(spawnData);
365 
366  // Add child markers if any
367  foreach (PropChildSocketData ChildSocket in nodeData.ChildSockets)
368  {
369  Matrix4x4 childTransform = transform * ChildSocket.Offset;
370  var childMarker = markers.EmitMarker(ChildSocket.SocketType, childTransform, marker.gridPosition, marker.cellId);
371  data.socket.childMarkers.Add(childMarker);
372  }
373 
374  if (nodeData.ConsumeOnAttach)
375  {
376  // Attach no more on this socket
377  break;
378  }
379  }
380  }
381  }
382  }
