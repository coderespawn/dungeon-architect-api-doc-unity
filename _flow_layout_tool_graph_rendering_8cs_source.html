360  void OrientLinkToNodes()
361  {
362  var headThickness = settings.Thickness * FlowLayout3DConstants.LinkHeadThicknessMultiplier;
363  float headLength = headThickness;
364  float headWidth = headThickness;
365  if (settings.OneWay)
366  {
367  headLength *= 2;
368  }
369 
370  var startTransform = SxSceneGraphUtils.AccumulateTransforms(settings.StartNode);
371  var endTransform = SxSceneGraphUtils.AccumulateTransforms(settings.EndNode);
372 
373  var start = Matrix.GetTranslation(ref startTransform);
374  var end = Matrix.GetTranslation(ref endTransform);
375 
376  var direction = (end - start).normalized;
377  start = start + direction * settings.StartRadius;
378  end = end - direction * (settings.EndRadius + headLength);
379 
380  var length = (end - start).magnitude;
381 
382  Rotation = Quaternion.FromToRotation(new Vector3(1, 0, 0), direction);
383  Position = start + (end - start) * 0.5f;
384  Scale = new Vector3(length / 2.0f, settings.Thickness / 2.0f, 1);
385 
386  {
387  var headScale = Vector3.Scale(Vector3.one / 2.0f, new Vector3(headLength, headWidth, 1));
388  headScale = MathUtils.Divide(headScale, Scale);
389  headActor.Scale = headScale;
390  headActor.Position = MathUtils.Divide(new Vector3(length * 0.5f + headLength * 0.5f, 0, 0), Scale);
391  }
