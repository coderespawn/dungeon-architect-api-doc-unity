206  public static bool HandleNodeInput(GraphNode node, Event e, GraphEditor graphEditor, UISystem uiSystem)
207  {
208  bool inputProcessed = false;
209  if (!node.Dragging)
210  {
211  // let the pins handle the input first
212  foreach (var pin in node.InputPins)
213  {
214  if (inputProcessed) break;
215  inputProcessed |= HandlePinInput(pin, e, graphEditor, uiSystem);
216  }
217  foreach (var pin in node.OutputPins)
218  {
219  if (inputProcessed) break;
220  inputProcessed |= HandlePinInput(pin, e, graphEditor, uiSystem);
221  }
222  }
223 
224  var mousePosition = e.mousePosition;
225  var mousePositionWorld = graphEditor.Camera.ScreenToWorld(mousePosition);
226  int dragButton = 0;
227  // If the pins didn't already handle the input, then let the node handle it
228  if (!inputProcessed)
229  {
230  bool insideRect = node.Bounds.Contains(mousePositionWorld);
231  if (e.type == EventType.MouseDown && insideRect && e.button == dragButton)
232  {
233  node.Dragging = true;
234  inputProcessed = true;
235  }
236  else if (e.type == EventType.MouseUp && insideRect && e.button == dragButton)
237  {
238  node.Dragging = false;
239  }
240  }
241 
242  if (node.Dragging && !node.Selected)
243  {
244  node.Dragging = false;
245  }
246 
247  if (node.Dragging && e.type == EventType.MouseDrag)
248  {
249  inputProcessed = true;
250  }
251 
252  return inputProcessed;
253  }

259  public static bool HandlePinInput(GraphPin pin, Event e, GraphEditor graphEditor, UISystem uiSystem)
260  {
261  var camera = graphEditor.Camera;
262  var mousePosition = e.mousePosition;
263  var mousePositionWorld = camera.ScreenToWorld(mousePosition);
264  int buttonIdDrag = 0;
265  int buttonIdDestroyLinks = 1;
266  if (pin.ContainsPoint(mousePositionWorld))
267  {
268  if (e.type == EventType.MouseDown && e.button == buttonIdDrag)
269  {
270  pin.ClickState = GraphPinMouseState.Clicked;
271  return true;
272  }
273 
274  if (e.button == buttonIdDestroyLinks)
275  {
276  if (e.type == EventType.MouseDown)
277  {
278  pin.RequestLinkDeletionInitiated = true;
279  }
280  else if (e.type == EventType.MouseDrag)
281  {
282  pin.RequestLinkDeletionInitiated = false;
283  }
284  else if (e.type == EventType.MouseUp)
285  {
286  if (pin.RequestLinkDeletionInitiated)
287  {
288  DestroyPinLinks(pin, uiSystem.Undo);
289  if (pin.Node != null && pin.Node.Graph != null)
290  {
291  graphEditor.HandleGraphStateChanged(uiSystem);
292  }
293  }
294  }
295  return true;
296  }
297 
298  if (pin.ClickState != GraphPinMouseState.Clicked)
299  {
300  pin.ClickState = GraphPinMouseState.Hover;
301  }
302  }
303  else
304  {
305  pin.ClickState = GraphPinMouseState.None;
306  }
307 
308  return false;
309  }
