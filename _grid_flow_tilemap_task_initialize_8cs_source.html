138  y0 = y1 = y;

143  x1 = Mathf.Max(x1, x);

145  y1 = Mathf.Max(y1, y);
146  }

192  void PerturbRoomSizes(FlowTilemap tilemap, FlowLayoutGraph graph, TilemapFlowNodeInfo[,] tileNodes, System.Random random)
193  {
194  // Perturb horizontally
195  {
196  //perturbAmount = Mathf.Min(perturbAmount, tilemapSizePerNode * 0.5f - corridorLaneWidth);
197  //perturbAmount = Mathf.Max(0, perturbAmount);
198 
199  var maxPerturbAmount = Mathf.FloorToInt((tilemapSizePerNode.x - 2 - corridorLaneWidth) * 0.5f);
200  maxPerturbAmount = Mathf.Max(0, maxPerturbAmount);
201 
202  var perturbX = Mathf.Min(perturbAmount.x, maxPerturbAmount);
203  perturbX = Mathf.Max(0, perturbX);
204 
205  for (int ny = 0; ny < nodeHeight; ny++)
206  {
207  for (int nx = -1; nx < nodeWidth; nx++)
208  {
209  var nodeA = (nx >= 0) ? tileNodes[nx, ny] : null;
210  var nodeB = (nx + 1 < nodeWidth) ? tileNodes[nx + 1, ny] : null;
211 
212  bool connected = false;
213  if (nodeA != null && nodeB != null)
214  {
215  var link = graph.GetLink(nodeA.node, nodeB.node, true);
216  connected = (link != null && link.state.type != FlowLayoutGraphLinkType.Unconnected);
217  }
218 
219  if (connected)
220  {
221  float perturb = random.Range(-perturbX, perturbX);
222  if (nodeA != null) nodeA.x1 += perturb;
223  if (nodeB != null) nodeB.x0 += perturb;
224  }
225  else
226  {
227  float perturbA = perturbX * random.NextFloat();
228  float perturbB = perturbX * random.NextFloat();
229  if (nodeA != null) nodeA.x1 -= perturbA;
230  if (nodeB != null) nodeB.x0 += perturbB;
231  }
232  }
233  }
234  }
235 
236  // Perturb vertically
237  {
238  var maxPerturbAmount = Mathf.FloorToInt((tilemapSizePerNode.y - 2 - corridorLaneWidth) * 0.5f);
239  maxPerturbAmount = Mathf.Max(0, maxPerturbAmount);
240 
241  var perturbY = Mathf.Min(perturbAmount.y, maxPerturbAmount);
242  perturbY = Mathf.Max(0, perturbY);
243 
244  for (int nx = 0; nx < nodeWidth; nx++)
245  {
246  for (int ny = -1; ny < nodeHeight; ny++)
247  {
248  var nodeA = (ny >= 0) ? tileNodes[nx, ny] : null;
249  var nodeB = (ny + 1 < nodeHeight) ? tileNodes[nx, ny + 1] : null;
250 
251  bool connected = false;
252  if (nodeA != null && nodeB != null)
253  {
254  var link = graph.GetLink(nodeA.node, nodeB.node, true);
255  connected = (link != null && link.state.type != FlowLayoutGraphLinkType.Unconnected);
256  }
257 
258  if (connected)
259  {
260  bool canMoveDown = (nodeA.x0 >= nodeB.x0 && nodeA.x1 <= nodeB.x1);
261  bool canMoveUp = (nodeB.x0 >= nodeA.x0 && nodeB.x1 <= nodeA.x1);
262 
263  if (!canMoveUp && !canMoveDown) continue;
264 
265  if (canMoveUp && canMoveDown)
266  {
267  // Move randomly on either one direction
268  if (random.NextFloat() < 0.5f)
269  {
270  canMoveUp = false;
271  }
272  else
273  {
274  canMoveDown = false;
275  }
276  }
277 
278  float perturbDirection = (canMoveUp ? -1 : 1);
279  float perturb = random.NextFloat() * perturbY * perturbDirection;
280  nodeA.y1 += perturb;
281  nodeB.y0 += perturb;
282  }
283  else
284  {
285  float perturbA = perturbY * random.NextFloat();
286  float perturbB = perturbY * random.NextFloat();
287  if (nodeA != null) nodeA.y1 -= perturbA;
288  if (nodeB != null) nodeB.y0 += perturbB;
289  }
290  }
291  }
292  }
293  }

301  void FixCorridorSizes(TilemapFlowNodeInfo[,] tileNodes, FlowLayoutGraph graph)
302  {

304  {
305  var node = tileNode.node;
306  if (GetRoomType(node) == GridFlowLayoutNodeRoomType.Corridor)

309  var outgoingNodes = graph.GetOutgoingNodes(node);
310  if (incomingNodes.Length == 0 || outgoingNodes.Length == 0) continue;
311 
312  var incomingNode = incomingNodes[0];
313  var outgoingNode = outgoingNodes[0];

317  if (vertical)
318  {
319  tileNode.x0 = Mathf.Max(tileNode.x0, tileNode.midX - corridorLaneWidth);
320  tileNode.x1 = Mathf.Min(tileNode.x1, tileNode.midX + corridorLaneWidth);
321  }
322  else
323  {
324  tileNode.y0 = Mathf.Max(tileNode.y0, tileNode.midY - corridorLaneWidth);

326  }

330  void RasterizeRoomCorridors(TilemapFlowNodeInfo[,] tileNodes, FlowTilemap tilemap)
331  {
332  foreach (var tileNode in tileNodes)
333  {
334  if (!tileNode.node.active) continue;
335  bool wallsAsTiles = wallGenerationMethod == TilemapFlowNodeWallGenerationMethod.WallAsTiles;
336  var b = NodeTilemapBounds.Build(tileNode, tilemap.Width, tilemap.Height, wallsAsTiles);
337 
338  if (GetRoomType(tileNode.node) == GridFlowLayoutNodeRoomType.Cave)
339  {
340  // Render the caves in another pass
341  continue;
342  }
343 
344  var nodeCoord = GetNodeCoord(tileNode.node);
345  // Rasterize the tiles
346  for (int y = b.y0; y <= b.y1; y++)
347  {
348  for (int x = b.x0; x <= b.x1; x++)
349  {
350  var cell = tilemap.Cells.GetCell(x, y);
351  if (cell == null) continue;
352  cell.LayoutCell = true;
353 
354  cell.NodeCoord = nodeCoord;
355  if (wallsAsTiles)
356  {
357  cell.CellType = (x == b.x0 || x == b.x1 || y == b.y0 || y == b.y1)
358  ? FlowTilemapCellType.Wall
359  : FlowTilemapCellType.Floor;
360  }
361  else
362  {
363  if (x < b.x1 && y < b.y1)
364  {
365  cell.CellType = FlowTilemapCellType.Floor;
366  }
367  }
368 
369  var nodeColor = tileNode.node.color;
370  cell.CustomColor = ColorUtils.BrightenColor(nodeColor, roomColorSaturation, roomColorBrightness);
371 
372  if (cell.CellType == FlowTilemapCellType.Floor)
373  {
374  cell.UseCustomColor = true;
375  }
376 
377  else if (cell.CellType == FlowTilemapCellType.Wall)
378  {
379  FlowTilemapCellWallInfo wallInfo = cell.Userdata as FlowTilemapCellWallInfo;
380  if (wallInfo == null)
381  {
382  wallInfo = new FlowTilemapCellWallInfo();
383  cell.Userdata = wallInfo;
384  }
385 
386  wallInfo.owningNodes.Add(nodeCoord);
387  }
388  }
389  }
390 
391  // Rasterize the edges
392  if (!wallsAsTiles)
393  {
394  for (int y = b.y0; y < b.y1; y++)
395  {
396  var edge1 = tilemap.Edges.GetVertical(b.x0, y);
397  var edge2 = tilemap.Edges.GetVertical(b.x1, y);
398  edge1.EdgeType = FlowTilemapEdgeType.Wall;
399  edge2.EdgeType = FlowTilemapEdgeType.Wall;
400  }
401  for (int x = b.x0; x < b.x1; x++)
402  {
403  var edge1 = tilemap.Edges.GetHorizontal(x, b.y0);
404  var edge2 = tilemap.Edges.GetHorizontal(x, b.y1);
405  edge1.EdgeType = FlowTilemapEdgeType.Wall;
406  edge2.EdgeType = FlowTilemapEdgeType.Wall;
407  }
408  }
409  }

414  private void RasterizeBaseCaveBlocks(TilemapFlowNodeInfo[,] tileNodes, FlowTilemap tilemap, FlowLayoutGraph graph)
415  {
416  foreach (var tileNode in tileNodes)
417  {
418  if (!tileNode.node.active) continue;
419  bool wallsAsTiles = wallGenerationMethod == TilemapFlowNodeWallGenerationMethod.WallAsTiles;
420  var b = NodeTilemapBounds.Build(tileNode, tilemap.Width, tilemap.Height, wallsAsTiles);
421 
422  if (GetRoomType(tileNode.node) != GridFlowLayoutNodeRoomType.Cave)
423  {
424  // Only build the caves in this pass
425  continue;
426  }
427 
428  var nodeCoord = GetNodeCoord(tileNode.node);
429  var caveNode = tileNode.node;
430  var blockLeft = ShouldBlockCaveBoundary(graph, caveNode, -1, 0);
431  var blockRight = ShouldBlockCaveBoundary(graph, caveNode, 1, 0);
432  var blockTop = ShouldBlockCaveBoundary(graph, caveNode, 0, -1);
433  var blockBottom = ShouldBlockCaveBoundary(graph, caveNode, 0, 1);
434  for (int y = b.y0; y <= b.y1; y++)
435  {
436  for (int x = b.x0; x <= b.x1; x++)
437  {
438  var cell = tilemap.Cells.GetCell(x, y);
439  if (cell == null) continue;
440 
441  if (cell.CellType == FlowTilemapCellType.Empty)
442  {
443  var makeFloor = true;
444  if (wallsAsTiles)
445  {
446  if (blockLeft && x == b.x0) makeFloor = false;
447  if (blockRight && x == b.x1) makeFloor = false;
448  if (blockTop && y == b.y0) makeFloor = false;
449  if (blockBottom && y == b.y1) makeFloor = false;
450  }
451  else
452  {
453  if (blockRight && x == b.x1) makeFloor = false;
454  if (blockBottom && y == b.y1) makeFloor = false;
455 
456  if (blockLeft && x == b.x0 && y == b.y0) makeFloor = false;
457  if (blockLeft && x == b.x0 && y == b.y1) makeFloor = false;
458 
459  if (blockTop && x == b.x0 && y == b.y1) makeFloor = false;
460  if (blockTop && x == b.x1 && y == b.y1) makeFloor = false;
461  }
462 
463  if (makeFloor)
464  {
465  cell.NodeCoord = nodeCoord;
466  cell.CellType = FlowTilemapCellType.Floor;
467  cell.UseCustomColor = true;
468  var nodeColor = tileNode.node.color;
469  cell.CustomColor = ColorUtils.BrightenColor(nodeColor, roomColorSaturation, roomColorBrightness);
470  }
471  }
472  }
473  }

500  private void BuildCaveStep_BuildRocks(CaveCellBuildTile[,] caveMap, FlowTilemap tilemap, System.Random random)
501  {
502  foreach (var caveCell in caveMap)
503  {
504  var tileCell = tilemap.Cells[caveCell.tileCoord.x, caveCell.tileCoord.y];
505  if (caveThickness > 0)
506  {
507  var rockProbability = Mathf.Exp(-tileCell.DistanceFromMainPath / caveThickness);
508  caveCell.rockTile = random.NextFloat() < rockProbability;
509  }
510  else
511  {
512  caveCell.rockTile = (tileCell.DistanceFromMainPath == 0);

516  private void BuildCaveStep_SimulateGrowth(CaveCellBuildTile[,] caveMap, FlowTilemap tilemap, System.Random random)
517  {
518  var width = caveMap.GetLength(0);
519  var height = caveMap.GetLength(1);
520 
521  for (int i = 0; i < caveAutomataIterations; i++)
522  {
523  CaveCellBuildTile[,] oldMap = new CaveCellBuildTile[width, height];
524  for (int y = 0; y < height; y++)
525  {
526  for (int x = 0; x < width; x++)
527  {
528  oldMap[x, y] = caveMap[x, y].Clone();
529  }
530  }
531 
532  for (int y = 0; y < height; y++)
533  {
534  for (int x = 0; x < width; x++)
535  {
536  int nrocks = 0;
537  for (int dy = -1; dy <= 1; dy++)
538  {
539  for (int dx = -1; dx <= 1; dx++)
540  {
541  if (dx == 0 && dy == 0) continue;
542  int nx = x + dx;
543  int ny = y + dy;
544  if (nx < 0 || ny < 0 || nx >= width || ny >= height) continue;
545  if (oldMap[nx, ny].rockTile)
546  {
547  nrocks++;
548  }
549  }
550  }
551 
552 
553  if (nrocks >= caveAutomataNeighbors)
554  {
555  caveMap[x, y].rockTile = true;
556  }
557  }
558  }
559 
560  }
561 
562  for (int y = 0; y < height; y++)
563  {
564  for (int x = 0; x < width; x++)
565  {
566  if (caveMap[x, y].valid && !caveMap[x, y].rockTile)
567  {
568  tilemap.Cells[x, y].CellType = FlowTilemapCellType.Empty;
569  tilemap.Cells[x, y].UseCustomColor = false;
570  caveMap[x, y].valid = false;
571  }
572  }
573  }

575  private void BuildCaveStep_Cleanup(CaveCellBuildTile[,] caveMap, TilemapFlowNodeInfo[,] tileNodes, FlowTilemap tilemap)
576  {
577  var width = tilemap.Width;
578  var height = tilemap.Height;
579  var traversibleCaveTiles = new bool[width, height];
580 
581  var childOffsets = new int[]
582  {
583  -1, 0,
584  1, 0,
585  0, -1,
586  0, 1
587  };
588 
589  foreach (var tileNode in tileNodes)
590  {
591  if (GetRoomType(tileNode.node) != GridFlowLayoutNodeRoomType.Cave)
592  {
593  // Only process the caves
594  continue;
595  }
596 
597  if (!tileNode.node.active)
598  {
599  // Do not process inactive nodes
600  continue;
601  }
602 
603  var nodeCoord = GetNodeCoord(tileNode.node);
604  var tileCenter = NodeCoordToTileCoord(nodeCoord);
605 
606  if (traversibleCaveTiles[tileCenter.x, tileCenter.y])
607  {
608  // Already processed from another adjacent node
609  continue;
610  }
611 
612  // Flood fill from the center of this node
613  var queue = new Queue<IntVector2>();
614  queue.Enqueue(tileCenter);
615  while (queue.Count > 0)
616  {
617  var front = queue.Dequeue();
618  if (traversibleCaveTiles[front.x, front.y])
619  {
620  // Already processed
621  continue;
622  }
623 
624  traversibleCaveTiles[front.x, front.y] = true;
625 
626  // Traverse the children
627  for (int i = 0; i < 4; i++)
628  {
629  var childCoord = new IntVector2(
630  front.x + childOffsets[i * 2 + 0],
631  front.y + childOffsets[i * 2 + 1]);
632 
633  if (childCoord.x >= 0 && childCoord.y >= 0 && childCoord.x < width && childCoord.y < height)
634  {
635  if (caveMap[childCoord.x, childCoord.y].valid)
636  {
637  var visited = traversibleCaveTiles[childCoord.x, childCoord.y];
638  if (!visited)
639  {
640  queue.Enqueue(childCoord);
641  }
642  }
643  }
644  }
645  }
646  }
647 
648  // Assign the valid traversable paths
649  for (int y = 0; y < height; y++)
650  {
651  for (int x = 0; x < width; x++)
652  {
653  var cell = tilemap.Cells[x, y];
654  var nodeCoord = cell.NodeCoord;
655  var tileNode = tileNodes[nodeCoord.x, nodeCoord.y];
656  if (GetRoomType(tileNode.node) == GridFlowLayoutNodeRoomType.Cave)
657  {
658  var valid = traversibleCaveTiles[x, y];
659  caveMap[x, y].valid = valid;
660  if (!valid)
661  {
662  tilemap.Cells[x, y].CellType = FlowTilemapCellType.Empty;
663  tilemap.Cells[x, y].UseCustomColor = false;
664  }
