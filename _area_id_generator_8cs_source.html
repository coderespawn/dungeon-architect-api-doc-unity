200  public Area[] ToArray()
201  {
202  Area[] areas = new Area[triCount];
203 
204  int i = 0;
205  foreach (var tri in tris)
206  {
207  areas[i] = defaultArea;
208 
209  foreach (var condition in conditions)

212 
213  i++;
214  }
