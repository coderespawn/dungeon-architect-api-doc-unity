33  {
34  BBox3 bounds = triangles.GetBoundingBox(settings.CellSize);
35  var hf = new Heightfield(bounds, settings);
36  hf.RasterizeTriangles(triangles);
37  hf.FilterLedgeSpans(settings.VoxelAgentHeight, settings.VoxelMaxClimb);
38  hf.FilterLowHangingWalkableObstacles(settings.VoxelMaxClimb);
39  hf.FilterWalkableLowHeightSpans(settings.VoxelAgentHeight);
40  var chf = new CompactHeightfield(hf, settings);
41  chf.Erode(settings.VoxelAgentRadius);

43  chf.BuildRegions(2, settings.MinRegionSize, settings.MergedRegionSize);
