141  {

155  var visited = new HashSet<IntVector2>();
156  var queue = new Queue<IntVector2>();
157  queue.Enqueue(player);
158  visited.Add(player);
159  while (queue.Count > 0)
160  {
161  var coord = queue.Dequeue();
162  result.Add(coord);
163 
164  var cell = tilemap.Cells.GetCell(coord.x, coord.y);
165  if (cell == null) continue;
166  if (cell.CellType == FlowTilemapCellType.Wall) continue;
167  if (cell.CellType == FlowTilemapCellType.Door) continue;
168 
169  for (int dy = -1; dy <= 1; dy++)
170  {
171  for (int dx = -1; dx <= 1; dx++)
172  {
173  if (dx == 0 && dy == 0)
174  {
175  continue;
176  }
177 
178  var cx = coord.x + dx;
179  var cy = coord.y + dy;
180 
181  if (cx < 0 || cy < 0 || cx >= tilemap.Width || cy >= tilemap.Height) continue;
182 
183  var childCoord = new IntVector2(cx, cy);
184  if (visited.Contains(childCoord) || !valid.Contains(childCoord))
185  {
186  continue;
187  }

189  queue.Enqueue(childCoord);
190  }

211  var gridCoord = new Vector2(
212  localPosition.x / config.gridSize.x,
213  localPosition.z / config.gridSize.z);
214 
215  item.position = gridCoord;
216  item.radius = trackedObject.fogOfWarNumTileRadius;
217  item.falloffStart = trackedObject.fogOfWarLightFalloffStart;
218  items.Add(item);

222  foreach (var item in items)
223  {
224  int cx = Mathf.FloorToInt(item.position.x);
225  int cy = Mathf.FloorToInt(item.position.y);
226 
227  int x0 = Mathf.FloorToInt(item.position.x - item.radius);
228  int x1 = Mathf.FloorToInt(item.position.x + item.radius);
229  int y0 = Mathf.FloorToInt(item.position.y - item.radius);
230  int y1 = Mathf.FloorToInt(item.position.y + item.radius);
231 
232  cx = Mathf.Clamp(cx, 0, tilemap.Width - 1);
233  cy = Mathf.Clamp(cy, 0, tilemap.Height - 1);
234 
235  x0 = Mathf.Clamp(x0, 0, tilemap.Width - 1);
236  x1 = Mathf.Clamp(x1, 0, tilemap.Width - 1);
237  y0 = Mathf.Clamp(y0, 0, tilemap.Height - 1);
238  y1 = Mathf.Clamp(y1, 0, tilemap.Height - 1);
239  var falloff = Mathf.Clamp01(1 - item.falloffStart);
240  var tilesToProcess = new List<IntVector2>();
241  for (int y = y0; y <= y1; y++)
242  {
243  for (int x = x0; x <= x1; x++)
244  {
245  tilesToProcess.Add(new IntVector2(x, y));
246  }
247  }
248 
249  var visibleTiles = FogOfWarFilterVisibility(tilesToProcess.ToArray(), new IntVector2(cx, cy));
250  foreach (var tile in visibleTiles)
251  {
252  float distance = (new Vector2(tile.x, tile.y) - item.position).magnitude;
253  distance = Mathf.Clamp01(distance / item.radius);
254  var weight = 1 - distance;
255

257 
258  weight = Mathf.Clamp01(weight);
259

261  fogMask[index].r = Mathf.Max(weight, fogMask[index].r);
262  }
