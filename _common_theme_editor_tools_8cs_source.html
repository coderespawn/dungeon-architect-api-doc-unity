118  static void CreateDefaultMarkersFor(GraphEditor graphEditor, Type builderType, UISystem uiSystem)
119  {
120  // Remove unused nodes
121  // Grab the names of all the markers nodes in the graph
122  var markerNames = new List<string>();
123  foreach (var node in graphEditor.Graph.Nodes)
124  {
125  if (node is MarkerNode)
126  {
127  var markerNode = node as MarkerNode;
128  markerNames.Add(markerNode.Caption);
129  }
130  }
131 
132  var unusedMarkers = new List<string>(markerNames.ToArray());
133 
134 
135  // Remove markers from the unused list that have child nodes attached to it
136  foreach (var node in graphEditor.Graph.Nodes)
137  {
138  if (node is VisualNode)
139  {
140  var visualNode = node as VisualNode;
141  foreach (var parentNode in visualNode.GetParentNodes())
142  {
143  if (parentNode is MarkerNode)
144  {
145  var markerNode = parentNode as MarkerNode;
146  unusedMarkers.Remove(markerNode.Caption);
147  }
148  }
149  }
150  }
151 
152  // Remove markers from the unused list that are referenced by other marker emitters
153  foreach (var node in graphEditor.Graph.Nodes)
154  {
155  if (node is MarkerEmitterNode)
156  {
157  var emitterNode = node as MarkerEmitterNode;
158  string markerName = emitterNode.Caption;
159  // this marker is referenced by an emitter. Remove it from the unused list
160  unusedMarkers.Remove(markerName);
161  }
162  }
163 
164  // Remove markers from the unused list that are required by the new builder type
165  var defaultMarkerRepository = new DungeonBuilderDefaultMarkers();
166  var builderMarkers = defaultMarkerRepository.GetDefaultMarkers(builderType);
167  foreach (var builderMarker in builderMarkers)
168  {
169  unusedMarkers.Remove(builderMarker);
170  }
171 
172  // Remove all the unused markers
173  var markerNodesToDelete = new List<MarkerNode>();
174  foreach (var node in graphEditor.Graph.Nodes)
175  {
176  if (node is MarkerNode)
177  {
178  var markerNode = node as MarkerNode;
179  if (unusedMarkers.Contains(markerNode.Caption)) {
180  markerNodesToDelete.Add(markerNode);
181  }
182  }
183  }
184 
185  graphEditor.DeleteNodes(markerNodesToDelete.ToArray(), uiSystem);
186 
187 
188  // Grab the names of all the markers nodes in the graph
189  markerNames.Clear();
190  foreach (var node in graphEditor.Graph.Nodes)
191  {
192  if (node is MarkerNode)
193  {
194  var markerNode = node as MarkerNode;
195  markerNames.Add(markerNode.Caption);
196  }
197  }
198 
199  var markersToCreate = new List<string>();
200  foreach (var builderMarker in builderMarkers)
201  {
202  if (!markerNames.Contains(builderMarker))
203  {
204  markersToCreate.Add(builderMarker);
205  }
206  }
207 
208  var existingBounds = new List<Rect>();
209  foreach (var node in graphEditor.Graph.Nodes)
210  {
211  existingBounds.Add(node.Bounds);
212  }
213 
214  // Add the new nodes
215  const int INTER_NODE_X = 200;
216  const int INTER_NODE_Y = 300;
217  int itemsPerRow = 5;
218  int positionIndex = 0;
219  int ix, iy, x, y;
220  var markerNodeSize = new Vector2(120, 50);
221  for (int i = 0; i < markersToCreate.Count; i++)
222  {
223  bool overlaps;
224  int numOverlapTries = 0;
225  int MAX_OVERLAP_TRIES = 100;
226  do
227  {
228  ix = positionIndex % itemsPerRow;
229  iy = positionIndex / itemsPerRow;
230  x = ix * INTER_NODE_X;
231  y = iy * INTER_NODE_Y;
232  positionIndex++;
233 
234  overlaps = false;
235  var newNodeBounds = new Rect(x, y, markerNodeSize.x, markerNodeSize.y);
236  foreach (var existingBound in existingBounds)
237  {
238  if (newNodeBounds.Overlaps(existingBound))
239  {
240  overlaps = true;
241  break;
242  }
243  }
244  numOverlapTries++;
245  } while (overlaps && numOverlapTries < MAX_OVERLAP_TRIES);
246 
247  var newNode = GraphOperations.CreateNode<MarkerNode>(graphEditor.Graph, uiSystem.Undo);
248  GraphEditorUtils.AddToAsset(new UnityEditorUIPlatform(), graphEditor.Graph, newNode);
249  newNode.Position = new Vector2(x, y);
250  newNode.Caption = markersToCreate[i];
251 
252  }
253  }
254  }
