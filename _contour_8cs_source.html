115  public void MergeWith(Contour contour)
116  {
117  int lengthA = vertices.Length;
118  int lengthB = contour.vertices.Length;
119 
120  int ia, ib;
121  GetClosestIndices(this, contour, out ia, out ib);
122 
123  //create a list with the capacity set to the max number of possible verts to avoid expanding the list.
124  var newVerts = new List<ContourVertex>(vertices.Length + contour.vertices.Length + 2);
125 
126  //copy contour A
127  for (int i = 0; i <= lengthA; i++)
128  newVerts.Add(vertices[(ia + i) % lengthA]);
129 
130  //add contour B (other contour) to contour A (this contour)
131  for (int i = 0; i <= lengthB; i++)
132  newVerts.Add(contour.vertices[(ib + i) % lengthB]);
133 
134  vertices = newVerts.ToArray();
135 
136  //delete the other contour
137  contour.vertices = null;

147  private static void GetClosestIndices(Contour a, Contour b, out int indexA, out int indexB)
148  {
149  int closestDistance = int.MaxValue;
150  int lengthA = a.vertices.Length;
151  int lengthB = b.vertices.Length;
152 
153  indexA = -1;
154  indexB = -1;
155 
156  for (int i = 0; i < lengthA; i++)
157  {
158  int vertA = i;
159  int vertANext = (i + 1) % lengthA;
160  int vertAPrev = (i + lengthA - 1) % lengthA;
161 
162  for (int j = 0; j < lengthB; j++)
163  {
164  int vertB = j;
165 
166  //vertB must be infront of vertA
167  if (ContourVertex.IsLeft(ref a.vertices[vertAPrev], ref a.vertices[vertA], ref b.vertices[vertB]) &&
168  ContourVertex.IsLeft(ref a.vertices[vertA], ref a.vertices[vertANext], ref b.vertices[vertB]))
169  {
170  int dx = b.vertices[vertB].X - a.vertices[vertA].X;
171  int dz = b.vertices[vertB].Z - a.vertices[vertA].Z;
172  int tempDist = dx * dx + dz * dz;
173  if (tempDist < closestDistance)
174  {
175  indexA = i;
176  indexB = j;
177  closestDistance = tempDist;
178  }
179  }
180  }
181  }

192  public static void Simplify(List<ContourVertex> rawVerts, List<ContourVertex> simplified, float maxError, int maxEdgeLen, ContourBuildFlags buildFlags)
193  {
194  bool tesselateWallEdges = (buildFlags & ContourBuildFlags.TessellateWallEdges) == ContourBuildFlags.TessellateWallEdges;
195  bool tesselateAreaEdges = (buildFlags & ContourBuildFlags.TessellateAreaEdges) == ContourBuildFlags.TessellateAreaEdges;
196 
197  //add initial points
198  bool hasConnections = false;
199  for (int i = 0; i < rawVerts.Count; i++)
200  {
201  if (rawVerts[i].RegionId.Id != 0)
202  {
203  hasConnections = true;
204  break;
205  }
206  }
207 
208  if (hasConnections)
209  {
210  //contour has some portals to other regions
211  //add new point to every location where region changes
212  for (int i = 0, end = rawVerts.Count; i < end; i++)
213  {
214  int ii = (i + 1) % end;
215  bool differentRegions = rawVerts[i].RegionId.Id != rawVerts[ii].RegionId.Id;
216  bool areaBorders = RegionId.HasFlags(rawVerts[i].RegionId, RegionFlags.AreaBorder) != RegionId.HasFlags(rawVerts[ii].RegionId, RegionFlags.AreaBorder);
217 
218  if (differentRegions || areaBorders)
219  {
220  simplified.Add(new ContourVertex(rawVerts[i], i));
221  }
222  }
223  }
224 
225  //add some points if thhere are no connections
226  if (simplified.Count == 0)
227  {
228  //find lower-left and upper-right vertices of contour
229  int lowerLeftX = rawVerts[0].X;
230  int lowerLeftY = rawVerts[0].Y;
231  int lowerLeftZ = rawVerts[0].Z;
232  RegionId lowerLeftI = RegionId.Null;
233 
234  int upperRightX = rawVerts[0].X;
235  int upperRightY = rawVerts[0].Y;
236  int upperRightZ = rawVerts[0].Z;
237  RegionId upperRightI = RegionId.Null;
238 
239  //iterate through points
240  for (int i = 0; i < rawVerts.Count; i++)
241  {
242  int x = rawVerts[i].X;
243  int y = rawVerts[i].Y;
244  int z = rawVerts[i].Z;
245 
246  if (x < lowerLeftX || (x == lowerLeftX && z < lowerLeftZ))
247  {
248  lowerLeftX = x;
249  lowerLeftY = y;
250  lowerLeftZ = z;
251  lowerLeftI = new RegionId(i);
252  }
253 
254  if (x > upperRightX || (x == upperRightX && z > upperRightZ))
255  {
256  upperRightX = x;
257  upperRightY = y;
258  upperRightZ = z;
259  upperRightI = new RegionId(i);
260  }
261  }
262 
263  //save the points
264  simplified.Add(new ContourVertex(lowerLeftX, lowerLeftY, lowerLeftZ, lowerLeftI));
265  simplified.Add(new ContourVertex(upperRightX, upperRightY, upperRightZ, upperRightI));
266  }
267 
268  //add points until all points are within error tolerance of simplified slope
269  int numPoints = rawVerts.Count;
270  for (int i = 0; i < simplified.Count;)
271  {
272  int ii = (i + 1) % simplified.Count;
273 
274  //obtain (x, z) coordinates, along with region id
275  int ax = simplified[i].X;
276  int az = simplified[i].Z;
277  int ai = (int)simplified[i].RegionId;
278 
279  int bx = simplified[ii].X;
280  int bz = simplified[ii].Z;
281  int bi = (int)simplified[ii].RegionId;
282 
283  float maxDeviation = 0;
284  int maxi = -1;
285  int ci, countIncrement, endi;
286 
287  //traverse segment in lexilogical order (try to go from smallest to largest coordinates?)
288  if (bx > ax || (bx == ax && bz > az))
289  {
290  countIncrement = 1;
291  ci = (int)(ai + countIncrement) % numPoints;
292  endi = (int)bi;
293  }
294  else
295  {
296  countIncrement = numPoints - 1;
297  ci = (int)(bi + countIncrement) % numPoints;
298  endi = (int)ai;
299  }
300 
301  //tessellate only outer edges or edges between areas
302  if (rawVerts[ci].RegionId.Id == 0 || RegionId.HasFlags(rawVerts[ci].RegionId, RegionFlags.AreaBorder))
303  {
304  //find the maximum deviation
305  while (ci != endi)
306  {
307  float deviation = Distance.PointToSegment2DSquared(rawVerts[ci].X, rawVerts[ci].Z, ax, az, bx, bz);
308 
309  if (deviation > maxDeviation)
310  {
311  maxDeviation = deviation;
312  maxi = ci;
313  }
314 
315  ci = (ci + countIncrement) % numPoints;
316  }
317  }
318 
319  //If max deviation is larger than accepted error, add new point
320  if (maxi != -1 && maxDeviation > (maxError * maxError))
321  {
322  simplified.Insert(i + 1, new ContourVertex(rawVerts[maxi], maxi));
323  }
324  else
325  {
326  i++;
327  }
328  }
329 
330  //split too long edges
331  if (maxEdgeLen > 0 && (tesselateAreaEdges || tesselateWallEdges))
332  {
333  for (int i = 0; i < simplified.Count;)
334  {
335  int ii = (i + 1) % simplified.Count;
336 
337  //get (x, z) coordinates along with region id
338  int ax = simplified[i].X;
339  int az = simplified[i].Z;
340  int ai = (int)simplified[i].RegionId;
341 
342  int bx = simplified[ii].X;
343  int bz = simplified[ii].Z;
344  int bi = (int)simplified[ii].RegionId;
345 
346  //find maximum deviation from segment
347  int maxi = -1;
348  int ci = (int)(ai + 1) % numPoints;
349 
350  //tessellate only outer edges or edges between areas
351  bool tess = false;
352 
353  //wall edges
354  if (tesselateWallEdges && rawVerts[ci].RegionId.Id == 0)
355  tess = true;
356 
357  //edges between areas
358  if (tesselateAreaEdges && RegionId.HasFlags(rawVerts[ci].RegionId, RegionFlags.AreaBorder))
359  tess = true;
360 
361  if (tess)
362  {
363  int dx = bx - ax;
364  int dz = bz - az;
365  if (dx * dx + dz * dz > maxEdgeLen * maxEdgeLen)
366  {
367  //round based on lexilogical direction (smallest to largest cooridinates, first by x.
368  //if x coordinates are equal, then compare z coordinates)
369  int n = bi < ai ? (bi + numPoints - ai) : (bi - ai);
370 
371  if (n > 1)
372  {
373  if (bx > ax || (bx == ax && bz > az))
374  maxi = (int)(ai + n / 2) % numPoints;
375  else
376  maxi = (int)(ai + (n + 1) / 2) % numPoints;
377  }
378  }
379  }
380 
381  //add new point
382  if (maxi != -1)
383  {
384  simplified.Insert(i + 1, new ContourVertex(rawVerts[maxi], maxi));
385  }
386  else
387  {
388  i++;
389  }
390  }
391  }
392 
393  for (int i = 0; i < simplified.Count; i++)
394  {
395  ContourVertex sv = simplified[i];
396 
397  //take edge vertex flag from current raw point and neighbor region from next raw point
398  int ai = ((int)sv.RegionId + 1) % numPoints;
399  RegionId bi = sv.RegionId;
400 
401  //save new region id
402  sv.RegionId = RegionId.FromRawBits(((int)rawVerts[ai].RegionId & (RegionId.MaskId | (int)RegionFlags.AreaBorder)) | ((int)rawVerts[(int)bi].RegionId & (int)RegionFlags.VertexBorder));
403 
404  simplified[i] = sv;
405  }
406  }
