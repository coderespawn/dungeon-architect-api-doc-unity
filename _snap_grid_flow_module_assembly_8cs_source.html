151  doorCell.nodeId = targetSide.connectionIndices[i].nodeId;

190  public static void Build(Vector3 inChunkSize, SgfModuleDatabaseItem inModuleInfo, out SgfModuleAssembly outAssembly)
191  {
192  var numChunks = inModuleInfo.NumChunks;
193  outAssembly = new SgfModuleAssembly();
194  outAssembly.Initialize(numChunks);
195 
196  Vector3 baseOffset = -inModuleInfo.ModuleBounds.min;
197  for (int connectionIdx = 0; connectionIdx < inModuleInfo.Connections.Length; connectionIdx++) {
198  var connectionInfo = inModuleInfo.Connections[connectionIdx];
199  Vector3 connectionLocation = Matrix.GetTranslation(ref connectionInfo.Transform) + baseOffset;
200  Vector3 localChunkPosition = MathUtils.Divide(connectionLocation, inChunkSize);
201  var p = localChunkPosition;
202 
203  var attachmentDistances = new List<FAssemblyAttachmentDistance>();
204  attachmentDistances.Add(new FAssemblyAttachmentDistance(EAssemblySide.Front, Mathf.Abs(p.z)));
205  attachmentDistances.Add(new FAssemblyAttachmentDistance(EAssemblySide.Left, Mathf.Abs(p.x - numChunks.x)));
206  attachmentDistances.Add(new FAssemblyAttachmentDistance(EAssemblySide.Back, Mathf.Abs(p.z - numChunks.z)));
207  attachmentDistances.Add(new FAssemblyAttachmentDistance(EAssemblySide.Right, Mathf.Abs(p.x)));
208  attachmentDistances.Add(new FAssemblyAttachmentDistance(EAssemblySide.Down, Mathf.Abs(p.y)));
209  attachmentDistances.Add(new FAssemblyAttachmentDistance(EAssemblySide.Top, Mathf.Abs(p.y - numChunks.y)));
210 
211  EAssemblySide bestSide = EAssemblySide.Unknown;
212  float bestDistance = float.MaxValue;
213  foreach (var attachDistInfo in attachmentDistances) {
214  if (attachDistInfo.Distance < bestDistance) {
215  bestDistance = attachDistInfo.Distance;
216  bestSide = attachDistInfo.Side;
217  }
218  }
219 
220  var connectionCell = new SgfModuleAssemblySideCell(connectionIdx);
221 
222  if (bestSide == EAssemblySide.Front) {
223  // Front
224  int cx = Mathf.FloorToInt(p.x);
225  int cy = Mathf.FloorToInt(p.y);
226  outAssembly.front.Set(cx, cy, connectionCell);
227  //Debug.Log("FRONT: Location (%.2f, %.2f)", P.X, P.Y);
228  }
229  else if (bestSide == EAssemblySide.Left) {
230  // Left
231  int cx = Mathf.FloorToInt(p.z);
232  int cy = Mathf.FloorToInt(p.y);
233  outAssembly.left.Set(cx, cy, connectionCell);
234  //Debug.Log("LEFT: Location (%.2f, %.2f)", P.X, P.Y);
235  }
236  else if (bestSide == EAssemblySide.Back) {
237  // Back
238  int cx = numChunks.x - 1 - Mathf.FloorToInt(p.x);
239  int cy = Mathf.FloorToInt(p.y);
240  outAssembly.back.Set(cx, cy, connectionCell);
241  //Debug.Log("BACK: Location (%.2f, %.2f)", P.X, P.Y);
242  }
243  else if (bestSide == EAssemblySide.Right) {
244  // Right
245  int cx = numChunks.z - 1 - Mathf.FloorToInt(p.z);
246  int cy = Mathf.FloorToInt(p.y);
247  outAssembly.right.Set(cx, cy, connectionCell);
248  //Debug.Log("RIGHT: Location (%.2f, %.2f)", P.X, P.Y);
249  }
250  else if (bestSide == EAssemblySide.Down) {
251  // Down
252  int cx = Mathf.FloorToInt(p.x);
253  int cy = Mathf.FloorToInt(p.z);
254  outAssembly.down.Set(cx, cy, connectionCell);

267  public static void Build(FlowLayoutGraphQuery inGraphQuery, FlowLayoutPathNodeGroup group, FFAGConstraintsLink[] incomingNodes, out SgfModuleAssembly outAssembly)
268  {
269  var minCoordF = new Vector3();
270  var maxCoordF = new Vector3();
271 
272  for (int i = 0; i < group.GroupEdgeNodes.Count; i++) {
273  var edgeNodeId = group.GroupEdgeNodes[i];
274  var edgeNode = inGraphQuery.GetNode(edgeNodeId);
275  if (edgeNode == null) {
276  edgeNode = inGraphQuery.GetSubNode(edgeNodeId);
277  }
278  Debug.Assert(edgeNode != null);
279 
280  if (i == 0) {
281  minCoordF = maxCoordF = edgeNode.coord;
282  }
283  else {
284  minCoordF = MathUtils.ComponentMin(minCoordF, edgeNode.coord);
285  maxCoordF = MathUtils.ComponentMax(maxCoordF, edgeNode.coord);
286  }
287  }
288 
289  var minCoord = MathUtils.RoundToVector3Int(minCoordF);
290  var maxCoord = MathUtils.RoundToVector3Int(maxCoordF);
291 
292  var numChunks = maxCoord - minCoord + new Vector3Int(1, 1, 1);
293  outAssembly = new SgfModuleAssembly();
294  outAssembly.Initialize(numChunks);
295 
296  foreach (var link in incomingNodes) {
297  if (link.IncomingNode == null) continue;
298  var c = MathUtils.RoundToVector3Int(link.Node.coord) - minCoord;
299  var ic = MathUtils.RoundToVector3Int(link.IncomingNode.coord) - minCoord;
300 
301  var cell = new SgfModuleAssemblySideCell();
302  cell.connectionIdx = SgfModuleAssemblySide.IndexValidUnknown;
303  cell.nodeId = (link.Node != null) ? link.Node.nodeId : DungeonUID.Empty;
304  cell.linkedNodeId = (link.IncomingNode != null) ? link.IncomingNode.nodeId : DungeonUID.Empty;
305 
306  if (c.z > ic.z) {
307  // Front
308  outAssembly.front.Set(c.x, c.y, cell);
309  }
310  else if (ic.x > c.x) {
311  // Left
312  outAssembly.left.Set(c.z, c.y, cell);
313  }
314  else if (ic.z > c.z) {
315  // Back
316  outAssembly.back.Set(numChunks.x - 1 - c.x, c.y, cell);
317  }
318  else if (c.x > ic.x) {
319  // Right
320  outAssembly.right.Set(numChunks.z - 1 - c.z, c.y, cell);
321  }
322  else if (c.y > ic.y) {
323  // Down
324  outAssembly.down.Set(c.x, c.z, cell);
325  }
326  else if (ic.y > c.y) {
327  // Top
328  outAssembly.top.Set(c.x, c.z, cell);
329  }
330  }
331  }
