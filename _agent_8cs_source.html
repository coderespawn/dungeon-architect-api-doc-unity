226  public void Integrate(float dt)
227  {
228  //fake dyanmic constraint
229  float maxDelta = Parameters.MaxAcceleration * dt;
230  Vector3 dv = NVel - Vel;
231  float ds = dv.Length();
232  if (ds > maxDelta)
233  dv = dv * (maxDelta / ds);
234  Vel = Vel + dv;
235 
236  //integrate
237  if (Vel.Length() > 0.0001f)
238  currentPos = currentPos + Vel * dt;
239  else
240  Vel = new Vector3(0, 0, 0);
241  }
