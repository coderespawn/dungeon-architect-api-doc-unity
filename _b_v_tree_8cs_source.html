39  public BVTree(PolyVertex[] verts, PolyMesh.Polygon[] polys, int nvp, float cellSize, float cellHeight)
40  {
41  nodes = new Node[polys.Length * 2];
42  var items = new List<Node>();
43 
44  for (int i = 0; i < polys.Length; i++)
45  {
46  PolyMesh.Polygon p = polys[i];
47 
48  Node temp;
49  temp.Index = i;
50  temp.Bounds.Min = temp.Bounds.Max = verts[p.Vertices[0]];
51 
52  for (int j = 1; j < nvp; j++)
53  {
54  int vi = p.Vertices[j];
55  if (vi == PolyMesh.NullId)
56  break;
57 
58  var v = verts[vi];
59  PolyVertex.ComponentMin(ref temp.Bounds.Min, ref v, out temp.Bounds.Min);
60  PolyVertex.ComponentMax(ref temp.Bounds.Max, ref v, out temp.Bounds.Max);
61  }
62 
63  temp.Bounds.Min.Y = (int)Math.Floor((float)temp.Bounds.Min.Y * cellHeight / cellSize);
64  temp.Bounds.Max.Y = (int)Math.Ceiling((float)temp.Bounds.Max.Y * cellHeight / cellSize);
65

67  }

147  private int Subdivide(List<Node> items, int minIndex, int maxIndex, int curNode)
148  {
149  int numIndex = maxIndex - minIndex;
150  int curIndex = curNode;
151 
152  int oldNode = curNode;
153  curNode++;
154 
155  //Check if the current node is a leaf node
156  if (numIndex == 1)
157  nodes[oldNode] = items[minIndex];
158  else
159  {
160  PolyBounds bounds;
161  CalcExtends(items, minIndex, maxIndex, out bounds);
162  nodes[oldNode].Bounds = bounds;
163 
164  int axis = LongestAxis((int)(bounds.Max.X - bounds.Min.X), (int)(bounds.Max.Y - bounds.Min.Y), (int)(bounds.Max.Z - bounds.Min.Z));
165 
166  switch (axis)
167  {
168  case 0:
169  items.Sort(minIndex, numIndex, XComparer);
170  break;
171  case 1:
172  items.Sort(minIndex, numIndex, YComparer);
173  break;
174  case 2:
175  items.Sort(minIndex, numIndex, ZComparer);
176  break;
177  default:
178  break;
179  }
180 
181  int splitIndex = minIndex + (numIndex / 2);
182 
183  curNode = Subdivide(items, minIndex, splitIndex, curNode);
184  curNode = Subdivide(items, splitIndex, maxIndex, curNode);
185 
186  int escapeIndex = curNode - curIndex;
187  nodes[oldNode].Index = -escapeIndex;
188  }
189 
190  return curNode;
191  }
