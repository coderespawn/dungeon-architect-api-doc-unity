156  protected void ProcessMarkerOverrideVolumes()
157  {
158  var dungeon = GetComponent<Dungeon>();
159  // Process the theme override volumes
160  var replacementVolumes = GameObject.FindObjectsOfType<MarkerReplaceVolume>();
161  foreach (var replacementVolume in replacementVolumes)
162  {
163  if (replacementVolume.dungeon == dungeon) {
164  // Fill up the prop sockets with the defined mesh data
165  for (int i = 0; i < markers.Count; i++) {
166  PropSocket socket = markers[i];
167  var socketPosition = Matrix.GetTranslation (ref socket.Transform);
168  if (replacementVolume.GetBounds ().Contains (socketPosition)) {
169  foreach (var replacementEntry in replacementVolume.replacements) {
170  if (socket.SocketType == replacementEntry.fromMarker) {
171  socket.SocketType = replacementEntry.toMarker;
172  }
173  }
174  }
175  }
176  }
177  }
178  }
