47  public Heightfield(BBox3 b, float cellSize, float cellHeight)
48  {
49  if (!BBox3.IsValid(ref bounds))
50  throw new ArgumentException("The bounds are considered invalid. See BBox3.IsValid for details.");
51 
52  if (cellSize <= 0)
53  throw new ArgumentOutOfRangeException("cellSize", "Cell size must be greater than 0.");
54 
55  if (cellHeight <= 0)
56  throw new ArgumentOutOfRangeException("cellHeight", "Cell height must be greater than 0.");
57 
58  this.cellSize = cellSize;
59  this.cellHeight = cellHeight;
60  this.bounds = b;
61 
62  //make sure the bbox contains all the possible voxels.
63  width = (int)Math.Ceiling((b.Max.X - b.Min.X) / cellSize);
64  height = (int)Math.Ceiling((b.Max.Y - b.Min.Y) / cellHeight);
65  length = (int)Math.Ceiling((b.Max.Z - b.Min.Z) / cellSize);
66 
67  bounds.Max.X = bounds.Min.X + width * cellSize;
68  bounds.Max.Y = bounds.Min.Y + height * cellHeight;
69  bounds.Max.Z = bounds.Min.Z + length * cellSize;
70 
71  cells = new Cell[width * length];
72  for (int i = 0; i < cells.Length; i++)
73  cells[i] = new Cell(height);
74  }

250  public void FilterLowHangingWalkableObstacles(int walkableClimb)
251  {
252  //Loop through every cell in the Heightfield
253  for (int i = 0; i < cells.Length; i++)
254  {
255  Cell c = cells[i];
256  List<Span> spans = c.MutableSpans;
257 
258  //store the first span's data as the "previous" data
259  Area prevArea = Area.Null;
260  bool prevWalkable = prevArea != Area.Null;
261  int prevMax = 0;
262 
263  //iterate over all the spans in the cell
264  for (int j = 0; j < spans.Count; j++)
265  {
266  Span s = spans[j];
267  bool walkable = s.Area != Area.Null;
268 
269  //if the current span isn't walkable but there's a walkable span right below it,
270  //mark this span as walkable too.
271  if (!walkable && prevWalkable)
272  {
273  if (Math.Abs(s.Maximum - prevMax) < walkableClimb)
274  s.Area = prevArea;
275  }
276 
277  //save changes back to the span list.
278  spans[j] = s;
279 
280  //set the previous data for the next iteration
281  prevArea = s.Area;
282  prevWalkable = walkable;
283  prevMax = s.Maximum;
284  }
285  }
286  }

293  public void FilterWalkableLowHeightSpans(int walkableHeight)
294  {
295  for (int i = 0; i < cells.Length; i++)
296  {
297  Cell c = cells[i];
298  List<Span> spans = c.MutableSpans;
299 
300  //Iterate over all spans
301  for (int j = 0; j < spans.Count - 1; j++)
302  {
303  Span currentSpan = spans[j];
304 
305  //too low, not enough space to walk through
306  if ((spans[j + 1].Minimum - currentSpan.Maximum) <= walkableHeight)
307  {
308  currentSpan.Area = Area.Null;
309  spans[j] = currentSpan;
310  }

321  public void FilterLedgeSpans(int walkableHeight, int walkableClimb)
322  {
323  //Mark border spans.
324  //Parallel.For(0, length, y =>
325  for (int y = 0; y < length; y++)
326  {
327  for (int x = 0; x < width; x++)
328  {
329  Cell c = cells[x + y * width];
330  List<Span> spans = c.MutableSpans;
331 
332  //Examine all the spans in each cell
333  for (int i = 0; i < spans.Count; i++)
334  {
335  Span currentSpan = spans[i];
336 
337  // Skip non walkable spans.
338  if (currentSpan.Area == Area.Null)
339  continue;
340 
341  int bottom = (int)currentSpan.Maximum;
342  int top = (i == spans.Count - 1) ? int.MaxValue : spans[i + 1].Minimum;
343 
344  // Find neighbours minimum height.
345  int minHeight = int.MaxValue;
346 
347  // Min and max height of accessible neighbours.
348  int accessibleMin = currentSpan.Maximum;
349  int accessibleMax = currentSpan.Maximum;
350 
351  for (var dir = Direction.West; dir <= Direction.South; dir++)
352  {
353  int dx = x + dir.GetHorizontalOffset();
354  int dy = y + dir.GetVerticalOffset();
355 
356  // Skip neighbours which are out of bounds.
357  if (dx < 0 || dy < 0 || dx >= width || dy >= length)
358  {
359  minHeight = Math.Min(minHeight, -walkableClimb - bottom);
360  continue;
361  }
362 
363  // From minus infinity to the first span.
364  Cell neighborCell = cells[dy * width + dx];
365  List<Span> neighborSpans = neighborCell.MutableSpans;
366  int neighborBottom = -walkableClimb;
367  int neighborTop = neighborSpans.Count > 0 ? neighborSpans[0].Minimum : int.MaxValue;
368 
369  // Skip neightbour if the gap between the spans is too small.
370  if (Math.Min(top, neighborTop) - Math.Max(bottom, neighborBottom) > walkableHeight)
371  minHeight = Math.Min(minHeight, neighborBottom - bottom);
372 
373  // Rest of the spans.
374  for (int j = 0; j < neighborSpans.Count; j++)
375  {
376  Span currentNeighborSpan = neighborSpans[j];
377 
378  neighborBottom = currentNeighborSpan.Maximum;
379  neighborTop = (j == neighborSpans.Count - 1) ? int.MaxValue : neighborSpans[j + 1].Minimum;
380 
381  // Skip neightbour if the gap between the spans is too small.
382  if (Math.Min(top, neighborTop) - Math.Max(bottom, neighborBottom) > walkableHeight)
383  {
384  minHeight = Math.Min(minHeight, neighborBottom - bottom);
385 
386  // Find min/max accessible neighbour height.
387  if (Math.Abs(neighborBottom - bottom) <= walkableClimb)
388  {
389  if (neighborBottom < accessibleMin) accessibleMin = neighborBottom;
390  if (neighborBottom > accessibleMax) accessibleMax = neighborBottom;
391  }
392  }
393  }
394  }
395 
396  // The current span is close to a ledge if the drop to any
397  // neighbour span is less than the walkableClimb.
398  if (minHeight < -walkableClimb)
399  currentSpan.Area = Area.Null;
400 
401  // If the difference between all neighbours is too large,
402  // we are at steep slope, mark the span as ledge.
403  if ((accessibleMax - accessibleMin) > walkableClimb)
404  currentSpan.Area = Area.Null;
405 
406  //save span data
407  spans[i] = currentSpan;
408  }
409  }
410  }
411  //});
412  }
