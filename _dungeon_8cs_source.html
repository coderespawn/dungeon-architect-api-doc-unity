387  public void AddPaintCell(IntVector location, bool automaticRebuild, IDungeonSceneObjectInstantiator sceneInstantiator) {
388  bool overlappingCell = false;
389  IntVector overlappingCellValue = new IntVector();
390  var toolOverlayData = ActiveModel.ToolData;
391  if (toolOverlayData == null) return;
392 
393  foreach (var cellData in toolOverlayData.PaintedCells) {
394  if (cellData.x == location.x && cellData.z == location.z) {
395  if (cellData.y != location.y) {
396  overlappingCell = true;
397  overlappingCellValue = cellData;
398  break;
399  }
400  else {
401  // Cell with this data already exists. Ignore the request
402  return;
403  }
404  }
405  }
406  if (overlappingCell) {
407  toolOverlayData.PaintedCells.Remove(overlappingCellValue);
408  }
409 
410  toolOverlayData.PaintedCells.Add(location);
411  if (automaticRebuild) {
412  Build(sceneInstantiator);
413  }
414  }
