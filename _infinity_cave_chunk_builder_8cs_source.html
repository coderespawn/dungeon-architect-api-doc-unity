53  var random = new System.Random(chunkSeed);
54  var sx = (cdx + 1) * w;
55  var sz = (cdz + 1) * h;
56  for (int x = sx; x < sx + w; x++)
57  {
58  for (int z = sz; z < sz + h; z++)
59  {
60  world[x, z] = random.NextFloat() < 0.5f;
61  }
62  }
63  }

72  for (int x = 0; x < w; x++)

75  {
76  bool occupied = world[w + x, h + y];
77  var cellType = occupied ? MazeTileState.Rock : MazeTileState.Empty;
78  if (occupied)
79  {

106  var h = world.GetLength(1);
107 
108  var result = new bool[w, h];
109  for (int x = 0; x < w; x++)
110  {
111  for (int y = 0; y < h; y++)
112  {
113  var rocks = 0;
114  for (int dx = -1; dx <= 1; dx++)
115  {
116  for (int dy = -1; dy <= 1; dy++)
117  {
118  var cx = Mathf.Clamp(x + dx, 0, w - 1);
119  var cy = Mathf.Clamp(y + dy, 0, w - 1);
120  if (world[cx, cy])
