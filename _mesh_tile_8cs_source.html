138  public void ClosestPointOnPolyBoundary(Poly poly, Vector3 pos, out Vector3 closest)
139  {
140  //Clamp point to be inside the polygon
141  Vector3[] verts = new Vector3[PathfindingCommon.VERTS_PER_POLYGON];
142  float[] edgeDistance = new float[PathfindingCommon.VERTS_PER_POLYGON];
143  float[] edgeT = new float[PathfindingCommon.VERTS_PER_POLYGON];
144  int numPolyVerts = poly.VertCount;
145  for (int i = 0; i < numPolyVerts; i++)
146  verts[i] = Verts[poly.Verts[i]];
147 
148  bool inside = Distance.PointToPolygonEdgeSquared(pos, verts, numPolyVerts, edgeDistance, edgeT);
149  if (inside)
150  {
151  //Point is inside the polygon
152  closest = pos;
153  }
154  else
155  {
156  //Point is outside the polygon
157  //Clamp to nearest edge
158  float minDistance = float.MaxValue;
159  int minIndex = -1;
160  for (int i = 0; i < numPolyVerts; i++)
161  {
162  if (edgeDistance[i] < minDistance)
163  {
164  minDistance = edgeDistance[i];
165  minIndex = i;
166  }
167  }
168 
169  Vector3 va = verts[minIndex];
170  Vector3 vb = verts[(minIndex + 1) % numPolyVerts];
171  closest = Vector3.Lerp(va, vb, edgeT[minIndex]);
172  }
173  }

182  public bool ClosestHeight(int indexPoly, Vector3 pos, out float h)
183  {
184  Poly poly = Polys[indexPoly];
185  PolyMeshDetail.MeshData pd = DetailMeshes[indexPoly];
186 
187  //find height at the location
188  for (int j = 0; j < DetailMeshes[indexPoly].TriangleCount; j++)
189  {
190  PolyMeshDetail.TriangleData t = DetailTris[pd.TriangleIndex + j];
191  Vector3[] v = new Vector3[3];
192 
193  for (int k = 0; k < 3; k++)
194  {
195  if (t[k] < poly.VertCount)
196  v[k] = Verts[poly.Verts[t[k]]];
197  else
198  v[k] = DetailVerts[pd.VertexIndex + (t[k] - poly.VertCount)];
199  }
200 
201  if (Distance.PointToTriangle(pos, v[0], v[1], v[2], out h))
202  return true;
203  }
204 
205  h = float.MaxValue;
206  return false;
207  }
