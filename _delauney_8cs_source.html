66  public static List<Geometry.Triangle> Triangulate(List<Triangulator.Geometry.Point> Vertex)
67  {
68  int nv = Vertex.Count;
69  if (nv < 3)
70  throw new ArgumentException("Need at least three vertices for triangulation");
71 
72  int trimax = 4 * nv;
73 
74  // Find the maximum and minimum vertex bounds.
75  // This is to allow calculation of the bounding supertriangle
76  double xmin = Vertex[0].X;
77  double ymin = Vertex[0].Y;
78  double xmax = xmin;
79  double ymax = ymin;
80  for (int i = 1; i < nv; i++)
81  {
82  if (Vertex[i].X < xmin) xmin = Vertex[i].X;
83  if (Vertex[i].X > xmax) xmax = Vertex[i].X;
84  if (Vertex[i].Y < ymin) ymin = Vertex[i].Y;
85  if (Vertex[i].Y > ymax) ymax = Vertex[i].Y;
86  }
87 
88  double dx = xmax - xmin;
89  double dy = ymax - ymin;
90  double dmax = (dx > dy) ? dx : dy;
91 
92  double xmid = (xmax + xmin) * 0.5;
93  double ymid = (ymax + ymin) * 0.5;
94 
95 
96  // Set up the supertriangle
97  // This is a triangle which encompasses all the sample points.
98  // The supertriangle coordinates are added to the end of the
99  // vertex list. The supertriangle is the first triangle in
100  // the triangle list.
101  Vertex.Add(new Triangulator.Geometry.Point((xmid - 2 * dmax), (ymid - dmax)));
102  Vertex.Add(new Triangulator.Geometry.Point(xmid, (ymid + 2 * dmax)));
103  Vertex.Add(new Triangulator.Geometry.Point((xmid + 2 * dmax), (ymid - dmax)));
104  List<Geometry.Triangle> Triangle = new List<Geometry.Triangle>();
105  Triangle.Add(new Geometry.Triangle(nv, nv + 1, nv + 2)); //SuperTriangle placed at index 0
106 
107  // Include each point one at a time into the existing mesh
108  for (int i = 0; i < nv; i++)
109  {
110  List<Geometry.Edge> Edges = new List<Geometry.Edge>(); //[trimax * 3];
111  // Set up the edge buffer.
112  // If the point (Vertex(i).x,Vertex(i).y) lies inside the circumcircle then the
113  // three edges of that triangle are added to the edge buffer and the triangle is removed from list.
114  for (int j = 0; j < Triangle.Count; j++)
115  {
116  if (InCircle(Vertex[i], Vertex[Triangle[j].p1], Vertex[Triangle[j].p2], Vertex[Triangle[j].p3]))
117  {
118  Edges.Add(new Geometry.Edge(Triangle[j].p1, Triangle[j].p2));
119  Edges.Add(new Geometry.Edge(Triangle[j].p2, Triangle[j].p3));
120  Edges.Add(new Geometry.Edge(Triangle[j].p3, Triangle[j].p1));
121  Triangle.RemoveAt(j);
122  j--;
123  }
124  }
125  if (i >= nv) continue; //In case we the last duplicate point we removed was the last in the array
126 
127  // Remove duplicate edges
128  // Note: if all triangles are specified anticlockwise then all
129  // interior edges are opposite pointing in direction.
130  for (int j = Edges.Count - 2; j >= 0; j--)
131  {
132  for (int k = Edges.Count - 1; k >= j + 1; k--)
133  {
134  if (Edges[j].Equals(Edges[k]))
135  {
136  Edges.RemoveAt(k);
137  Edges.RemoveAt(j);
138  k--;
139  continue;
140  }
141  }
142  }
143  // Form new triangles for the current point
144  // Skipping over any tagged edges.
145  // All edges are arranged in clockwise order.
146  for (int j = 0; j < Edges.Count; j++)
147  {
148  if (Triangle.Count >= trimax)
149  throw new ApplicationException("Exceeded maximum edges");
150  Triangle.Add(new Geometry.Triangle(Edges[j].p1, Edges[j].p2, i));
151  }
152  Edges.Clear();
153  Edges = null;
154  }
155  // Remove triangles with supertriangle vertices
156  // These are triangles which have a vertex number greater than nv
157  for (int i = Triangle.Count - 1; i >= 0; i--)
158  {
159  if (Triangle[i].p1 >= nv || Triangle[i].p2 >= nv || Triangle[i].p3 >= nv)
160  Triangle.RemoveAt(i);
161  }
162  //Remove SuperTriangle vertices
163  Vertex.RemoveAt(Vertex.Count - 1);
164  Vertex.RemoveAt(Vertex.Count - 1);
165  Vertex.RemoveAt(Vertex.Count - 1);
166  Triangle.TrimExcess();
167  return Triangle;
168  }

181  private static bool InCircle(Geometry.Point p, Geometry.Point p1, Geometry.Point p2, Geometry.Point p3)
182  {
183  //Return TRUE if the point (xp,yp) lies inside the circumcircle
184  //made up by points (x1,y1) (x2,y2) (x3,y3)
185  //NOTE: A point on the edge is inside the circumcircle
186 
187  if (System.Math.Abs(p1.Y - p2.Y) <= double.Epsilon && System.Math.Abs(p2.Y - p3.Y) <= double.Epsilon)
188  {
189  //INCIRCUM - F - Points are coincident !!
190  return false;
191  }
192 
193  double m1, m2;
194  double mx1, mx2;
195  double my1, my2;
196  double xc, yc;
197 
198  if (System.Math.Abs(p2.Y - p1.Y) <= double.Epsilon)
199  {
200  m2 = -(p3.X - p2.X) / (p3.Y - p2.Y);
201  mx2 = (p2.X + p3.X) * 0.5;
202  my2 = (p2.Y + p3.Y) * 0.5;
203  //Calculate CircumCircle center (xc,yc)
204  xc = (p2.X + p1.X) * 0.5;
205  yc = m2 * (xc - mx2) + my2;
206  }
207  else if (System.Math.Abs(p3.Y - p2.Y) <= double.Epsilon)
208  {
209  m1 = -(p2.X - p1.X) / (p2.Y - p1.Y);
210  mx1 = (p1.X + p2.X) * 0.5;
211  my1 = (p1.Y + p2.Y) * 0.5;
212  //Calculate CircumCircle center (xc,yc)
213  xc = (p3.X + p2.X) * 0.5;
214  yc = m1 * (xc - mx1) + my1;
215  }
216  else
217  {
218  m1 = -(p2.X - p1.X) / (p2.Y - p1.Y);
219  m2 = -(p3.X - p2.X) / (p3.Y - p2.Y);
220  mx1 = (p1.X + p2.X) * 0.5;
221  mx2 = (p2.X + p3.X) * 0.5;
222  my1 = (p1.Y + p2.Y) * 0.5;
223  my2 = (p2.Y + p3.Y) * 0.5;
224  //Calculate CircumCircle center (xc,yc)
225  xc = (m1 * mx1 - m2 * mx2 + my2 - my1) / (m1 - m2);
226  yc = m1 * (xc - mx1) + my1;
227  }
228 
229  double dx = p2.X - xc;
230  double dy = p2.Y - yc;
231  double rsqr = dx * dx + dy * dy;
232  //double r = Math.Sqrt(rsqr); //Circumcircle radius
233  dx = p.X - xc;
234  dy = p.Y - yc;
235  double drsqr = dx * dx + dy * dy;
236 
237  return (drsqr <= rsqr);
238  }
239  }
