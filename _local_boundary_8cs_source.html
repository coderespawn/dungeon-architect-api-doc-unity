105  public void AddSegment(float dist, Segment s)
106  {
107  //insert neighbour based on distance
108  int segPos = 0;
109  if (segCount == 0)
110  {
111  segPos = 0;
112  }
113  else if (dist >= segs[segCount - 1].Dist)
114  {
115  //further than the last segment, skip
116  if (segCount >= MaxLocalSegs)
117  return;
118 
119  //last, trivial accept
120  segPos = segCount;
121  }
122  else
123  {
124  //insert inbetween
125  int i;
126  for (i = 0; i < segCount; i++)
127  if (dist <= segs[i].Dist)
128  break;
129  int tgt = i + 1;
130  int n = Math.Min(segCount - i, MaxLocalSegs - tgt);
131  if (n > 0)
132  {
133  for (int j = 0; j < n; j++)
134  segs[tgt + j] = segs[i + j];
135  }
136 
137  segPos = i;
138  }
139 
140  segs[segPos].Dist = dist;
141  segs[segPos].Start = s.Start;
142  segs[segPos].End = s.End;
143 
144  if (segCount < MaxLocalSegs)
145  segCount++;
146  }

155  public void Update(int reference, Vector3 pos, float collisionQueryRange, NavMeshQuery navquery)
156  {
157  const int MAX_SEGS_PER_POLY = PathfindingCommon.VERTS_PER_POLYGON;
158 
159  if (reference == 0)
160  {
161  this.center = new Vector3(float.MaxValue, float.MaxValue, float.MaxValue);
162  this.segCount = 0;
163  this.numPolys = 0;
164  return;
165  }
166 
167  this.center = pos;
168 
169  //first query non-overlapping polygons
170  int[] tempArray = new int[polys.Length];
171  navquery.FindLocalNeighbourhood(new NavPoint(reference, pos), collisionQueryRange, polys, tempArray, ref numPolys, MaxLocalPolys);
172 
173  //secondly, store all polygon edges
174  this.segCount = 0;
175  Segment[] segs = new Segment[MAX_SEGS_PER_POLY];
176  int numSegs = 0;
177  for (int j = 0; j < numPolys; j++)
178  {
179  tempArray = new int[segs.Length];
180  navquery.GetPolyWallSegments(polys[j], segs, tempArray, ref numSegs, MAX_SEGS_PER_POLY);
181  for (int k = 0; k < numSegs; k++)
182  {
183  //skip too distant segments
184  float tseg;
185  float distSqr = Distance.PointToSegment2DSquared(ref pos, ref segs[k].Start, ref segs[k].End, out tseg);
186  if (distSqr > collisionQueryRange * collisionQueryRange)
187  continue;
188  AddSegment(distSqr, segs[k]);
189  }
190  }
191  }
