78  void ShowTabConstraintRules()
79  {
80  EmitAddConstraintWidget();
81 
82  var ruleNode = target as SCRuleNode;
83  EmitEvaluationModeWidget(ruleNode);
84 
85  var constraints = ruleNode.constraints.Where(c => c != null);
86  foreach (var constraint in constraints)
87  {
88  if (!uiState.constraintStates.ContainsKey(constraint))
89  {
90  var state = new RuleNodeEditorUIState.ConstraintUIState();
91  state.foldout = true;
92  state.enabled = constraint.enabled;
93  uiState.constraintStates.Add(constraint, state);
94  }
95 
96  var constraintUI = uiState.constraintStates[constraint];
97  EmitConstraintScript(constraint, constraintUI);
98  }
99 
100  if (constraints.Count() == 0)
101  {

128  void AddConstraintRule(System.Type constraintType)
129  {
130  var constraint = CreateInstance(constraintType) as ConstraintRule;
131  constraint.ruleName = ConstraintRule.GetScriptName(constraintType);
132 
133  var ruleNode = target as SCRuleNode;
134  var constraintList = new List<ConstraintRule>(ruleNode.constraints);
135  constraintList.Add(constraint);
136  ruleNode.constraints = constraintList.ToArray();
137 
138  // Add this constraint to the theme graph's asset file so it is serialized and not garbage collected
139  AssetDatabase.AddObjectToAsset(constraint, GetAssetObject());
140  }

152  void DestroyConstraintRule(ConstraintRule rule)
153  {
154  var ruleNode = target as SCRuleNode;
155  var constraints = new List<ConstraintRule>(ruleNode.constraints);
156 
157  Undo.RegisterCompleteObjectUndo(GetOwningAsset(), "Delete Node Spatial Constraint");
158  Undo.RecordObject(ruleNode, "Delete Constraint");
159 
160  constraints.Remove(rule);
161  Undo.DestroyObjectImmediate(rule);
162 
163  ruleNode.constraints = constraints.ToArray();
164  }

166  void EmitConstraintScript(ConstraintRule constraint, RuleNodeEditorUIState.ConstraintUIState constraintUI)
167  {
