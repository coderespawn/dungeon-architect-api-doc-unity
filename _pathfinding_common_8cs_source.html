38  public static void RandomPointInConvexPoly(Vector3[] pts, int npts, float[] areas, float s, float t, out Vector3 pt)
39  {
40  //Calculate triangle areas
41  float areaSum = 0.0f;
42  float area;
43  for (int i = 2; i < npts; i++)
44  {
45  Triangle3.Area2D(ref pts[0], ref pts[i - 1], ref pts[i], out area);
46  areaSum += Math.Max(0.001f, area);
47  areas[i] = area;
48  }
49 
50  //Find sub triangle weighted by area
51  float threshold = s * areaSum;
52  float accumulatedArea = 0.0f;
53  float u = 0.0f;
54  int triangleVertex = 0;
55  for (int i = 2; i < npts; i++)
56  {
57  float currentArea = areas[i];
58  if (threshold >= accumulatedArea && threshold < (accumulatedArea + currentArea))
59  {
60  u = (threshold - accumulatedArea) / currentArea;
61  triangleVertex = i;
62  break;
63  }
64 
65  accumulatedArea += currentArea;
66  }
67 
68  float v = (float)Math.Sqrt(t);
69 
70  float a = 1 - v;
71  float b = (1 - u) * v;
72  float c = u * v;
73  Vector3 pointA = pts[0];
74  Vector3 pointB = pts[triangleVertex - 1];
75  Vector3 pointC = pts[triangleVertex];
76 
77  pt = a * pointA + b * pointB + c * pointC;
78  }
