49  public bool Generate(FlowTilemapCell cell, FlowTilemapCell incomingCell, System.Random random, ref float overlayValue)
50  {
51  var cellCoord = incomingCell.TileCoord;
52  var position = cellCoord.ToVector2() * noiseSettings.noiseFrequency;
53  var n = noiseTable.GetNoiseFBM(position, noiseSettings.noiseOctaves);
54  if (noiseSettings.noiseValuePower > 0.0f)
55  {
56  n = Mathf.Pow(n, noiseSettings.noiseValuePower);
57  }
58 
59  n = noiseSettings.noiseMinValue + (noiseSettings.noiseMaxValue - noiseSettings.noiseMinValue) * n;
60 
61  if (n > noiseSettings.noiseThreshold)
62  {
63  var distanceFromMainPath = incomingCell.DistanceFromMainPath;
64  float noiseFactor = (n - noiseSettings.noiseThreshold) / (1.0f - noiseSettings.noiseThreshold);
65  bool insertOverlay = (noiseFactor * distanceFromMainPath > noiseSettings.minDistFromMainPath);
66 
67  if (insertOverlay)
68  {
69  overlayValue = n;
70  return true;
71  }
72  }
73  return false;
74  }
