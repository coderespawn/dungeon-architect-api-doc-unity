38  // Draw the domain, if we are snapped
39  if (domainNode.IsSnapped && !domainNode.Dragging)
40  {
41  const float DomainSizeHi = 0.75f;
42  const float DomainSizeLo = 0.15f;
43  var domainRectSize = bounds.size;
44  if (domainNode.RuleDomain == SCRuleNodeDomain.Corner)
45  {
46  domainRectSize = Vector2.one * (SCBaseDomainNode.TileSize * DomainSizeLo);
47  }
48  else if (domainNode.RuleDomain == SCRuleNodeDomain.Tile)
49  {
50  domainRectSize = Vector2.one * (SCBaseDomainNode.TileSize * DomainSizeHi);
51  }
52  else if (domainNode.RuleDomain == SCRuleNodeDomain.Edge)
53  {
54  var coords = domainNode.GetHalfGridLogicalCoords();
55  var localSize = (coords.x == 1) ? new Vector2(DomainSizeHi, DomainSizeLo) : new Vector2(DomainSizeLo, DomainSizeHi);
56  domainRectSize = localSize * SCBaseDomainNode.TileSize;
57  }
58  domainRectSize /= camera.ZoomLevel;
59  var domainBounds = new Rect(bounds.center - domainRectSize / 2.0f, domainRectSize);
60  var domainColor = color;
61  domainColor.a *= 0.5f;
62  renderer.color = domainColor;
63  renderer.DrawRect(domainBounds, domainColor);
64  }
65  guiState.Restore();
