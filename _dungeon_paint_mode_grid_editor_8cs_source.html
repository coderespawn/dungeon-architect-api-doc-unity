21  Vector3 SnapToGrid(Vector3 value)
22  {
23  var result = value;
24  var mode = target as DungeonPaintModeGrid;
25  var config = mode.GetDungeonConfig() as GridDungeonConfig;
26  if (config == null)
27  {
28  return result;
29  }
30  var gridSize = config.GridCellSize;
31  result.x = Mathf.FloorToInt(value.x / gridSize.x) * gridSize.x;
32  result.y = mode.cursorLogicalHeight * gridSize.y;
33  result.z = Mathf.FloorToInt(value.z / gridSize.z) * gridSize.z;
34  return result;
35  }

47  void UpdateCursorPosition(DungeonPaintModeGrid mode) {
48  float distance;
49  var e = Event.current;
50  Plane plane;
51  if (mode.mode2D) {
52  var planePoint = Vector3.zero;
53  plane = new Plane(new Vector3(0, 0, 1), planePoint);
54  } else {
55  var cursorHeight = mode.GetCursorHeight();
56  var planePoint = new Vector3(0, cursorHeight, 0);
57  plane = new Plane(Vector3.up, planePoint);
58  }
59 
60  var ray = HandleUtility.GUIPointToWorldRay(e.mousePosition);
61  if (plane.Raycast(ray, out distance))
62  {
63  var hitpoint = ray.GetPoint(distance);
64  if (mode.mode2D) {
65  hitpoint = DMathUtils.FlipYZ(hitpoint);
66  }
67  hitpoint = SnapToGrid(hitpoint);
68  cursorPosition = hitpoint;
69 
70  if (!mode.mode2D) {
71  if (e.type == EventType.ScrollWheel)
72  {
73  var delta = (int)Mathf.Sign(e.delta.y);
74  mode.SetElevationDelta(delta);
75  }
76  }
77  }

128  void ProcessPaintRequest(Event e)
129  {
130  var mode = target as DungeonPaintModeGrid;
131  var config = mode.GetDungeonConfig() as GridDungeonConfig;
132  var gridSize = config.GridCellSize;
133  var cursorGridPosition = new IntVector(
134  Mathf.RoundToInt(cursorPosition.x / gridSize.x),
135  Mathf.RoundToInt(cursorPosition.y / gridSize.y),
136  Mathf.RoundToInt(cursorPosition.z / gridSize.z));
137  var dungeon = mode.GetDungeon();
138 
139  var cursorSize = Mathf.Max (1, mode.brushSize);
140  for (int dx = 0; dx < cursorSize; dx++) {
141  for (int dz = 0; dz < cursorSize; dz++) {
142  var position = cursorGridPosition + new IntVector(dx, 0, dz);
143  var toolData = dungeon.ActiveModel.ToolData;
144  if (modeDelete)
145  {
146  dungeon.RemovePaintCell(position, true, sceneObjectInstantiator);
147  if (toolData) {
148  Undo.RecordObject(toolData, "Delete Painted Cell");
149  }
150  }
151  else
152  {
153  dungeon.AddPaintCell(position, true, sceneObjectInstantiator);
154  if (toolData)
155  {
156  Undo.RecordObject(toolData, "Add Painted Cell");
157  }
158  }
159  }
160  }
161 
162  EditorUtility.SetDirty(dungeon.gameObject);
163  }
