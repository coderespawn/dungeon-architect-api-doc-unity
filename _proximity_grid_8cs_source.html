91  public void AddItem(T value, float minX, float minY, float maxX, float maxY)

132  public int QueryItems(float minX, float minY, float maxX, float maxY, T[] values, int maxVals)
133  {

137  int invMaxY = (int)Math.Floor(maxY * invCellSize);

139  int n = 0;
140 
141  for (int y = invMinY; y <= invMaxY; y++)
142  {
143  for (int x = invMinX; x <= invMaxX; x++)
144  {
145  int h = HashPos2(x, y, buckets.Length);
146  int idx = buckets[h];
147

149  {

151  {

155  i++;

163  }

167  }
