54  public void Update(int maxIters)
55  {
56  const int MAX_KEEP_ALIVE = 2; //in update ticks
57 
58  //update path request until there is nothing left to update
59  //or up to maxIters pathfinder iterations have been consumed
60  int iterCount = maxIters;
61 
62  for (int i = 0; i < MaxQueue; i++)
63  {
64  PathQuery q = queue[queueHead % MaxQueue];
65 
66  //skip inactive requests
67  if (q.Reference == Invalid)
68  {
69  queueHead++;
70  continue;
71  }
72 
73  //handle completed request
74  if (q.status == Status.Success || q.status == Status.Failure)
75  {
76  q.KeepAlive++;
77  if (q.KeepAlive > MAX_KEEP_ALIVE)
78  {
79  q.Reference = Invalid;
80  q.status = 0;
81  }
82 
83  queueHead++;
84  continue;
85  }
86 
87  //handle query start
88  if (q.status == 0)
89  {
90  q.status = navquery.InitSlicedFindPath(new NavPoint(q.StartRef, q.StartPos), new NavPoint(q.EndRef, q.EndPos)).ToStatus();
91  }
92 
93  //handle query in progress
94  if (q.status == Status.InProgress)
95  {
96  int iters = 0;
97  q.status = navquery.UpdateSlicedFindPath(iterCount, ref iters).ToStatus();
98 
99  iterCount -= iters;
100  }
101 
102  if (q.status == Status.Success)
103  {
104  q.status = navquery.FinalizeSlicedFindPath(q.Path, ref q.PathCount, maxPathSize).ToStatus();
105  }
106 
107  if (iterCount <= 0)
108  break;
109 
110  queueHead++;
111  }
112  }

122  public int Request(int startRef, int endRef, Vector3 startPos, Vector3 endPos)
123  {
124  //find empty slot
125  int slot = -1;
126  for (int i = 0; i < MaxQueue; i++)
127  {
128  if (queue[i].Reference == Invalid)
129  {
130  slot = i;
131  break;
132  }
133  }
134 
135  //could not find slot
136  if (slot == -1)
137  return Invalid;
138 
139  int reference = nextHandle++;
140  if (nextHandle == Invalid) nextHandle++;
141 
142  PathQuery q = queue[slot];
143  q.Reference = reference;
144  q.StartPos = startPos;
145  q.StartRef = startRef;
146  q.EndPos = endPos;
147  q.EndRef = endRef;
148 
149  q.status = 0;
150  q.PathCount = 0;
151  q.KeepAlive = 0;
152 
153  queue[slot] = q;
154 
155  return reference;
156  }

174  public bool GetPathResult(int reference, int[] path, ref int pathSize, int maxPath)
175  {
176  for (int i = 0; i < MaxQueue; i++)
177  {
178  if (queue[i].Reference == reference)
179  {
180  PathQuery q = queue[i];
181 
182  //free request for reuse
183  q.Reference = Invalid;
184  q.status = 0;
185 
186  //copy path
187  int n = Math.Min(q.PathCount, maxPath);
188  q.Path.CopyTo(path, 0);
189  pathSize = n;
190 
191  queue[i] = q;
192 
193  return true;
194  }
195  }
196 
197  return false;
