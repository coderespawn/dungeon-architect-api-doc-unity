49  public CompactHeightfield(Heightfield field, int walkableHeight, int walkableClimb)
50  {
51  this.bounds = field.Bounds;
52  this.width = field.Width;
53  this.height = field.Height;
54  this.length = field.Length;
55  this.cellSize = field.CellSizeXZ;
56  this.cellHeight = field.CellHeight;
57 
58  int spanCount = field.SpanCount;
59  cells = new CompactCell[width * length];
60  spans = new CompactSpan[spanCount];
61  areas = new Area[spanCount];
62 
63  //iterate over the Heightfield's cells
64  int spanIndex = 0;
65  for (int i = 0; i < cells.Length; i++)
66  {
67  //get the heightfield span list, skip if empty
68  var fs = field[i].Spans;
69  if (fs.Count == 0)
70  continue;
71 
72  CompactCell c = new CompactCell(spanIndex, 0);
73 
74  //convert the closed spans to open spans
75  int lastInd = fs.Count - 1;
76  for (int j = 0; j < lastInd; j++)
77  {
78  var s = fs[j];
79  if (s.Area.IsWalkable)
80  {
81  CompactSpan.FromMinMax(s.Maximum, fs[j + 1].Minimum, out spans[spanIndex]);
82  areas[spanIndex] = s.Area;
83  spanIndex++;
84  c.Count++;
85  }
86  }
87 
88  //the last closed span that has an "infinite" height
89  var lastS = fs[lastInd];
90  if (lastS.Area.IsWalkable)
91  {
92  spans[spanIndex] = new CompactSpan(fs[lastInd].Maximum, int.MaxValue);
93  areas[spanIndex] = lastS.Area;
94  spanIndex++;
95  c.Count++;
96  }
97 
98  cells[i] = c;
99  }
100 
101  //set neighbor connections
102  for (int z = 0; z < length; z++)
103  {
104  for (int x = 0; x < width; x++)
105  {
106  CompactCell c = cells[z * width + x];
107  for (int i = c.StartIndex, end = c.StartIndex + c.Count; i < end; i++)
108  {
109  CompactSpan s = spans[i];
110 
111  for (var dir = Direction.West; dir <= Direction.South; dir++)
112  {
113  int dx = x + dir.GetHorizontalOffset();
114  int dz = z + dir.GetVerticalOffset();
115 
116  if (dx < 0 || dz < 0 || dx >= width || dz >= length)
117  continue;
118 
119  CompactCell dc = cells[dz * width + dx];
120  for (int j = dc.StartIndex, cellEnd = dc.StartIndex + dc.Count; j < cellEnd; j++)
121  {
122  CompactSpan ds = spans[j];
123 
124  int overlapBottom, overlapTop;
125  CompactSpan.OverlapMin(ref s, ref ds, out overlapBottom);
126  CompactSpan.OverlapMax(ref s, ref ds, out overlapTop);
127 
128  //Make sure that the agent can walk to the next span and that the span isn't a huge drop or climb
129  if ((overlapTop - overlapBottom) >= walkableHeight && Math.Abs(ds.Minimum - s.Minimum) <= walkableClimb)
130  {
131  int con = j - dc.StartIndex;
132  CompactSpan.SetConnection(dir, con, ref spans[i]);
133  break;
134  }
135  }
136  }
137  }
138  }
139  }
140  }

341  public void BuildDistanceField()
342  {
343  if (distances == null)
344  distances = new int[spans.Length];
345 
346  //fill up all the values in src
347  CalculateDistanceField(distances);
348 
349  //blur the distances
350  BoxBlur(distances, 1, null);
351 
352  //find the maximum distance
353  this.maxDistance = 0;
354  for (int i = 0; i < distances.Length; i++)
355  this.maxDistance = Math.Max(distances[i], this.maxDistance);
356  }

366  public void Erode(int radius)
367  {
368  radius *= 2;
369 
370  //get a distance field
371  int[] dists = new int[spans.Length];
372  CalculateDistanceField(dists);
373 
374  //erode close-to-null areas to null areas.
375  for (int i = 0; i < spans.Length; i++)
376  if (dists[i] < radius)
377  areas[i] = Area.Null;
378 
379  //marking areas as null changes the distance field, so recalculate it.
380  if (distances != null)
381  BuildDistanceField();
382  }

390  public void BuildRegions(int borderSize, int minRegionArea, int mergeRegionArea)
391  {
392  if (distances == null)
393  BuildDistanceField();
394 
395  const int LogStackCount = 3;
396  const int StackCount = 1 << LogStackCount;
397  List<CompactSpanReference>[] stacks = new List<CompactSpanReference>[StackCount];
398  for (int i = 0; i < stacks.Length; i++)
399  stacks[i] = new List<CompactSpanReference>(1024);
400 
401  RegionId[] regions = new RegionId[spans.Length];
402  int[] floodDistances = new int[spans.Length];
403 
404  RegionId[] regionBuffer = new RegionId[spans.Length];
405  int[] distanceBuffer = new int[spans.Length];
406 
407  int regionIndex = 1;
408  int level = ((maxDistance + 1) / 2) * 2;
409 
410  const int ExpandIters = 8;
411 
412  if (borderSize > 0)
413  {
414  //make sure border doesn't overflow
415  int borderWidth = Math.Min(width, borderSize);
416  int borderHeight = Math.Min(length, borderSize);
417 
418  //fill regions
419  FillRectangleRegion(regions, new RegionId(regionIndex++, RegionFlags.Border), 0, borderWidth, 0, length);
420  FillRectangleRegion(regions, new RegionId(regionIndex++, RegionFlags.Border), width - borderWidth, width, 0, length);
421  FillRectangleRegion(regions, new RegionId(regionIndex++, RegionFlags.Border), 0, width, 0, borderHeight);
422  FillRectangleRegion(regions, new RegionId(regionIndex++, RegionFlags.Border), 0, width, length - borderHeight, length);
423 
424  this.borderSize = borderSize;
425  }
426 
427  int stackId = -1;
428  while (level > 0)
429  {
430  level = level >= 2 ? level - 2 : 0;
431  stackId = (stackId + 1) & (StackCount - 1);
432 
433  if (stackId == 0)
434  SortCellsByLevel(regions, stacks, level, StackCount, 1);
435  else
436  AppendStacks(stacks[stackId - 1], stacks[stackId], regions);
437 
438  //expand current regions until no new empty connected cells found
439  ExpandRegions(regions, floodDistances, ExpandIters, level, stacks[stackId], regionBuffer, distanceBuffer);
440 
441  //mark new regions with ids
442  for (int j = 0; j < stacks[stackId].Count; j++)
443  {
444  var spanRef = stacks[stackId][j];
445  if (spanRef.Index >= 0 && regions[spanRef.Index] == 0)
446  if (FloodRegion(regions, floodDistances, regionIndex, level, ref spanRef))
447  regionIndex++;
448  }
449  }
450 
451  //expand current regions until no new empty connected cells found
452  ExpandRegions(regions, floodDistances, ExpandIters * 8, 0, null, regionBuffer, distanceBuffer);
453 
454  //filter out small regions
455  this.maxRegions = FilterSmallRegions(regions, minRegionArea, mergeRegionArea, regionIndex);
456 
457  //write the result out
458  for (int i = 0; i < spans.Length; i++)
459  spans[i].Region = regions[i];
460  }

468  private static void AppendStacks(List<CompactSpanReference> source, List<CompactSpanReference> destination, RegionId[] regions)
469  {
470  for (int j = 0; j < source.Count; j++)
471  {
472  var spanRef = source[j];
473  if (spanRef.Index < 0 || regions[spanRef.Index] != 0)
474  continue;
475 
476  destination.Add(spanRef);
477  }

488  private int FilterSmallRegions(RegionId[] regionIds, int minRegionArea, int mergeRegionSize, int maxRegionId)
489  {
490  int numRegions = maxRegionId + 1;
491  Region[] regions = new Region[numRegions];
492 
493  //construct regions
494  for (int i = 0; i < numRegions; i++)
495  regions[i] = new Region(i);
496 
497  //find edge of a region and find connections around a contour
498  for (int y = 0; y < length; y++)
499  {
500  for (int x = 0; x < width; x++)
501  {
502  CompactCell c = cells[x + y * width];
503  for (int i = c.StartIndex, end = c.StartIndex + c.Count; i < end; i++)
504  {
505  CompactSpanReference spanRef = new CompactSpanReference(x, y, i);
506 
507  //HACK since the border region flag makes r negative, I changed r == 0 to r <= 0. Figure out exactly what maxRegionId's purpose is and see if Region.IsBorderOrNull is all we need.
508  int r = (int)regionIds[i];
509  if (r <= 0 || (int)r >= numRegions)
510  continue;
511 
512  Region reg = regions[(int)r];
513  reg.SpanCount++;
514 
515  //update floors
516  for (int j = c.StartIndex; j < end; j++)
517  {
518  if (i == j) continue;
519  RegionId floorId = regionIds[j];
520  if (floorId == 0 || (int)floorId >= numRegions)
521  continue;
522  reg.AddUniqueFloorRegion(floorId);
523  }
524 
525  //have found contour
526  if (reg.Connections.Count > 0)
527  continue;
528 
529  reg.AreaType = areas[i];
530 
531  //check if this cell is next to a border
532  for (var dir = Direction.West; dir <= Direction.South; dir++)
533  {
534  if (IsSolidEdge(regionIds, ref spanRef, dir))
535  {
536  //The cell is at a border.
537  //Walk around contour to find all neighbors
538  WalkContour(regionIds, spanRef, dir, reg.Connections);
539  break;
540  }
541  }
542  }
543  }
544  }
545 
546  //Remove too small regions
547  Stack<RegionId> stack = new Stack<RegionId>();
548  List<RegionId> trace = new List<RegionId>();
549  for (int i = 0; i < numRegions; i++)
550  {
551  Region reg = regions[i];
552  if (reg.IsBorderOrNull || reg.SpanCount == 0 || reg.Visited)
553  continue;
554 
555  //count the total size of all connected regions
556  //also keep track of the regions connections to a tile border
557  bool connectsToBorder = false;
558  int spanCount = 0;
559  stack.Clear();
560  trace.Clear();
561 
562  reg.Visited = true;
563  stack.Push(reg.Id);
564 
565  while (stack.Count > 0)
566  {
567  //pop
568  RegionId ri = stack.Pop();
569 
570  Region creg = regions[(int)ri];
571 
572  spanCount += creg.SpanCount;
573  trace.Add(ri);
574 
575  for (int j = 0; j < creg.Connections.Count; j++)
576  {
577  if (RegionId.HasFlags(creg.Connections[j], RegionFlags.Border))
578  {
579  connectsToBorder = true;
580  continue;
581  }
582 
583  Region neiReg = regions[(int)creg.Connections[j]];
584  if (neiReg.Visited || neiReg.IsBorderOrNull)
585  continue;
586 
587  //visit
588  stack.Push(neiReg.Id);
589  neiReg.Visited = true;
590  }
591  }
592 
593  //if the accumulated region size is too small, remove it
594  //do not remove areas which connect to tile borders as their size can't be estimated correctly
595  //and removing them can potentially remove necessary areas
596  if (spanCount < minRegionArea && !connectsToBorder)
597  {
598  //kill all visited regions
599  for (int j = 0; j < trace.Count; j++)
600  {
601  int index = (int)trace[j];
602 
603  regions[index].SpanCount = 0;
604  regions[index].Id = RegionId.Null;
605  }
606  }
607  }
608 
609  //Merge too small regions to neighbor regions
610  int mergeCount = 0;
611  do
612  {
613  mergeCount = 0;
614  for (int i = 0; i < numRegions; i++)
615  {
616  Region reg = regions[i];
617  if (reg.IsBorderOrNull || reg.SpanCount == 0)
618  continue;
619 
620  //check to see if region should be merged
621  if (reg.SpanCount > mergeRegionSize && reg.IsConnectedToBorder())
622  continue;
623 
624  //small region with more than one connection or region which is not connected to border at all
625  //find smallest neighbor that connects to this one
626  int smallest = int.MaxValue;
627  RegionId mergeId = reg.Id;
628  for (int j = 0; j < reg.Connections.Count; j++)
629  {
630  if (RegionId.HasFlags(reg.Connections[j], RegionFlags.Border))
631  continue;
632 
633  Region mreg = regions[(int)reg.Connections[j]];
634  if (mreg.IsBorderOrNull)
635  continue;
636 
637  if (mreg.SpanCount < smallest && reg.CanMergeWith(mreg) && mreg.CanMergeWith(reg))
638  {
639  smallest = mreg.SpanCount;
640  mergeId = mreg.Id;
641  }
642  }
643 
644  //found new id
645  if (mergeId != reg.Id)
646  {
647  RegionId oldId = reg.Id;
648  Region target = regions[(int)mergeId];
649 
650  //merge regions
651  if (target.MergeWithRegion(reg))
652  {
653  //fix regions pointing to current region
654  for (int j = 0; j < numRegions; j++)
655  {
656  if (regions[j].IsBorderOrNull)
657  continue;
658 
659  //if another regions was already merged into current region
660  //change the nid of the previous region too
661  if (regions[j].Id == oldId)
662  regions[j].Id = mergeId;
663 
664  //replace current region with new one if current region is neighbor
665  regions[j].ReplaceNeighbour(oldId, mergeId);
666  }
667 
668  mergeCount++;
669  }
670  }
671  }
672  }
673  while (mergeCount > 0);
674 
675  //Compress region ids
676  for (int i = 0; i < numRegions; i++)
677  {
678  regions[i].Remap = false;
679 
680  if (regions[i].IsBorderOrNull)
681  continue;
682 
683  regions[i].Remap = true;
684  }
685 
686  int regIdGen = 0;
687  for (int i = 0; i < numRegions; i++)
688  {
689  if (!regions[i].Remap)
690  continue;
691 
692  RegionId oldId = regions[i].Id;
693  RegionId newId = new RegionId(++regIdGen);
694  for (int j = i; j < numRegions; j++)
695  {
696  if (regions[j].Id == oldId)
697  {
698  regions[j].Id = newId;
699  regions[j].Remap = false;
700  }
701  }
702  }
703 
704  //Remap regions
705  for (int i = 0; i < spans.Length; i++)
706  {
707  if (!RegionId.HasFlags(regionIds[i], RegionFlags.Border))
708  regionIds[i] = regions[(int)regionIds[i]].Id;
709  }
710 
711  return regIdGen;
712  }

718  private void CalculateDistanceField(int[] src)
719  {
720  //initialize distance and points
721  for (int i = 0; i < spans.Length; i++)
722  src[i] = int.MaxValue;
723 
724  //mark boundary cells
725  for (int y = 0; y < length; y++)
726  {
727  for (int x = 0; x < width; x++)
728  {
729  CompactCell c = cells[y * width + x];
730  for (int i = c.StartIndex, end = c.StartIndex + c.Count; i < end; i++)
731  {
732  CompactSpan s = spans[i];
733  Area area = areas[i];
734 
735  bool isBoundary = false;
736  if (s.ConnectionCount != 4)
737  isBoundary = true;
738  else
739  {
740  for (var dir = Direction.West; dir <= Direction.South; dir++)
741  {
742  int dx = x + dir.GetHorizontalOffset();
743  int dy = y + dir.GetVerticalOffset();
744  int di = cells[dx + dy * width].StartIndex + CompactSpan.GetConnection(ref s, dir);
745  if (area != areas[di])
746  {
747  isBoundary = true;
748  break;
749  }
750  }
751  }
752 
753  if (isBoundary)
754  src[i] = 0;
755  }
756  }
757  }
758 
759  //pass 1
760  for (int y = 0; y < length; y++)
761  {
762  for (int x = 0; x < width; x++)
763  {
764  CompactCell c = cells[y * width + x];
765  for (int i = c.StartIndex, end = c.StartIndex + c.Count; i < end; i++)
766  {
767  CompactSpan s = spans[i];
768 
769  if (s.IsConnected(Direction.West))
770  {
771  //(-1, 0)
772  int dx = x + Direction.West.GetHorizontalOffset();
773  int dy = y + Direction.West.GetVerticalOffset();
774  int di = cells[dx + dy * width].StartIndex + CompactSpan.GetConnection(ref s, Direction.West);
775  CompactSpan ds = spans[di];
776  if (src[di] + 2 < src[i])
777  src[i] = src[di] + 2;
778 
779  //(-1, -1)
780  if (ds.IsConnected(Direction.South))
781  {
782  int ddx = dx + Direction.South.GetHorizontalOffset();
783  int ddy = dy + Direction.South.GetVerticalOffset();
784  int ddi = cells[ddx + ddy * width].StartIndex + CompactSpan.GetConnection(ref ds, Direction.South);
785  if (src[ddi] + 3 < src[i])
786  src[i] = src[ddi] + 3;
787  }
788  }
789 
790  if (s.IsConnected(Direction.South))
791  {
792  //(0, -1)
793  int dx = x + Direction.South.GetHorizontalOffset();
794  int dy = y + Direction.South.GetVerticalOffset();
795  int di = cells[dx + dy * width].StartIndex + CompactSpan.GetConnection(ref s, Direction.South);
796  CompactSpan ds = spans[di];
797  if (src[di] + 2 < src[i])
798  src[i] = src[di] + 2;
799 
800  //(1, -1)
801  if (ds.IsConnected(Direction.East))
802  {
803  int ddx = dx + Direction.East.GetHorizontalOffset();
804  int ddy = dy + Direction.East.GetVerticalOffset();
805  int ddi = cells[ddx + ddy * width].StartIndex + CompactSpan.GetConnection(ref ds, Direction.East);
806  if (src[ddi] + 3 < src[i])
807  src[i] = src[ddi] + 3;
808  }
809  }
810  }
811  }
812  }
813 
814  //pass 2
815  for (int y = length - 1; y >= 0; y--)
816  {
817  for (int x = width - 1; x >= 0; x--)
818  {
819  CompactCell c = cells[y * width + x];
820  for (int i = c.StartIndex, end = c.StartIndex + c.Count; i < end; i++)
821  {
822  CompactSpan s = spans[i];
823 
824  if (s.IsConnected(Direction.East))
825  {
826  //(1, 0)
827  int dx = x + Direction.East.GetHorizontalOffset();
828  int dy = y + Direction.East.GetVerticalOffset();
829  int di = cells[dx + dy * width].StartIndex + CompactSpan.GetConnection(ref s, Direction.East);
830  CompactSpan ds = spans[di];
831  if (src[di] + 2 < src[i])
832  src[i] = src[di] + 2;
833 
834  //(1, 1)
835  if (ds.IsConnected(Direction.North))
836  {
837  int ddx = dx + Direction.North.GetHorizontalOffset();
838  int ddy = dy + Direction.North.GetVerticalOffset();
839  int ddi = cells[ddx + ddy * width].StartIndex + CompactSpan.GetConnection(ref ds, Direction.North);
840  if (src[ddi] + 3 < src[i])
841  src[i] = src[ddi] + 3;
842  }
843  }
844 
845  if (s.IsConnected(Direction.North))
846  {
847  //(0, 1)
848  int dx = x + Direction.North.GetHorizontalOffset();
849  int dy = y + Direction.North.GetVerticalOffset();
850  int di = cells[dx + dy * width].StartIndex + CompactSpan.GetConnection(ref s, Direction.North);
851  CompactSpan ds = spans[di];
852  if (src[di] + 2 < src[i])
853  src[i] = src[di] + 2;
854 
855  //(-1, 1)
856  if (ds.IsConnected(Direction.West))
857  {
858  int ddx = dx + Direction.West.GetHorizontalOffset();
859  int ddy = dy + Direction.West.GetVerticalOffset();
860  int ddi = cells[ddx + ddy * width].StartIndex + CompactSpan.GetConnection(ref ds, Direction.West);
861  if (src[ddi] + 3 < src[i])
862  src[i] = src[ddi] + 3;
863  }
864  }
865  }
866  }
867  }

876  private void BoxBlur(int[] distances, int threshold, int[] buffer)
877  {
878  threshold *= 2;
879 
880  //if the optional buffer parameter wasn't passed in, or is too small, make a new one.
881  if (buffer == null || buffer.Length < distances.Length)
882  buffer = new int[distances.Length];
883 
884  Buffer.BlockCopy(distances, 0, buffer, 0, distances.Length * sizeof(int));
885 
886  //horizontal pass
887  for (int y = 0; y < length; y++)
888  {

952  private void ExpandRegions(RegionId[] regions, int[] floodDistances, int maxIterations, int level, List<CompactSpanReference> stack, RegionId[] regionBuffer, int[] distanceBuffer)
953  {
954  //generate buffers if they're not passed in or if they're too small.
955  if (regionBuffer == null || regionBuffer.Length < regions.Length)
956  regionBuffer = new RegionId[regions.Length];
957 
958  if (distanceBuffer == null || distanceBuffer.Length < floodDistances.Length)
959  distanceBuffer = new int[floodDistances.Length];
960 
961  //copy existing data into the buffers.
962  Array.Copy(regions, 0, regionBuffer, 0, regions.Length);
963  Array.Copy(floodDistances, 0, distanceBuffer, 0, floodDistances.Length);
964 
965  //find cells that are being expanded to.
966  if (stack == null)
967  {
968  stack = new List<CompactSpanReference>();
969  for (int y = 0; y < length; y++)
970  {
971  for (int x = 0; x < width; x++)
972  {
973  CompactCell c = cells[x + y * width];
974  for (int i = c.StartIndex, end = c.StartIndex + c.Count; i < end; i++)
975  {
976  //a cell is being expanded to if it's distance is greater than the current level,
977  //but no region has been asigned yet. It must also not be in a null area.
978  if (this.distances[i] >= level && regions[i] == 0 && areas[i].IsWalkable)
979  stack.Add(new CompactSpanReference(x, y, i));
980  }
981  }
982  }
983  }
984  else
985  {
986  for (int j = 0; j < stack.Count; j++)
987  {
988  if (regions[stack[j].Index] != 0)
989  stack[j] = CompactSpanReference.Null;
990  }
991  }
992 
993  //assign regions to all the cells that are being expanded to.
994  //will run until it's done or it runs maxIterations times.
995  int iter = 0;
996  while (stack.Count > 0)
997  {
998  //spans in the stack that are skipped:
999  // - assigned a region ID in an earlier iteration
1000  // - not neighboring any spans with region IDs
1001  int skipped = 0;
1002 
1003  for (int j = 0; j < stack.Count; j++)
1004  {
1005  CompactSpanReference spanRef = stack[j];
1006  int x = spanRef.X;
1007  int y = spanRef.Y;
1008  int i = spanRef.Index;
1009 
1010  //skip regions already assigned to
1011  if (i < 0)
1012  {
1013  skipped++;
1014  continue;
1015  }
1016 
1017  RegionId r = regions[i];
1018  Area area = areas[i];
1019  CompactSpan s = spans[i];
1020 
1021  //search direct neighbors for the one with the smallest distance value
1022  int minDist = int.MaxValue;
1023  for (var dir = Direction.West; dir <= Direction.South; dir++)
1024  {
1025  if (!s.IsConnected(dir))
1026  continue;
1027 
1028  int dx = x + dir.GetHorizontalOffset();
1029  int dy = y + dir.GetVerticalOffset();
1030  int di = cells[dx + dy * width].StartIndex + CompactSpan.GetConnection(ref s, dir);
1031 
1032  if (areas[di] != area)
1033  continue;
1034 
1035  //compare distance to previous best
1036  RegionId ri = regions[di];
1037  int dist = floodDistances[di];
1038  if (!(ri.IsNull || RegionId.HasFlags(ri, RegionFlags.Border)))
1039  {
1040  //set region and distance if better
1041  if (dist + 2 < minDist)
1042  {
1043  r = ri;
1044  minDist = dist + 2;
1045  }
1046  }
1047  }
1048 
1049  if (r != 0)
1050  {
1051  //set the region and distance for this span
1052  regionBuffer[i] = r;
1053  distanceBuffer[i] = minDist;
1054 
1055  //mark this item in the stack as assigned for the next iteration.
1056  stack[j] = CompactSpanReference.Null;
1057  }
1058  else
1059  {
1060  //skip spans that don't neighbor any regions
1061  skipped++;
1062  }
1063  }
1064 
1065  //if the entire stack is being skipped, we're done.
1066  if (skipped == stack.Count)
1067  break;
1068 
1069  //Copy from the buffers back to the original arrays. This is done after each iteration
1070  //because changing it in-place has some side effects for the other spans in the stack.
1071  Array.Copy(regionBuffer, 0, regions, 0, regions.Length);
1072  Array.Copy(distanceBuffer, 0, floodDistances, 0, floodDistances.Length);
1073 
1074  if (level > 0)
1075  {
1076  //if we hit maxIterations before expansion is done, break out anyways.
1077  ++iter;
1078  if (iter >= maxIterations)
1079  break;
1080  }
1081  }
