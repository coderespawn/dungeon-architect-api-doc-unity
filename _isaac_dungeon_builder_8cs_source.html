66  void GenerateLevelLayout()
67  {
68  var queue = new Queue<LevelGrowthNode>();
69  var visited = new HashSet<IntVector>();
70 
71  var roomFactory = new IsaacRoomFactory();
72  rooms.Clear();
73  doors.Clear();
74 
75  var start = new LevelGrowthNode();
76  start.room = roomFactory.CreateRoom(IntVector.Zero);
77  start.moveDirection = 0;
78  rooms.Add(start.room);
79 
80  queue.Enqueue(start);
81  visited.Add(start.room.position);
82 
83  var numRooms = random.Range(isaacConfig.minRooms, isaacConfig.maxRooms);
84  bool isSpawnRoom = true;
85  while (queue.Count > 0)
86  {
87  var top = queue.Dequeue();
88  if (isSpawnRoom)
89  {
90  // in the spawn room. Spawn on all 4 sides
91  for (int d = 0; d < 4; d++)
92  {
93  AddNextRoomNode(roomFactory, queue, visited, numRooms, top.room, d, isaacConfig.spawnRoomBranchProbablity);
94  }
95  isSpawnRoom = false;
96  }
97  else
98  {
99  // Grow forward
100  AddNextRoomNode(roomFactory, queue, visited, numRooms, top.room, top.moveDirection, isaacConfig.growForwardProbablity);
101 
102  // Grow sideways
103  AddNextRoomNode(roomFactory, queue, visited, numRooms, top.room, (top.moveDirection + 1) % 4, isaacConfig.growSidewaysProbablity);
104  AddNextRoomNode(roomFactory, queue, visited, numRooms, top.room, (top.moveDirection + 3) % 4, isaacConfig.growSidewaysProbablity);
105  }
106 
107  if (rooms.Count >= numRooms)
108  {
109  break;
110  }
111  }
112 
113  // Generate the tile layout of the rooms
114  var layoutBuilder = GetComponent<IsaacRoomLayoutBuilder>();
115  foreach (var room in rooms)
116  {
117  GenerateRoomLayout(layoutBuilder, room);
118  }
119 
120  isaacModel.rooms = rooms.ToArray();
121  isaacModel.doors = doors.ToArray();
122  rooms.Clear();
123  doors.Clear();
124  }

126  void AddNextRoomNode(IsaacRoomFactory roomFactory,
127  Queue<LevelGrowthNode> queue, HashSet<IntVector> visited, int maxRooms,
128  IsaacRoom parentRoom, int direction, float probability)
129  {
130  if (random.NextFloat() > probability) return;
131  if (rooms.Count >= maxRooms) return;
132 
133  var nextPosition = parentRoom.position + directions[direction];
134  if (!visited.Contains(nextPosition))
135  {
136  var nextRoom = roomFactory.CreateRoom(nextPosition);
137  rooms.Add(nextRoom);
138  var nextNode = new LevelGrowthNode();
139  nextNode.room = nextRoom;
140  nextNode.moveDirection = direction;
141  queue.Enqueue(nextNode);
142  visited.Add(nextPosition);
143 
144  // Create a door between the two rooms
145  ConnectRoomsWithDoors(parentRoom, nextRoom);
146  }
147  else
148  {
149  // See if we can connect to the other room
150  // first make sure we don't already have a connection between the two
151  var nextRoom = GetRoomAt(nextPosition);
152  if (!ContainsDoorBetween(parentRoom.roomId, nextRoom.roomId))
153  {
154  float loopTest = random.NextFloat();
155  if (loopTest < isaacConfig.cycleProbability)
156  {
157  // Connect the two rooms together
158  if (nextRoom != null)
159  {
160  // Create a door between the two rooms
161  ConnectRoomsWithDoors(parentRoom, nextRoom);
162  }
163  }
164  }
165  }

200  void CreateDoor(IsaacRoom roomA, IsaacRoom roomB, float ratio)
201  {
202  var door = new IsaacDoor();
203  door.roomA = roomA.roomId;
204  door.roomB = roomB.roomId;
205  door.ratio = ratio;
206  doors.Add(door);

208  // Create the door tile
209  var roomWidth = isaacConfig.roomWidth;
210  var roomHeight = isaacConfig.roomHeight;
211  bool horizontal = (roomA.position.z - roomB.position.z) == 0; // Are the two room horizontal
212  var size = horizontal ? isaacConfig.roomHeight : isaacConfig.roomWidth;
213  var location1D = Mathf.FloorToInt(size * door.ratio);
214 
215  var leftRoom = roomA;
216  var rightRoom = roomB;
217  if (horizontal && leftRoom.position.x > rightRoom.position.x)
218  {
219  // Swap
220  leftRoom = roomB;
221  rightRoom = roomA;
222  }
223  else if (!horizontal && leftRoom.position.z > rightRoom.position.z)
224  {
225  // Swap
226  leftRoom = roomB;
227  rightRoom = roomA;
228  }
229 
230 
231  IntVector leftRoomDoor;
232  IntVector rightRoomDoor;
233 
234  if (horizontal)
235  {
236  leftRoomDoor = new IntVector(roomWidth, 0, location1D);
237  rightRoomDoor = new IntVector(-1, 0, location1D);
238  }
239  else
240  {

242  rightRoomDoor = new IntVector(location1D, 0, -1);
243  }

300 
301  for (int x = -1; x < roomWidth + 1; x++)
302  {
303  for (int z = -1; z < roomHeight + 1; z++)
304  {

306  var tileOffset = tilePosition * tileSize;
