137  List<MergeMeshItem[]> SplitPerVertexLimit(List<MergeMeshItem> patchItems, int vertexLimit)
138  {
139  var result = new List<MergeMeshItem[]>();
140  var currentList = new List<MergeMeshItem>();
141  int vertexCount = 0;
142  foreach (var item in patchItems)
143  {
144  if (item == null || item.mesh == null) continue;
145 
146  if (vertexCount + item.mesh.vertexCount >= vertexLimit)
147  {
148  if (currentList.Count > 0)
149  {
150  result.Add(currentList.ToArray());
151  }
152 
153  currentList = new List<MergeMeshItem>();
154  vertexCount = 0;
155  }
156 
157  currentList.Add(item);
158  vertexCount += item.mesh.vertexCount;
159  }
160 
161  if (currentList.Count > 0)
162  {
163  result.Add(currentList.ToArray());

169  Dictionary<IntVector2, List<MergeMeshItem>> SpatialPartitionByPatchSize(List<MergeMeshItem> items)
170  {
171  var patches = new Dictionary<IntVector2, List<MergeMeshItem>>();
172  foreach (var item in items)
173  {
174  var position = Matrix.GetTranslation(ref item.transform);
175  var patchIndex = new IntVector2(
176  Mathf.FloorToInt(position.x / mergePatchSize),
177  Mathf.FloorToInt(position.z / mergePatchSize));
178 
179  if (!patches.ContainsKey(patchIndex))
180  {
181  patches.Add(patchIndex, new List<MergeMeshItem>());
182  }
183 
184  patches[patchIndex].Add(item);
185  }
186

190  static Mesh CreateMergedMesh(MergeMeshItem[] items)
191  {
192  var vertices = new List<Vector3>();
193  var uv = new List<Vector2>();
194  var triangles = new List<int>();
195 
196  int baseIndex = 0;
197  foreach (var item in items)
198  {
199  foreach (var localVertex in item.mesh.vertices)
200  {
201  var worldVertex = item.transform.MultiplyPoint(localVertex);
202  vertices.Add(worldVertex);
203  }
204 
205  foreach (var index in item.mesh.triangles)
206  {
207  triangles.Add(baseIndex + index);
208  }
209 
210  baseIndex += item.mesh.vertices.Length;
211 
212  uv.AddRange(item.mesh.uv);
213  }
214 
215  var mesh = new Mesh();
216  mesh.SetVertices(vertices);
217  mesh.SetUVs(0, uv);
218  mesh.SetTriangles(triangles.ToArray(), 0);
219  mesh.RecalculateNormals();
220  return mesh;
221  }
