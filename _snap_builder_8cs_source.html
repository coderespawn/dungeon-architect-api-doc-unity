181  void FixupDoorStates(ModuleBuildNode rootNode)
182  {
183  var moduleConnections = new Dictionary<GameObject, SnapConnection[]>();
184  TraverseTree(rootNode, delegate (ModuleBuildNode node)
185  {
186  if (!moduleConnections.ContainsKey(node.spawnedModule))

189  moduleConnections.Add(node.spawnedModule, connections);
190  }
191  });

193  // Set everything to wall

195  {

202  var stack = new Stack<ModuleBuildNode>();
203  stack.Push(rootNode);
204  while (stack.Count > 0)
205  {
206  ModuleBuildNode top = stack.Pop();
207  if (top == null) continue;
208  ModuleBuildNode parent = top.Parent;
209  if (parent != null)

211  if (top.spawnedModule != null && parent.spawnedModule != null)

214  int TopDoorIndex = top.AttachmentConfig.AttachedModuleDoorIndex;
215  var parentConnection = GetArrayEntry<SnapConnection>(ParentDoorIndex, moduleConnections[parent.spawnedModule]);
216  var topConnection = GetArrayEntry<SnapConnection>(TopDoorIndex, moduleConnections[top.spawnedModule]);
217

312  bool FindAttachmentConfiguration(ModuleInfo TargetModule, ModuleInfo IncomingModule, ref Matrix4x4 IncomingModuleTransform,
313  int IncomingDoorIndex, HashSet<Bounds> OccupiedBounds, ref SnapAttachmentConfiguration OutAttachmentConfig)
314  {
315  int TargetNumDoors = TargetModule.ConnectionTransforms.Length;
316  //if (IncomingDoorIndex >= TargetNumDoors) return false;
317 
318  if (IncomingDoorIndex < 0 || IncomingModule == null)
319  {
320  OutAttachmentConfig.AttachedModule = TargetModule;
321  OutAttachmentConfig.AttachedModuleDoorIndex = random.Range(0, TargetNumDoors - 1);
322  OutAttachmentConfig.AttachedModuleWorldBounds = GetModulePrefabBounds(TargetModule.ModuleTemplate.module, TargetModule.Bounds);
323  OutAttachmentConfig.AttachedModuleTransform = IncomingModuleTransform;
324  return true;
325  }
326 
327  //if (IncomingDoorIndex >= TargetNumDoors) return false;
328  Matrix4x4 IncomingDoorTransform = IncomingModule.ConnectionTransforms[IncomingDoorIndex];
329  string IncomingDoorCategory = IncomingModule.ConnectionCategory[IncomingDoorIndex];
330 
331  bool bFoundValid = false;
332  int[] ShuffledIndices = MathUtils.GetShuffledIndices(TargetNumDoors, random);
333  for (int si = 0; si < ShuffledIndices.Length; si++)
334  {
335  int Index = ShuffledIndices[si];
336  string AttachmentDoorCategory = TargetModule.ConnectionCategory[Index];
337  if (AttachmentDoorCategory != IncomingDoorCategory)
338  {
339  // The doors do not fit
340  continue;
341  }
342 
343  Matrix4x4 AttachmentDoorTransform = TargetModule.ConnectionTransforms[Index];
344 
345  // Align the module with a door that fits the incoming door
346  Matrix4x4[] ModuleTransforms = FindAttachmentTransforms(ref IncomingModuleTransform, ref IncomingDoorTransform, ref AttachmentDoorTransform);
347 
348  foreach (var ModuleTransform in ModuleTransforms)
349  {
350  if (!snapConfig.RotateModulesToFit)
351  {
352  // Rotation is not allowed. Check if we rotated the module
353  Matrix4x4 ModuleTransformCopy = ModuleTransform;
354  var moduleRotation = Matrix.GetRotation(ref ModuleTransformCopy);
355  if (Mathf.Abs(moduleRotation.eulerAngles.y) > 0.1f)
356  {
357  // Module was rotated
358  continue;
359  }
360  }
361 
362  {
363  // Calculate the bounds of the module
364  Bounds ModuleWorldBounds = GetModulePrefabBounds(TargetModule.ModuleTemplate.module, TargetModule.Bounds);
365  ModuleWorldBounds = MathUtils.TransformBounds(ModuleTransform, ModuleWorldBounds);
366  Bounds ContractedModuleWorldBounds = ExpandBounds(ModuleWorldBounds, -1 * (snapConfig.CollisionTestContraction + 1e-4f));
367 
368  // Check if this module would intersect with any of the existing modules
369  bool bIntersects = false;
370  foreach (var OccupiedBound in OccupiedBounds)
371  {
372  if (ContractedModuleWorldBounds.Intersects(OccupiedBound))
373  {
374  // intersects. Do not spawn a module here
375  bIntersects = true;
376  break;
377  }
378  }
379  if (bIntersects)
380  {
381  continue;
382  }
383 
384  // We found a valid module. Use this
385  OutAttachmentConfig.AttachedModule = TargetModule;
386  OutAttachmentConfig.AttachedModuleDoorIndex = Index;
387  OutAttachmentConfig.AttachedModuleWorldBounds = ModuleWorldBounds;
388  OutAttachmentConfig.AttachedModuleTransform = ModuleTransform;
389  bFoundValid = true;
390  break;
391  }
392  }
393  }
394 
395  return bFoundValid;
396  }

474  ModuleBuildNode BuildLayoutRecursive(ModuleGrowthNode GrowthNode, ref HashSet<Bounds> _OccupiedBounds, RNode graphNode, SnapLayoutBuildState RecursiveState)
475  {
476  RecursiveState.NumTries++;
477  if (RecursiveState.NumTries >= snapConfig.MaxProcessingPower)
478  {
479  return null;
480  }
481 
482  if (graphNode == null || graphNode.Payload == null || graphNode.Payload.nodeType == null)
483  {
484  return null;
485  }
486 
487  ModuleGrowthNode Top = GrowthNode;
488 
489  // Pick a door from this module to extend
490  ModuleBuildNode BestBuildNode = null;
491 
492  HashSet<Bounds> OccupiedBounds = new HashSet<Bounds>(_OccupiedBounds);
493 
494  int[] ModuleListIndices;
495  ModuleListIndices = FindFilteredModuleList(RecursiveState.ModuleInfoList, graphNode.Payload.nodeType.nodeName);
496  MathUtils.Shuffle(ModuleListIndices, random);
497 
498  for (int si = 0; si < ModuleListIndices.Length; si++)
499  {
500  int Index = ModuleListIndices[si];
501 
502  ModuleInfo Module = RecursiveState.ModuleInfoList[Index];
503 
504  var AttachmentConfig = new SnapAttachmentConfiguration();
505  if (!FindAttachmentConfiguration(Module, Top.IncomingModule, ref Top.ModuleTransform, Top.IncomingModuleDoorIndex, OccupiedBounds, ref AttachmentConfig))
506  {
507  continue;
508  }
509 
510  var BuildNode = new ModuleBuildNode();
511  BuildNode.AttachmentConfig = AttachmentConfig;
512  BuildNode.IncomingDoorIndex = Top.IncomingModuleDoorIndex;
513  BuildNode.Module = Module;
514 
515  // We found a valid module. Use this
516  {
517  Bounds contractedModuleWorldBounds = ExpandBounds(AttachmentConfig.AttachedModuleWorldBounds, -1 * snapConfig.CollisionTestContraction);
518  OccupiedBounds.Add(contractedModuleWorldBounds);
519  }
520 
521  int AttachmentDoorIndex = AttachmentConfig.AttachedModuleDoorIndex;
522 
523  var childGraphNodes = graphNode.Outgoing.ToArray();
524 
525  bool allChildrenFound = true;
526  foreach (var childGraphNode in childGraphNodes)
527  {
528  ModuleBuildNode ChildBuildNode = null;
529  HashSet<Bounds> ChildOccupiedBounds = null;
530 
531  // Extend from this door further
532  var ExtensionDoorIndices = MathUtils.GetShuffledIndices(Module.ConnectionTransforms.Length, random);
533  for (int ExtensionDoorIndexRef = 0; ExtensionDoorIndexRef < ExtensionDoorIndices.Length; ExtensionDoorIndexRef++)
534  {
535  int ExtensionDoorIndex = ExtensionDoorIndices[ExtensionDoorIndexRef];
536  if (ExtensionDoorIndex == AttachmentDoorIndex && Top.IncomingModuleDoorIndex != -1)
537  {
538  // Don't want to extend from the door we came in through
539  continue;
540  }
541 
543 
544  // Grow this branch further
545  var NextNode = new ModuleGrowthNode();
546  NextNode.IncomingModuleDoorIndex = ExtensionDoorIndex;
547  NextNode.ModuleTransform = AttachmentConfig.AttachedModuleTransform;
548  NextNode.IncomingModule = Module;
549  var ExtensionOccupiedBounds = new HashSet<Bounds>(OccupiedBounds);
550  ModuleBuildNode ExtensionNode = BuildLayoutRecursive(NextNode, ref ExtensionOccupiedBounds, childGraphNode, RecursiveState);
551 
552  if (ExtensionNode != null)
553  {
554  ChildBuildNode = ExtensionNode;
555  ChildOccupiedBounds = ExtensionOccupiedBounds;
556  break;
557  }
558  }
559 
560  if (ChildBuildNode != null)
561  {
562  // We cannot grow the child branches
563  ChildBuildNode.Parent = BuildNode;
564  BuildNode.Extensions.Add(ChildBuildNode);
565 
566  OccupiedBounds = ChildOccupiedBounds;
567  }
568  else
569  {
570  allChildrenFound = false;
571  break;
572  }
573  }
574 
575  if (allChildrenFound)
576  {
577  BestBuildNode = BuildNode;
578  _OccupiedBounds = OccupiedBounds;
579  break;
580  }
581  }
582 
583  return BestBuildNode;
584  }
