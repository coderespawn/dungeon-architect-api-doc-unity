243  public void Update(float dt)
244  {
245  velocitySampleCount = 0;
246 
247  int numAgents = GetActiveAgents(agents);
248 
249  //check that all agents have valid paths
250  CheckPathValidity(agents, numAgents, dt);
251 
252  //update async move requests and path finder
253  UpdateMoveRequest();
254 
255  //optimize path topology
256  UpdateTopologyOptimization(agents, numAgents, dt);
257 
258  //register agents to proximity grid
259  grid.Clear();
260  for (int i = 0; i < numAgents; i++)
261  {
262  Agent a = agents[i];
263 
264  Vector3 p = a.Position;
265  float r = a.Parameters.Radius;
266  grid.AddItem(a, p.X - r, p.Z - r, p.X + r, p.Z + r);
267  }
268 
269  //get nearby navmesh segments and agents to collide with
270  for (int i = 0; i < numAgents; i++)
271  {
272  if (agents[i].State != AgentState.Walking)
273  continue;
274 
275  //update the collision boundary after certain distance has passed or if it has become invalid
276  float updateThr = agents[i].Parameters.CollisionQueryRange * 0.25f;
277  if (Vector3Extensions.Distance2D(agents[i].Position, agents[i].Boundary.Center) > updateThr * updateThr || !agents[i].Boundary.IsValid(navQuery))
278  {
279  agents[i].Boundary.Update(agents[i].Corridor.GetFirstPoly(), agents[i].Position, agents[i].Parameters.CollisionQueryRange, navQuery);
280  }
281 
282  //query neighbour agents
283  agents[i].NeighborCount = GetNeighbours(agents[i].Position, agents[i].Parameters.Height, agents[i].Parameters.CollisionQueryRange, agents[i], agents[i].Neighbors, AgentMaxNeighbours, agents, grid);
284 
285  for (int j = 0; j < agents[i].NeighborCount; j++)
286  agents[i].Neighbors[j].Index = GetAgentIndex(agents[agents[i].Neighbors[j].Index]);
287  }
288 
289  //find the next corner to steer to
290  for (int i = 0; i < numAgents; i++)
291  {
292  if (agents[i].State != AgentState.Walking)
293  continue;
294  if (agents[i].TargetState == TargetState.None ||
295  agents[i].TargetState == TargetState.Velocity)
296  continue;
297 
298  //find corners for steering
299  agents[i].CornerCount = agents[i].Corridor.FindCorners(
300  agents[i].CornerVerts, agents[i].CornerFlags, agents[i].CornerPolys, AgentMaxCorners, navQuery);
301 
302  //check to see if the corner after the next corner is directly visible
303  if (((agents[i].Parameters.UpdateFlags & UpdateFlags.OptimizeVis) != 0) && agents[i].CornerCount > 0)
304  {
305  Vector3 target = agents[i].CornerVerts[Math.Min(1, agents[i].CornerCount - 1)];
306  agents[i].Corridor.OptimizePathVisibility(target, agents[i].Parameters.PathOptimizationRange, navQuery);
307  }
308  }
309 
310  //trigger off-mesh connections (depends on corners)
311  for (int i = 0; i < numAgents; i++)
312  {
313  if (agents[i].State != AgentState.Walking)
314  continue;
315  if (agents[i].TargetState == TargetState.None ||
316  agents[i].TargetState == TargetState.Velocity)
317  continue;
318 
319  //check
320  float triggerRadius = agents[i].Parameters.Radius * 2.25f;
321  if (OverOffmeshConnection(agents[i], triggerRadius))
322  {
323  //prepare to off-mesh connection
324  int idx = i;
325 
326  //adjust the path over the off-mesh connection
327  int[] refs = new int[2];
328  if (agents[i].Corridor.MoveOverOffmeshConnection(agents[i].CornerPolys[agents[i].CornerCount - 1], refs, ref agentAnims[idx].StartPos, ref agentAnims[idx].EndPos, navQuery))
329  {
330  agentAnims[idx].InitPos = agents[i].Position;
331  agentAnims[idx].PolyRef = refs[1];
332  agentAnims[idx].Active = true;
333  agentAnims[idx].T = 0.0f;
334  agentAnims[idx].TMax = (Vector3Extensions.Distance2D(agentAnims[idx].StartPos, agentAnims[idx].EndPos)
335  / agents[i].Parameters.MaxSpeed) * 0.5f;
336 
337  agents[i].State = AgentState.Offmesh;
338  agents[i].CornerCount = 0;
339  agents[i].NeighborCount = 0;
340  continue;
341  }
342  }
343  }
344 
345  //calculate steering
346  for (int i = 0; i < numAgents; i++)
347  {
348  if (agents[i].State != AgentState.Walking)
349  continue;
350  if (agents[i].TargetState == TargetState.None)
351  continue;
352 
353  Vector3 dvel = new Vector3(0, 0, 0);
354 
355  if (agents[i].TargetState == TargetState.Velocity)
356  {
357  dvel = agents[i].TargetPosition;
358  agents[i].DesiredSpeed = agents[i].TargetPosition.Length();
359  }
360  else
361  {
362  //calculate steering direction
363  if ((agents[i].Parameters.UpdateFlags & UpdateFlags.AnticipateTurns) != 0)
364  CalcSmoothSteerDirection(agents[i], ref dvel);
365  else
366  CalcStraightSteerDirection(agents[i], ref dvel);
367 
368  //calculate speed scale, which tells the agent to slowdown at the end of the path
369  float slowDownRadius = agents[i].Parameters.Radius * 2;
370  float speedScale = GetDistanceToGoal(agents[i], slowDownRadius) / slowDownRadius;
371 
372  agents[i].DesiredSpeed = agents[i].Parameters.MaxSpeed;
373  dvel = dvel * (agents[i].DesiredSpeed * speedScale);
374  }
375 
376  //separation
377  if ((agents[i].Parameters.UpdateFlags & UpdateFlags.Separation) != 0)
378  {
379  float separationDist = agents[i].Parameters.CollisionQueryRange;
380  float invSeparationDist = 1.0f / separationDist;
381  float separationWeight = agents[i].Parameters.SeparationWeight;
382 
383  float w = 0;
384  Vector3 disp = new Vector3(0, 0, 0);
385 
386  for (int j = 0; j < agents[i].NeighborCount; j++)
387  {
388  Agent nei = agents[agents[i].Neighbors[j].Index];
389 
390  Vector3 diff = agents[i].Position - nei.Position;
391  diff.Y = 0;
392 
393  float distSqr = diff.LengthSquared();
394  if (distSqr < 0.00001f)
395  continue;
396  if (distSqr > separationDist * separationDist)
397  continue;
398  float dist = (float)Math.Sqrt(distSqr);
399  float weight = separationWeight * (1.0f - (dist * invSeparationDist) * (dist * invSeparationDist));
400 
401  disp = disp + diff * (weight / dist);
402  w += 1.0f;
403  }
404 
405  if (w > 0.0001f)
406  {
407  //adjust desired veloctiy
408  dvel = dvel + disp * (1.0f / w);
409 
410  //clamp desired velocity to desired speed
411  float speedSqr = dvel.LengthSquared();
412  float desiredSqr = agents[i].DesiredSpeed * agents[i].DesiredSpeed;
413  if (speedSqr > desiredSqr)
414  dvel = dvel * (desiredSqr / speedSqr);
415  }
416  }
417 
418  //set the desired velocity
419  agents[i].DesiredVel = dvel;
420  }
421 
422  //velocity planning
423  for (int i = 0; i < numAgents; i++)
424  {
425  if (agents[i].State != AgentState.Walking)
426  continue;
427 
428  if ((agents[i].Parameters.UpdateFlags & UpdateFlags.ObstacleAvoidance) != 0)
429  {
430  this.obstacleQuery.Reset();
431 
432  //add neighhbors as obstacles
433  for (int j = 0; j < agents[i].NeighborCount; j++)
434  {
435  Agent nei = agents[agents[i].Neighbors[j].Index];
436  obstacleQuery.AddCircle(nei.Position, nei.Parameters.Radius, nei.Vel, nei.DesiredVel);
437  }
438 
439  //append neighbour segments as obstacles
440  for (int j = 0; j < agents[i].Boundary.SegCount; j++)
441  {
442  LocalBoundary.Segment s = agents[i].Boundary.Segs[j];
443  if (Triangle3.Area2D(agents[i].Position, s.Start, s.End) < 0.0f)
444  continue;
445  obstacleQuery.AddSegment(s.Start, s.End);
446  }
447 
448  //sample new safe velocity
449  bool adaptive = true;
450  int ns = 0;
451 
452  ObstacleAvoidanceQuery.ObstacleAvoidanceParams parameters = obstacleQueryParams[agents[i].Parameters.ObstacleAvoidanceType];
453 
454  if (adaptive)
455  {
456  ns = obstacleQuery.SampleVelocityAdaptive(agents[i].Position, agents[i].Parameters.Radius, agents[i].DesiredSpeed, agents[i].Vel, agents[i].DesiredVel, ref agents[i].NVel, parameters);
457  }
458  else
459  {
460  ns = obstacleQuery.SampleVelocityGrid(agents[i].Position, agents[i].Parameters.Radius, agents[i].DesiredSpeed, agents[i].Vel, agents[i].DesiredVel, ref agents[i].NVel, parameters);
461  }
462 
463  this.velocitySampleCount += ns;
464  }
465  else
466  {
467  //if not using velocity planning, new velocity is directly the desired velocity
468  agents[i].NVel = agents[i].DesiredVel;
469  }
470  }
471 
472  //integrate
473  for (int i = 0; i < numAgents; i++)
474  {
475  Agent ag = agents[i];
476 
477  if (ag.State != AgentState.Walking)
478  continue;
479 
480  ag.Integrate(dt);
481  }
482 
483  //handle collisions
484  const float COLLISION_RESOLVE_FACTOR = 0.7f;
485 
486  for (int iter = 0; iter < 4; iter++)
487  {
488  for (int i = 0; i < numAgents; i++)
489  {
490  int idx0 = GetAgentIndex(agents[i]);
491 
492  if (agents[i].State != AgentState.Walking)
493  continue;
494 
495  agents[i].Disp = new Vector3(0, 0, 0);
496 
497  float w = 0;
498 
499  for (int j = 0; j < agents[i].NeighborCount; j++)
500  {
501  Agent nei = agents[agents[i].Neighbors[j].Index];
502  int idx1 = GetAgentIndex(nei);
503 
504  Vector3 diff = agents[i].Position - nei.Position;
505  diff.Y = 0;
506 
507  float dist = diff.LengthSquared();
508  if (dist > (agents[i].Parameters.Radius + nei.Parameters.Radius) * (agents[i].Parameters.Radius + nei.Parameters.Radius))
509  continue;
510  dist = (float)Math.Sqrt(dist);
511  float pen = (agents[i].Parameters.Radius + nei.Parameters.Radius) - dist;
512  if (dist < 0.0001f)
513  {
514  //agents on top of each other, try to choose diverging separation directions
515  if (idx0 > idx1)
516  diff = new Vector3(-agents[i].DesiredVel.Z, 0, agents[i].DesiredVel.X);
517  else
518  diff = new Vector3(agents[i].DesiredVel.Z, 0, -agents[i].DesiredVel.X);
519  pen = 0.01f;
520  }
521  else
522  {
523  pen = (1.0f / dist) * (pen * 0.5f) * COLLISION_RESOLVE_FACTOR;
524  }
525 
526  agents[i].Disp = agents[i].Disp + diff * pen;
527 
528  w += 1.0f;
529  }
530 
531  if (w > 0.0001f)
532  {
533  float iw = 1.0f / w;
534  agents[i].Disp = agents[i].Disp * iw;
535  }
536  }
537 
538  for (int i = 0; i < numAgents; i++)
539  {
540  if (agents[i].State != AgentState.Walking)
541  continue;
542 
543  //move along navmesh
544  agents[i].Corridor.MovePosition(agents[i].Position, navQuery);
545 
546  //get valid constrained position back
547  agents[i].Position = agents[i].Corridor.Pos;
548 
549  //if not using path, truncate the corridor to just one poly
550  if (agents[i].TargetState == TargetState.None ||
551  agents[i].TargetState == TargetState.Velocity)
552  {
553  agents[i].Corridor.Reset(agents[i].Corridor.GetFirstPoly(), agents[i].Position);
554  agents[i].IsPartial = false;
555  }
556  }
557 
558  //update agents using offmesh connections
559  for (int i = 0; i < maxAgents; i++)
560  {
561  if (!agentAnims[i].Active)
562  continue;
563 
564  agentAnims[i].T += dt;
565  if (agentAnims[i].T > agentAnims[i].TMax)
566  {
567  //reset animation
568  agentAnims[i].Active = false;
569 
570  //prepare agent for walking
571  agents[i].State = AgentState.Walking;
572 
573  continue;
574  }
575 
576  //update position
577  float ta = agentAnims[i].TMax * 0.15f;
578  float tb = agentAnims[i].TMax;
579  if (agentAnims[i].T < ta)
580  {
581  float u = MathHelper.Normalize(agentAnims[i].T, 0.0f, ta);
582  Vector3 lerpOut;
583  Vector3.Lerp(ref agentAnims[i].InitPos, ref agentAnims[i].StartPos, u, out lerpOut);
584  agents[i].Position = lerpOut;
585  }
586  else
587  {
588  float u = MathHelper.Normalize(agentAnims[i].T, ta, tb);
589  Vector3 lerpOut;
590  Vector3.Lerp(ref agentAnims[i].StartPos, ref agentAnims[i].EndPos, u, out lerpOut);
591  agents[i].Position = lerpOut;
592  }
593 
594  agents[i].Vel = new Vector3(0, 0, 0);
595  agents[i].DesiredVel = new Vector3(0, 0, 0);
596  }
597  }
598  }

603  public void UpdateMoveRequest()
604  {
605  const int PATH_MAX_AGENTS = 8;
606  Agent[] queue = new Agent[PATH_MAX_AGENTS];
607  int numQueue = 0;
608  Status status;
609 
610  //fire off new requests
611  for (int i = 0; i < maxAgents; i++)
612  {
613  if (!agents[i].IsActive)
614  continue;
615  if (agents[i].State == AgentState.Invalid)
616  continue;
617  if (agents[i].TargetState == TargetState.None || agents[i].TargetState == TargetState.Velocity)
618  continue;
619 
620  if (agents[i].TargetState == TargetState.Requesting)
621  {
622  int[] path = agents[i].Corridor.Path;
623  int npath = agents[i].Corridor.PathCount;
624 
625  const int MAX_RES = 32;
626  Vector3 reqPos = new Vector3();
627  int[] reqPath = new int[MAX_RES];
628  int reqPathCount = 0;
629 
630  //quick search towards the goal
631  const int MAX_ITER = 20;
632  navQuery.InitSlicedFindPath(new NavPoint(path[0], agents[i].Position), new NavPoint(agents[i].TargetRef, agents[i].TargetPosition));
633  int tempInt = 0;
634  navQuery.UpdateSlicedFindPath(MAX_ITER, ref tempInt);
635  status = Status.Failure;
636  if (agents[i].TargetReplan)
637  {
638  //try to use an existing steady path during replan if possible
639  status = navQuery.FinalizedSlicedPathPartial(path, npath, reqPath, ref reqPathCount, MAX_RES).ToStatus();
640  }
641  else
642  {
643  //try to move towards the target when the goal changes
644  status = navQuery.FinalizeSlicedFindPath(reqPath, ref reqPathCount, MAX_RES).ToStatus();
645  }
646 
647  if (status != Status.Failure && reqPathCount > 0)
648  {
649  //in progress or succeed
650  if (reqPath[reqPathCount - 1] != agents[i].TargetRef)
651  {
652  //partial path, constrain target position in last polygon
653  bool tempBool;
654  status = navQuery.ClosestPointOnPoly(reqPath[reqPathCount - 1], agents[i].TargetPosition, out reqPos, out tempBool).ToStatus();
655  if (status == Status.Failure)
656  reqPathCount = 0;
657  }
658  else
659  {
660  reqPos = agents[i].TargetPosition;
661  }
662  }
663  else
664  {
665  reqPathCount = 0;
666  }
667 
668  if (reqPathCount == 0)
669  {
670  //could not find path, start the request from the current location
671  reqPos = agents[i].Position;
672  reqPath[0] = path[0];
673  reqPathCount = 1;
674  }
675 
676  agents[i].Corridor.SetCorridor(reqPos, reqPath, reqPathCount);
677  agents[i].Boundary.Reset();
678  agents[i].IsPartial = false;
679 
680  if (reqPath[reqPathCount - 1] == agents[i].TargetRef)
681  {
682  agents[i].TargetState = TargetState.Valid;
683  agents[i].TargetReplanTime = 0.0f;
684  }
685  else
686  {
687  //the path is longer or potentially unreachable, full plan
688  agents[i].TargetState = TargetState.WaitingForQueue;
689  }
690  }
691 
692  if (agents[i].TargetState == TargetState.WaitingForQueue)
693  {
694  numQueue = AddToPathQueue(agents[i], queue, numQueue, PATH_MAX_AGENTS);
695  }
696  }
697 
698  for (int i = 0; i < numQueue; i++)
699  {
700  queue[i].TargetPathqRef = pathq.Request(queue[i].Corridor.GetLastPoly(), queue[i].TargetRef, queue[i].Corridor.Target, queue[i].TargetPosition);
701  if (queue[i].TargetPathqRef != PathQueue.Invalid)
702  queue[i].TargetState = TargetState.WaitingForPath;
703  }
704 
705  //update requests
706  pathq.Update(MaxItersPerUpdate);
707 
708  //process path results
709  for (int i = 0; i < maxAgents; i++)
710  {
711  if (!agents[i].IsActive)
712  continue;
713  if (agents[i].TargetState == TargetState.None || agents[i].TargetState == TargetState.Velocity)
714  continue;
715 
716  if (agents[i].TargetState == TargetState.WaitingForPath)
717  {
718  //poll path queue
719  status = pathq.GetRequestStatus(agents[i].TargetPathqRef);
720  if (status == Status.Failure)
721  {
722  //path find failed, retry if the target location is still valid
723  agents[i].TargetPathqRef = PathQueue.Invalid;
724  if (agents[i].TargetRef != 0)
725  agents[i].TargetState = TargetState.Requesting;
726  else
727  agents[i].TargetState = TargetState.Failed;
728  agents[i].TargetReplanTime = 0.0f;
729  }
730  else if (status == Status.Success)
731  {
732  int[] path = agents[i].Corridor.Path;
733  int npath = agents[i].Corridor.PathCount;
734 
735  //apply results
