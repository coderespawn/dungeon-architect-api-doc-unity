115  public bool MovePosition(Vector3 npos, NavMeshQuery navquery)
116  {
117  //move along navmesh and update new position
118  Vector3 result = new Vector3();
119  const int MaxVisited = 16;
120  int[] visited = new int[MaxVisited];
121  List<int> listVisited = new List<int>(MaxVisited);
122  bool status = navquery.MoveAlongSurface(new NavPoint(path[0], pos), npos, ref result, listVisited);
123  visited = listVisited.ToArray();
124 
125  if (status == true)
126  {
127  pathCount = MergeCorridorStartMoved(path, pathCount, maxPath, visited, listVisited.Count);
128 
129  //adjust the position to stay on top of the navmesh
130  float h = pos.Y;
131  navquery.GetPolyHeight(path[0], result, ref h);
132  result.Y = h;
133  pos = result;
134  return true;
135  }
136

140  public int FindCorners(Vector3[] cornerVerts, int[] cornerFlags, int[] cornerPolys, int maxCorners, NavMeshQuery navquery)
141  {
142  const float MinTargetDist = 0.01f;
143 
144  int numCorners = 0;
145  navquery.FindStraightPath(pos, target, path, pathCount, cornerVerts, cornerFlags, cornerPolys, ref numCorners, maxCorners, 0);
146 
147  //prune points in the beginning of the path which are too close
148  while (numCorners > 0)
149  {
150  if (((cornerFlags[0] & PathfindingCommon.STRAIGHTPATH_OFFMESH_CONNECTION) != 0) ||
151  Vector3Extensions.Distance2D(cornerVerts[0], pos) > MinTargetDist)
152  break;
153  numCorners--;
154  if (numCorners > 0)
155  {
156  for (int i = 0; i < numCorners; i++)
157  {
158  cornerFlags[i] = cornerFlags[i + 1];
159  cornerPolys[i] = cornerPolys[i + 1];
160  cornerVerts[i] = cornerVerts[i + 1];
161  }
162  }
163  }
164 
165  //prune points after an off-mesh connection
166  for (int i = 0; i < numCorners; i++)
167  {
168  if ((cornerFlags[i] & PathfindingCommon.STRAIGHTPATH_OFFMESH_CONNECTION) != 0)
169  {
170  numCorners = i + 1;
171  break;
172  }
173  }
174 
175  return numCorners;
176  }

183  public bool OptimizePathTopology(NavMeshQuery navquery)
184  {
185  if (pathCount < 3)
186  return false;
187 
188  const int MaxIter = 32;
189  const int MaxRes = 32;
190 
191  int[] res = new int[MaxRes];
192  int numRes = 0;
193  int tempInt = 0;
194  navquery.InitSlicedFindPath(new NavPoint(path[0], pos), new NavPoint(path[pathCount - 1], target));
195  navquery.UpdateSlicedFindPath(MaxIter, ref tempInt);
196  bool status = navquery.FinalizedSlicedPathPartial(path, pathCount, res, ref numRes, MaxRes);
197 
198  if (status == true && numRes > 0)
199  {
200  pathCount = MergeCorridorStartShortcut(path, pathCount, maxPath, res, numRes);
201  return true;
202  }
203 
204  return false;

213  public void OptimizePathVisibility(Vector3 next, float pathOptimizationRange, NavMeshQuery navquery)
214  {
215  //clamp the ray to max distance
216  Vector3 goal = next;
217  float dist = Vector3Extensions.Distance2D(pos, goal);
218 
219  //if too close to the goal, do not try to optimize
220  if (dist < 0.01f)
221  return;
222 
223  dist = Math.Min(dist + 0.01f, pathOptimizationRange);
224 
225  //adjust ray length
226  Vector3 delta = goal - pos;
227  goal = pos + delta * (pathOptimizationRange / dist);
228 
229  const int MaxRes = 32;
230  int[] res = new int[MaxRes];
231  float t = 0;
232  Vector3 norm = new Vector3();
233  int nres = 0;
234  navquery.Raycast(new NavPoint(path[0], pos), goal, ref t, ref norm, res, ref nres, MaxRes);
235  if (nres > 1 && t > 0.99f)
236  {
237  pathCount = MergeCorridorStartShortcut(path, pathCount, maxPath, res, nres);
238  }
239  }

250  public int MergeCorridorStartMoved(int[] path, int npath, int maxPath, int[] visited, int nvisited)
251  {
252  int furthestPath = -1;
253  int furthestVisited = -1;
254 
255  //find furthest common polygon
256  for (int i = npath - 1; i >= 0; i--)
257  {
258  bool found = false;
259  for (int j = nvisited - 1; j >= 0; j--)
260  {
261  if (path[i] == visited[j])
262  {
263  furthestPath = i;
264  furthestVisited = j;
265  found = true;
266  }
267  }
268 
269  if (found)
270  break;
271  }
272 
273  //if no intersection found just return current path
274  if (furthestPath == -1 || furthestVisited == -1)
275  return npath;
276 
277  //concatenate paths
278 
279  //adjust beginning of buffer to include the visited
280  int req = nvisited - furthestVisited;
281  int orig = Math.Min(furthestPath + 1, npath);
282  int size = Math.Max(0, npath - orig);
283  if (req + size > maxPath)
284  size = maxPath - req;
285  if (size > 0)
286  {
287  for (int i = 0; i < size; i++)
288  path[req + i] = path[orig + i];
289  }
290 
291  //store visited
292  for (int i = 0; i < req; i++)
293  path[i] = visited[(nvisited - 1) - i];
294 
295  return req + size;
296  }

307  public int MergeCorridorStartShortcut(int[] path, int npath, int maxPath, int[] visited, int nvisited)
308  {
309  int furthestPath = -1;
310  int furthestVisited = -1;
311 
312  //find furthest common polygon
313  for (int i = npath - 1; i >= 0; i--)
314  {
315  bool found = false;
316  for (int j = nvisited - 1; j >= 0; j--)
317  {
318  if (path[i] == visited[j])
319  {
320  furthestPath = i;
321  furthestVisited = j;
322  found = true;
323  }
324  }
325 
326  if (found)
327  break;
328  }
329 
330  //if no intersection found, return current path
331  if (furthestPath == -1 || furthestVisited == -1)
332  return npath;
333 
334  //concatenate paths
335  //adjust beginning of the buffer to include the visited
336  int req = furthestVisited;
337  if (req <= 0)
338  return npath;
339 
340  int orig = furthestPath;
341  int size = Math.Max(0, npath - orig);
342  if (req + size > maxPath)
343  size = maxPath - req;
344  for (int i = 0; i < size; i++)
345  path[req + i] = path[orig + i];
346 
347  //store visited
348  for (int i = 0; i < req; i++)
349  path[i] = visited[i];
350 
351  return req + size;
352  }

354  public bool MoveOverOffmeshConnection(int offMeshConRef, int[] refs, ref Vector3 startPos, ref Vector3 endPos, NavMeshQuery navquery)
355  {
356  //advance the path up to and over the off-mesh connection
357  int prevRef = 0, polyRef = path[0];
358  int npos = 0;
359  while (npos < pathCount && polyRef != offMeshConRef)
360  {
361  prevRef = polyRef;
362  polyRef = path[npos];
363  npos++;
364  }
365 
366  if (npos == pathCount)
367  {
368  //could not find offMeshConRef
369  return false;
370  }
371 
372  //prune path
373  for (int i = npos; i < pathCount; i++)
374  path[i - npos] = path[i];
375  pathCount -= npos;
376 
377  refs[0] = prevRef;
378  refs[1] = polyRef;
379 
380  TiledNavMesh nav = navquery.NavMesh;
381 
382  if (nav.GetOffMeshConnectionPolyEndPoints(refs[0], refs[1], ref startPos, ref endPos) == true)
383  {
384  pos = endPos;
385  return true;
386  }
387 
388  return false;
389  }
