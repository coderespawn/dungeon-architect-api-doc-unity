140  internal static bool SegmentPoly2D(Vector3 p0, Vector3 p1, Vector3[] verts, int nverts, out float tmin, out float tmax, out int segMin, out int segMax)
141  {
142  const float Epsilon = 0.00000001f;
143 
144  tmin = 0;
145  tmax = 1;
146  segMin = -1;
147  segMax = -1;
148 
149  Vector3 dir = p1 - p0;
150 
151  for (int i = 0, j = nverts - 1; i < nverts; j = i++)
152  {
153  Vector3 edge = verts[i] - verts[j];
154  Vector3 diff = p0 - verts[j];
155  float n = edge.Z * diff.X - edge.X * diff.Z;
156  float d = dir.Z * edge.X - dir.X * edge.Z;
157  if (Math.Abs(d) < Epsilon)
158  {
159  //S is nearly parallel to this edge
160  if (n < 0)
161  return false;
162  else
163  continue;
164  }
165 
166  float t = n / d;
167  if (d < 0)
168  {
169  //segment S is entering across this edge
170  if (t > tmin)
171  {
172  tmin = t;
173  segMin = j;
174 
175  //S enters after leaving the polygon
176  if (tmin > tmax)
177  return false;
178  }
179  }
180  else
181  {
182  //segment S is leaving across this edge
183  if (t < tmax)
184  {
185  tmax = t;
186  segMax = j;
187 
188  //S leaves before entering the polygon
189  if (tmax < tmin)
190  return false;
191  }
192  }
193  }
194 
195  return true;
196  }

199  {

201  Vector3Extensions.Dot2D(ref axis, ref poly[0], out rmax);
