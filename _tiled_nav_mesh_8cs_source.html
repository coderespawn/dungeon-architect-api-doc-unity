110  public bool InitTileNavMesh(TiledNavMeshParams parameters)
111  {
112  this.parameters = parameters;
113  origin = this.parameters.Origin;
114  tileWidth = parameters.TileWidth;
115  tileHeight = parameters.TileHeight;
116 
117  //init tiles
118  maxTiles = parameters.MaxTiles;
119  tileLookupTableSize = MathHelper.NextPowerOfTwo(parameters.MaxTiles / 4);
120  if (tileLookupTableSize == 0)
121  tileLookupTableSize = 1;
122  tileLookupTableMask = tileLookupTableSize - 1;
123 
124  tiles = new MeshTile[maxTiles];
125  posLookup = new MeshTile[tileLookupTableSize];
126  for (int i = 0; i < tiles.Length; i++)
127  tiles[i] = new MeshTile();
128  for (int i = 0; i < posLookup.Length; i++)
129  posLookup[i] = null;
130 
131  //create a linked list of tiles
132  nextFree = null;
133  for (int i = maxTiles - 1; i >= 0; i--)
134  {
135  tiles[i].Salt = 1;
136  tiles[i].Next = nextFree;
137  nextFree = tiles[i];
138  }
139 
140  //init ID generator values
141  tileBits = MathHelper.Log2(MathHelper.NextPowerOfTwo(parameters.MaxTiles));
142  polyBits = MathHelper.Log2(MathHelper.NextPowerOfTwo(parameters.MaxPolys));
143 
144  //only allow 31 salt bits, since salt mask is calculated using 32-bit int and it will overflow
145  saltBits = Math.Min(31, 32 - tileBits - polyBits);
146  if (saltBits < 10)
147  return false;
148 
149  return true;
150  }

159  public void AddTile(NavMeshBuilder data, int lastRef, ref int result)
160  {
161  //make sure data is in right format
162  PathfindingCommon.NavMeshInfo header = data.Header;
163 
164  //make sure location is free
165  if (GetTileAt(header.X, header.Y, header.Layer) != null)
166  return;
167 
168  //allocate a tile
169  MeshTile tile = null;
170  if (lastRef == 0)
171  {
172  if (nextFree != null)
173  {
174  tile = nextFree;
175  nextFree = tile.Next;
176  tile.Next = null;
177  }
178  }
179  else
180  {
181  //try to relocate tile to specific index with the same salt
182  int tileIndex = DecodePolyIdTile(lastRef);
183  if (tileIndex >= maxTiles)
184  return;
185 
186  //try to find specific tile id from free list
187  MeshTile target = tiles[tileIndex];
188  MeshTile prev = null;
189  tile = nextFree;
190  while (tile != null && tile != target)
191  {
192  prev = tile;
193  tile = tile.Next;
194  }
195 
196  //couldn't find correct location
197  if (tile != target)
198  return;
199 
200  //remove from freelist
201  if (prev == null)
202  nextFree = tile.Next;
203  else
204  prev.Next = tile.Next;
205 
206  //restore salt
207  tile.Salt = DecodePolyIdSalt(lastRef);
208  }
209 
210  //make sure we could allocate a tile
211  if (tile == null)
212  return;
213 
214  //insert tile into position LookUp Table (lut)
215  int h = ComputeTileHash(header.X, header.Y, tileLookupTableMask);
216  tile.Next = posLookup[h];
217  posLookup[h] = tile;
218 
219  if (header.BvNodeCount == 0)
220  tile.BVTree = null;
221 
222  //patch header
223  tile.Verts = data.NavVerts;
224  tile.Polys = data.NavPolys;
225  tile.DetailMeshes = data.NavDMeshes;
226  tile.DetailVerts = data.NavDVerts;
227  tile.DetailTris = data.NavDTris;
228  tile.BVTree = data.NavBvTree;
229  tile.OffMeshConnections = data.OffMeshCons;
230 
231  //build links freelist
232  tile.LinksFreeList = 0;
233  tile.Links = new Link[header.MaxLinkCount];
234  for (int i = 0; i < header.MaxLinkCount; i++)
235  tile.Links[i] = new Link();
236 
237  tile.Links[header.MaxLinkCount - 1].Next = Link.Null;
238  for (int i = 0; i < header.MaxLinkCount - 1; i++)
239  tile.Links[i].Next = i + 1;
240 
241  //init tile
242  tile.Header = header;
243  tile.Data = data;
244 
245  ConnectIntLinks(ref tile);
246  BaseOffMeshLinks(ref tile);
247 
248  //create connections with neighbor tiles
249  MeshTile[] neis = new MeshTile[32];
250  int nneis;
251 
252  //connect with layers in current tile
253  nneis = GetTilesAt(header.X, header.Y, neis);
254  for (int j = 0; j < nneis; j++)
255  {
256  if (neis[j] != tile)
257  {
258  ConnectExtLinks(ref tile, ref neis[j], BoundarySide.Internal);
259  ConnectExtLinks(ref neis[j], ref tile, BoundarySide.Internal);
260  }
261 
262  ConnectExtOffMeshLinks(ref tile, ref neis[j], BoundarySide.Internal);
263  ConnectExtOffMeshLinks(ref neis[j], ref tile, BoundarySide.Internal);
264  }
265 
266  //connect with neighbour tiles
267  for (int i = 0; i < 8; i++)
268  {
269  BoundarySide b = (BoundarySide)i;
270  BoundarySide bo = b.GetOpposite();
271  nneis = GetNeighbourTilesAt(header.X, header.Y, b, neis);
272  for (int j = 0; j < nneis; j++)
273  {
274  ConnectExtLinks(ref tile, ref neis[j], b);
275  ConnectExtLinks(ref neis[j], ref tile, bo);
276  ConnectExtOffMeshLinks(ref tile, ref neis[j], b);
277  ConnectExtOffMeshLinks(ref neis[j], ref tile, bo);
278  }
279  }
280 
281  result = GetTileRef(tile);
282  }

288  public void ConnectIntLinks(ref MeshTile tile)
289  {
290  if (tile == null)
291  return;
292 
293  int polyBase = GetPolyRefBase(tile);
294 
295  //Iterate through all the polygons
296  for (int i = 0; i < tile.Header.PolyCount; i++)
297  {
298  //The polygon links will end in a null link
299  tile.Polys[i].FirstLink = Link.Null;
300 
301  //Avoid Off-Mesh Connection polygons
302  if (tile.Polys[i].PolyType == PolygonType.OffMeshConnection)
303  continue;
304 
305  //Build edge links
306  for (int j = tile.Polys[i].VertCount - 1; j >= 0; j--)
307  {
308  //Skip hard and non-internal edges
309  if (tile.Polys[i].Neis[j] == 0 || IsExternalLink(tile.Polys[i].Neis[j]))
310  continue;
311 
312  //Allocate a new link if possible
313  int idx = AllocLink(tile);
314 
315  //Allocation of link should be successful
316  if (IsLinkAllocated(idx))
317  {
318  //Initialize a new link
319  tile.Links[idx].Reference = GetReference(polyBase, tile.Polys[i].Neis[j] - 1);
320  tile.Links[idx].Edge = j;
321  tile.Links[idx].Side = BoundarySide.Internal;
322  tile.Links[idx].BMin = tile.Links[idx].BMax = 0;
323 
324  //Add to polygon's links list
325  tile.Links[idx].Next = tile.Polys[i].FirstLink;
326  tile.Polys[i].FirstLink = idx;
327  }
328  }
329  }

336  public void BaseOffMeshLinks(ref MeshTile tile)
337  {
338  if (tile == null)
339  return;
340 
341  int polyBase = GetPolyRefBase(tile);
342 
343  //Base off-mesh connection start points
344  for (int i = 0; i < tile.Header.OffMeshConCount; i++)
345  {
346  int con = i;
347  int poly = tile.OffMeshConnections[con].Poly;
348 
349  Vector3 extents = new Vector3(tile.OffMeshConnections[con].Radius, tile.Header.WalkableClimb, tile.OffMeshConnections[con].Radius);
350 
351  //Find polygon to connect to
352  Vector3 p = tile.OffMeshConnections[con].Pos0;
353  Vector3 nearestPt = new Vector3();
354  int reference = FindNearestPolyInTile(tile, p, extents, ref nearestPt);
355  if (reference == 0)
356  continue;
357 
358  //Do extra checks
359  if ((nearestPt.X - p.X) * (nearestPt.X - p.X) + (nearestPt.Z - p.Z) * (nearestPt.Z - p.Z) >
360  tile.OffMeshConnections[con].Radius * tile.OffMeshConnections[con].Radius)
361  continue;
362 
363  //Make sure location is on current mesh
364  tile.Verts[tile.Polys[poly].Verts[0]] = nearestPt;
365 
366  //Link off-mesh connection to target poly
367  int idx = AllocLink(tile);
368  if (IsLinkAllocated(idx))
369  {
370  //Initialize a new link
371  tile.Links[idx].Reference = reference;
372  tile.Links[idx].Edge = 0;
373  tile.Links[idx].Side = BoundarySide.Internal;
374  tile.Links[idx].BMin = tile.Links[idx].BMax = 0;
375 
376  //Add to polygon's links list
377  tile.Links[idx].Next = tile.Polys[poly].FirstLink;
378  tile.Polys[poly].FirstLink = idx;
379  }
380 
381  //Start end-point always conects back to off-mesh connection
382  int tidx = AllocLink(tile);
383  if (IsLinkAllocated(tidx))
384  {
385  //Initialize a new link
386  int landPolyIdx = DecodePolyIdPoly(reference);
387  tile.Links[idx].Reference = GetReference(polyBase, tile.OffMeshConnections[con].Poly);
388  tile.Links[idx].Edge = 0xff;
389  tile.Links[idx].Side = BoundarySide.Internal;
390  tile.Links[idx].BMin = tile.Links[idx].BMax = 0;
391 
392  //Add to polygon's links list
393  tile.Links[idx].Next = tile.Polys[landPolyIdx].FirstLink;
394  tile.Polys[landPolyIdx].FirstLink = tidx;
395  }
396  }

405  public void ConnectExtLinks(ref MeshTile tile, ref MeshTile target, BoundarySide side)
406  {
407  if (tile == null)
408  return;
409 
410  //Connect border links
411  for (int i = 0; i < tile.Header.PolyCount; i++)
412  {
413  int numPolyVerts = tile.Polys[i].VertCount;
414 
415  for (int j = 0; j < numPolyVerts; j++)
416  {
417  //Skip non-portal edges
418  if ((tile.Polys[i].Neis[j] & Link.External) == 0)
419  continue;
420 
421  BoundarySide dir = (BoundarySide)(tile.Polys[i].Neis[j] & 0xff);
422  if (side != BoundarySide.Internal && dir != side)
423  continue;
424 
425  //Create new links
426  Vector3 va = tile.Verts[tile.Polys[i].Verts[j]];
427  Vector3 vb = tile.Verts[tile.Polys[i].Verts[(j + 1) % numPolyVerts]];
428  List<int> nei = new List<int>(4);
429  List<float> neia = new List<float>(4 * 2);
430  FindConnectingPolys(va, vb, target, dir.GetOpposite(), nei, neia);
431 
432  //Iterate through neighbors
433  for (int k = 0; k < nei.Count; k++)
434  {
435  //Allocate a new link if possible
436  int idx = AllocLink(tile);
437 
438  if (IsLinkAllocated(idx))
439  {
440  tile.Links[idx].Reference = nei[k];
441  tile.Links[idx].Edge = j;
442  tile.Links[idx].Side = dir;
443 
444  tile.Links[idx].Next = tile.Polys[i].FirstLink;
445  tile.Polys[i].FirstLink = idx;
446 
447  //Compress portal limits to a value
448  if (dir == BoundarySide.PlusX || dir == BoundarySide.MinusX)
449  {
450  float tmin = (neia[k * 2 + 0] - va.Z) / (vb.Z - va.Z);
451  float tmax = (neia[k * 2 + 1] - va.Z) / (vb.Z - va.Z);
452 
453  if (tmin > tmax)
454  {
455  float temp = tmin;
456  tmin = tmax;
457  tmax = temp;
458  }
459 
460  tile.Links[idx].BMin = (int)(MathHelper.Clamp(tmin, 0.0f, 1.0f) * 255.0f);
461  tile.Links[idx].BMax = (int)(MathHelper.Clamp(tmax, 0.0f, 1.0f) * 255.0f);
462  }
463  else if (dir == BoundarySide.PlusZ || dir == BoundarySide.MinusZ)
464  {
465  float tmin = (neia[k * 2 + 0] - va.X) / (vb.X - va.X);
466  float tmax = (neia[k * 2 + 1] - va.X) / (vb.X - va.X);
467 
468  if (tmin > tmax)
469  {
470  float temp = tmin;
471  tmin = tmax;
472  tmax = temp;
473  }
474 
475  tile.Links[idx].BMin = (int)(MathHelper.Clamp(tmin, 0.0f, 1.0f) * 255.0f);
476  tile.Links[idx].BMax = (int)(MathHelper.Clamp(tmax, 0.0f, 1.0f) * 255.0f);
477  }
478  }
479  }
480  }
481  }

490  public void ConnectExtOffMeshLinks(ref MeshTile tile, ref MeshTile target, BoundarySide side)
491  {
492  if (tile == null)
493  return;
494 
495  //Connect off-mesh links, specifically links which land from target tile to this tile
496  BoundarySide oppositeSide = side.GetOpposite();
497 
498  //Iterate through all the off-mesh connections of target tile
499  for (int i = 0; i < target.Header.OffMeshConCount; i++)
500  {
501  OffMeshConnection targetCon = target.OffMeshConnections[i];
502  if (targetCon.Side != oppositeSide)
503  continue;
504 
505  Poly targetPoly = target.Polys[targetCon.Poly];
506 
507  //Skip off-mesh connections which start location could not be connected at all
508  if (!IsLinkAllocated(targetPoly.FirstLink))
509  continue;
510 
511  Vector3 extents = new Vector3(targetCon.Radius, target.Header.WalkableClimb, targetCon.Radius);
512 
513  //Find polygon to connect to
514  Vector3 p = targetCon.Pos1;
515  Vector3 nearestPt = new Vector3();
516  int reference = FindNearestPolyInTile(tile, p, extents, ref nearestPt);
517  if (reference == 0)
518  continue;
519 
520  //Further checks
521  if ((nearestPt.X - p.X) * (nearestPt.X - p.X) + (nearestPt.Z - p.Z) * (nearestPt.Z - p.Z) >
522  (targetCon.Radius * targetCon.Radius))
523  continue;
524 
525  //Make sure the location is on the current mesh
526  target.Verts[targetPoly.Verts[1]] = nearestPt;
527 
528  //Link off-mesh connection to target poly
529  int idx = AllocLink(target);
530  if (IsLinkAllocated(idx))
531  {
532  target.Links[idx].Reference = reference;
533  target.Links[idx].Edge = i;
534  target.Links[idx].Side = oppositeSide;
535  target.Links[idx].BMin = target.Links[idx].BMax = 0;
536 
537  //add to linked list
538  target.Links[idx].Next = target.Polys[i].FirstLink;
539  target.Polys[i].FirstLink = idx;
540  }
541 
542  //link target poly to off-mesh connection
543  if ((targetCon.Flags & OffMeshConnectionFlags.Bidirectional) != 0)
544  {
545  int tidx = AllocLink(tile);
546  if (IsLinkAllocated(tidx))
547  {
548  int landPolyIdx = DecodePolyIdPoly(reference);
549  tile.Links[tidx].Reference = GetReference(GetPolyRefBase(target), targetCon.Poly);
550  tile.Links[tidx].Edge = 0xff;
551  tile.Links[tidx].Side = side;
552  tile.Links[tidx].BMin = tile.Links[tidx].BMax = 0;
553 
554  //add to linked list
555  tile.Links[tidx].Next = tile.Polys[landPolyIdx].FirstLink;
556  tile.Polys[landPolyIdx].FirstLink = tidx;
557  }
558  }
559  }

570  public bool GetOffMeshConnectionPolyEndPoints(int prevRef, int polyRef, ref Vector3 startPos, ref Vector3 endPos)
571  {
572  int salt = 0, indexTile = 0, indexPoly = 0;
573 
574  if (polyRef == 0)
575  return false;
576 
577  //get current polygon
578  DecodePolyId(polyRef, ref salt, ref indexTile, ref indexPoly);
579  if (indexTile >= maxTiles)
580  return false;
581  if (tiles[indexTile].Salt != salt || tiles[indexTile].Header == null)
582  return false;
583  MeshTile tile = tiles[indexTile];
584  if (indexPoly >= tile.Header.PolyCount)
585  return false;
586  Poly poly = tile.Polys[indexPoly];
587 
588  if (poly.PolyType != PolygonType.OffMeshConnection)
589  return false;
590 
591  int idx0 = 0, idx1 = 1;
592 
593  //find the link that points to the first vertex
594  for (int i = poly.FirstLink; i != Link.Null; i = tile.Links[i].Next)
595  {
596  if (tile.Links[i].Edge == 0)
597  {
598  if (tile.Links[i].Reference != prevRef)
599  {
600  idx0 = 1;
601  idx1 = 0;
602  }
603 
604  break;
605  }
606  }
607 
608  startPos = tile.Verts[poly.Verts[idx0]];
609  endPos = tile.Verts[poly.Verts[idx1]];
610 
611  return true;
612  }

623  public void FindConnectingPolys(Vector3 va, Vector3 vb, MeshTile tile, BoundarySide side, List<int> con, List<float> conarea)
624  {
625  if (tile == null)
626  return;
627 
628  Vector2 amin = Vector2.Zero;
629  Vector2 amax = Vector2.Zero;
630  CalcSlabEndPoints(va, vb, amin, amax, side);
631  float apos = GetSlabCoord(va, side);
632 
633  //Remove links pointing to 'side' and compact the links array
634  Vector2 bmin = Vector2.Zero;
635  Vector2 bmax = Vector2.Zero;
636 
637  int polyBase = GetPolyRefBase(tile);
638 
639  //Iterate through all the tile's polygons
640  for (int i = 0; i < tile.Header.PolyCount; i++)
641  {
642  int numPolyVerts = tile.Polys[i].VertCount;
643 
644  //Iterate through all the vertices
645  for (int j = 0; j < numPolyVerts; j++)
646  {
647  //Skip edges which do not point to the right side
648  if (tile.Polys[i].Neis[j] != (Link.External | (int)side))
649  continue;
650 
651  //Grab two adjacent vertices
652  Vector3 vc = tile.Verts[tile.Polys[i].Verts[j]];
653  Vector3 vd = tile.Verts[tile.Polys[i].Verts[(j + 1) % numPolyVerts]];
654  float bpos = GetSlabCoord(vc, side);
655 
656  //Segments are not close enough
657  if (Math.Abs(apos - bpos) > 0.01f)
658  continue;
659 
660  //Check if the segments touch
661  CalcSlabEndPoints(vc, vd, bmin, bmax, side);
662 
663  //Skip if slabs don't overlap
664  if (!OverlapSlabs(amin, amax, bmin, bmax, 0.01f, tile.Header.WalkableClimb))
665  continue;
666 
667  //Add return value
668  if (con.Count < con.Capacity)
669  {
670  conarea.Add(Math.Max(amin.X, bmin.X));
671  conarea.Add(Math.Min(amax.X, bmax.X));
672  con.Add(GetReference(polyBase, i));
673  }
674 
675  break;
676  }
677  }
678  }

688  public void CalcSlabEndPoints(Vector3 va, Vector3 vb, Vector2 bmin, Vector2 bmax, BoundarySide side)
689  {
690  if (side == BoundarySide.PlusX || side == BoundarySide.MinusX)
691  {
692  if (va.Z < vb.Z)
693  {
694  bmin.X = va.Z;
695  bmin.Y = va.Y;
696 
697  bmax.X = vb.Z;
698  bmax.Y = vb.Y;
699  }
700  else
701  {
702  bmin.X = vb.Z;
703  bmin.Y = vb.Y;
704 
705  bmax.X = va.Z;
706  bmax.Y = va.Y;
707  }
708  }
709  else if (side == BoundarySide.PlusZ || side == BoundarySide.MinusZ)
710  {
711  if (va.X < vb.X)
712  {
713  bmin.X = va.X;
714  bmin.Y = va.Y;
715 
716  bmax.X = vb.X;
717  bmax.Y = vb.Y;
718  }
719  else
720  {
721  bmin.X = vb.X;
722  bmin.Y = vb.Y;
723 
724  bmax.X = va.X;
725  bmax.Y = va.Y;
726  }
727  }

756  public bool OverlapSlabs(Vector2 amin, Vector2 amax, Vector2 bmin, Vector2 bmax, float px, float py)
757  {
758  //Check for horizontal overlap
759  //Segment shrunk a little so that slabs which touch at endpoints aren't connected
760  float minX = Math.Max(amin.X + px, bmin.X + px);
761  float maxX = Math.Min(amax.X - px, bmax.X - px);
762  if (minX > maxX)
763  return false;
764 
765  //Check vertical overlap
766  float leftSlope = (amax.Y - amin.Y) / (amax.X - amin.X);
767  float leftConstant = amin.Y - leftSlope * amin.X;
768  float rightSlope = (bmax.Y - bmin.Y) / (bmax.X - bmin.X);
769  float rightConstant = bmin.Y - rightSlope * bmin.X;
770  float leftMinY = leftSlope * minX + leftConstant;
771  float leftMaxY = leftSlope * maxX + leftConstant;
772  float rightMinY = rightSlope * minX + rightConstant;
773  float rightMaxY = rightSlope * maxX + rightConstant;
774  float dmin = rightMinY - leftMinY;
775  float dmax = rightMaxY - leftMaxY;
776 
777  //Crossing segments always overlap
778  if (dmin * dmax < 0)
779  return true;
780 
781  //Check for overlap at endpoints

836  public int QueryPolygonsInTile(MeshTile tile, BBox3 qbounds, List<int> polys)
837  {
838  if (tile.BVTree.Count != 0)
839  {
840  int node = 0;
841  int end = tile.Header.BvNodeCount;
842  Vector3 tbmin = tile.Header.Bounds.Min;
843  Vector3 tbmax = tile.Header.Bounds.Max;
844 
845  //Clamp query box to world box
846  Vector3 qbmin = qbounds.Min;
847  Vector3 qbmax = qbounds.Max;
848  PolyBounds b;
849  float bminx = MathHelper.Clamp(qbmin.X, tbmin.X, tbmax.X) - tbmin.X;
850  float bminy = MathHelper.Clamp(qbmin.Y, tbmin.Y, tbmax.Y) - tbmin.Y;
851  float bminz = MathHelper.Clamp(qbmin.Z, tbmin.Z, tbmax.Z) - tbmin.Z;
852  float bmaxx = MathHelper.Clamp(qbmax.X, tbmin.X, tbmax.X) - tbmin.X;
853  float bmaxy = MathHelper.Clamp(qbmax.Y, tbmin.Y, tbmax.Y) - tbmin.Y;
854  float bmaxz = MathHelper.Clamp(qbmax.Z, tbmin.Z, tbmax.Z) - tbmin.Z;
855 
856  const int MinMask = unchecked((int)0xfffffffe);
857 
858  b.Min.X = (int)(bminx * tile.Header.BvQuantFactor) & MinMask;
859  b.Min.Y = (int)(bminy * tile.Header.BvQuantFactor) & MinMask;
860  b.Min.Z = (int)(bminz * tile.Header.BvQuantFactor) & MinMask;
861  b.Max.X = (int)(bmaxx * tile.Header.BvQuantFactor + 1) | 1;
862  b.Max.Y = (int)(bmaxy * tile.Header.BvQuantFactor + 1) | 1;
863  b.Max.Z = (int)(bmaxz * tile.Header.BvQuantFactor + 1) | 1;
864 
865  //traverse tree
866  int polyBase = GetPolyRefBase(tile);
867 
868  while (node < end)
869  {
870  BVTree.Node bvNode = tile.BVTree[node];
871  bool overlap = PolyBounds.Overlapping(ref b, ref bvNode.Bounds);
872  bool isLeafNode = bvNode.Index >= 0;
873 
874  if (isLeafNode && overlap)
875  {
876  if (polys.Count < polys.Capacity)
877  polys.Add(GetReference(polyBase, bvNode.Index));
878  }
879 
880  if (overlap || isLeafNode)
881  {
882  node++;
883  }
884  else
885  {
886  int escapeIndex = -bvNode.Index;
887  node += escapeIndex;
888  }
889  }
890 
891  return polys.Count;
892  }
893  else
894  {
895  BBox3 b;
896  int polyBase = GetPolyRefBase(tile);
897 
898  for (int i = 0; i < tile.Header.PolyCount; i++)
899  {
900  var poly = tile.Polys[i];
901 
902  //don't return off-mesh connection polygons
903  if (poly.PolyType == PolygonType.OffMeshConnection)
904  continue;
905 
906  //calculate polygon bounds
907  b.Max = b.Min = tile.Verts[poly.Verts[0]];
908  for (int j = 1; j < poly.VertCount; j++)
909  {
910  Vector3 v = tile.Verts[poly.Verts[j]];
911  Vector3Extensions.ComponentMin(ref b.Min, ref v, out b.Min);
912  Vector3Extensions.ComponentMax(ref b.Max, ref v, out b.Max);
913  }
914 
915  if (BBox3.Overlapping(ref qbounds, ref b))
916  {
917  if (polys.Count < polys.Capacity)
918  polys.Add(GetReference(polyBase, i));
919  }
920  }
921 
922  return polys.Count;
923  }
