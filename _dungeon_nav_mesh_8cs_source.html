125  void BuildVisualization() {
126  //if (visualization == null)
127  {
128  var filter = GetComponent<MeshFilter>();
129  visualization = new Mesh();
130  filter.mesh = visualization;
131  }
132  if (polyMesh == null) return;

136 
137  var tile = navMesh.GetTileAt(0, 0, 0);
138  var vertices = new List<Vector3>();
139  var triangles = new List<int>();
140 
141  for (int i = 0; i < tile.Polys.Length; i++)
142  {
143  for (int j = 1; j + 1 < tile.Polys[i].VertCount; j++)
144  {
145  int vertIndex0 = tile.Polys[i].Verts[0];
146  int vertIndex1 = tile.Polys[i].Verts[j];
147  int vertIndex2 = tile.Polys[i].Verts[j + 1];
148 
149  var v = tile.Verts[vertIndex0];
150  var v0 = ToV3(v);
151 
152  v = tile.Verts[vertIndex1];
153  var v1 = ToV3(v);
154 
155  v = tile.Verts[vertIndex2];
156  var v2 = ToV3(v);
157 
158  var offset = vertices.Count;
159  vertices.Add(v0);
160  vertices.Add(v1);
161  vertices.Add(v2);
162 
163  triangles.Add (offset);
164  triangles.Add (offset + 1);
165  triangles.Add (offset + 2);
166  }
167  }
168 
169  if (visualize2D) {
170  // Flip YZ
171  float t;
172  for (int i = 0; i < vertices.Count; i++) {
173  var v = vertices[i];
174  t = v.y;
175  v.y = v.z;
176  v.z = t;
177  vertices[i] = v;
178  }
179  }
180 
181  visualization.vertices = vertices.ToArray();
182  visualization.SetIndices(triangles.ToArray(), MeshTopology.Triangles, 0);
183  }
