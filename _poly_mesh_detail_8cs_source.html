65  public PolyMeshDetail(PolyMesh mesh, CompactHeightfield compactField, float sampleDist, float sampleMaxError)
66  {
67  if (mesh.VertCount == 0 || mesh.PolyCount == 0)
68  return;
69 
70  Vector3 origin = mesh.Bounds.Min;
71 
72  int maxhw = 0, maxhh = 0;
73 
74  BBox2i[] bounds = new BBox2i[mesh.PolyCount];
75  Vector3[] poly = new Vector3[mesh.NumVertsPerPoly];
76 
77  var storedVertices = new List<Vector3>();
78  var storedTriangles = new List<TriangleData>();
79 
80  //find max size for polygon area
81  for (int i = 0; i < mesh.PolyCount; i++)
82  {
83  var p = mesh.Polys[i];
84 
85  int xmin = compactField.Width;
86  int xmax = 0;
87  int zmin = compactField.Length;
88  int zmax = 0;
89 
90  for (int j = 0; j < mesh.NumVertsPerPoly; j++)
91  {
92  var pj = p.Vertices[j];
93  if (pj == PolyMesh.NullId)
94  break;
95 
96  var v = mesh.Verts[pj];
97 
98  xmin = Math.Min(xmin, v.X);
99  xmax = Math.Max(xmax, v.X);
100  zmin = Math.Min(zmin, v.Z);
101  zmax = Math.Max(zmax, v.Z);
102  }
103 
104  xmin = Math.Max(0, xmin - 1);
105  xmax = Math.Min(compactField.Width, xmax + 1);
106  zmin = Math.Max(0, zmin - 1);
107  zmax = Math.Min(compactField.Length, zmax + 1);
108 
109  if (xmin >= xmax || zmin >= zmax)
110  continue;
111 
112  maxhw = Math.Max(maxhw, xmax - xmin);
113  maxhh = Math.Max(maxhh, zmax - zmin);
114 
115  bounds[i] = new BBox2i(xmin, zmin, xmax, zmax);
116  }
117 
118  HeightPatch hp = new HeightPatch(0, 0, maxhw, maxhh);
119 
120  this.meshes = new MeshData[mesh.PolyCount];
121 
122  for (int i = 0; i < mesh.PolyCount; i++)
123  {
124  var p = mesh.Polys[i];
125 
126  //store polygon vertices for processing
127  int npoly = 0;
128  for (int j = 0; j < mesh.NumVertsPerPoly; j++)
129  {
130  int pvi = p.Vertices[j];
131  if (pvi == PolyMesh.NullId)
132  break;
133 
134  PolyVertex pv = mesh.Verts[pvi];
135  Vector3 v = new Vector3(pv.X, pv.Y, pv.Z);
136  v.X *= mesh.CellSize;
137  v.Y *= mesh.CellHeight;
138  v.Z *= mesh.CellSize;
139  poly[j] = v;
140  npoly++;
141  }
142 
143  //get height data from area of polygon
144  BBox2i bound = bounds[i];
145  hp.Resize(bound.Min.X, bound.Min.Y, bound.Max.X - bound.Min.X, bound.Max.Y - bound.Min.Y);
146  GetHeightData(compactField, p, npoly, mesh.Verts, mesh.BorderSize, hp);
147 
148  List<Vector3> tempVerts = new List<Vector3>();
149  List<TriangleData> tempTris = new List<TriangleData>(128);
150  List<EdgeInfo> edges = new List<EdgeInfo>(16);
151  List<SamplingData> samples = new List<SamplingData>(128);
152  BuildPolyDetail(poly, npoly, sampleDist, sampleMaxError, compactField, hp, tempVerts, tempTris, edges, samples);
153 
154  //more detail verts
155  for (int j = 0; j < tempVerts.Count; j++)
156  {
157  Vector3 tv = tempVerts[j];
158 
159  Vector3 v;
160  v.X = tv.X + origin.X;
161  v.Y = tv.Y + origin.Y + compactField.CellHeight;
162  v.Z = tv.Z + origin.Z;
163 
164  tempVerts[j] = v;
165  }
166 
167  for (int j = 0; j < npoly; j++)
168  {
169  Vector3 po = poly[j];
170 
171  po.X += origin.X;
172  po.Y += origin.Y;
173  po.Z += origin.Z;
174 
175  poly[j] = po;
176  }
177 
178  //save data
179  this.meshes[i].VertexIndex = storedVertices.Count;
180  this.meshes[i].VertexCount = tempVerts.Count;
181  this.meshes[i].TriangleIndex = storedTriangles.Count;
182  this.meshes[i].TriangleCount = tempTris.Count;
183 
184  //store vertices
185  storedVertices.AddRange(tempVerts);
186 
187  //store triangles
188  for (int j = 0; j < tempTris.Count; j++)
189  {
190  storedTriangles.Add(new TriangleData(tempTris[j], tempVerts, poly, npoly));
191  }
192  }
193 
194  this.verts = storedVertices.ToArray();
195  this.tris = storedTriangles.ToArray();
196  }

320  private static int GetEdgeFlags(Vector3 va, Vector3 vb, Vector3[] vpoly, int npoly)
321  {
322  //true if edge is part of polygon
323  float thrSqr = 0.001f * 0.001f;
324 
325  for (int i = 0, j = npoly - 1; i < npoly; j = i++)
326  {
327  Vector3 pt1 = va;
328  Vector3 pt2 = vb;
329 
330  //the vertices pt1 (va) and pt2 (vb) are extremely close to the polygon edge
331  if (Distance.PointToSegment2DSquared(ref pt1, ref vpoly[j], ref vpoly[i]) < thrSqr
332  && Distance.PointToSegment2DSquared(ref pt2, ref vpoly[j], ref vpoly[i]) < thrSqr)
333  return 1;
334  }
335 
336  return 0;
337  }

339  private void GetHeightData(CompactHeightfield compactField, PolyMesh.Polygon poly, int polyCount, PolyVertex[] verts, int borderSize, HeightPatch hp)
340  {
341  var stack = new List<CompactSpanReference>();
342  bool empty = true;
343  hp.Clear();
344 
345  for (int y = 0; y < hp.Length; y++)
346  {
347  int hy = hp.Y + y + borderSize;
348  for (int x = 0; x < hp.Width; x++)
349  {
350  int hx = hp.X + x + borderSize;
351  var cells = compactField.Cells[hy * compactField.Width + hx];
352  for (int i = cells.StartIndex, end = cells.StartIndex + cells.Count; i < end; i++)
353  {
354  var span = compactField.Spans[i];
355 
356  if (span.Region == poly.RegionId)
357  {
358  hp[x, y] = span.Minimum;
359  empty = false;
360 
361  bool border = false;
362  for (var dir = Direction.West; dir <= Direction.South; dir++)
363  {
364  if (span.IsConnected(dir))
365  {
366  int ax = hx + dir.GetHorizontalOffset();
367  int ay = hy + dir.GetVerticalOffset();
368  int ai = compactField.Cells[ay * compactField.Width + ax].StartIndex + CompactSpan.GetConnection(ref span, dir);
369 
370  if (compactField.Spans[ai].Region != poly.RegionId)
371  {
372  border = true;
373  break;
374  }
375  }
376  }
377 
378  if (border)
379  stack.Add(new CompactSpanReference(hx, hy, i));
380 
381  break;
382  }
383  }
384  }
385  }
386 
387  if (empty)
388  GetHeightDataSeedsFromVertices(compactField, poly, polyCount, verts, borderSize, hp, stack);
389 
390  const int RetractSize = 256;
391  int head = 0;
392 
393  while (head < stack.Count)
394  {
395  var cell = stack[head++];
396  var cs = compactField[cell];
397 
398  if (head >= RetractSize)
399  {
400  head = 0;
401  if (stack.Count > RetractSize)
402  {
403  for (int i = 0; i < stack.Count - RetractSize; i++)
404  stack[i] = stack[i + RetractSize];
405  }
406 
407  int targetSize = stack.Count % RetractSize;
408  while (stack.Count > targetSize)
409  stack.RemoveAt(stack.Count - 1);
410  }
411 
412  //loop in all four directions
413  for (var dir = Direction.West; dir <= Direction.South; dir++)
414  {
415  //skip
416  if (!cs.IsConnected(dir))
417  continue;
418 
419  int ax = cell.X + dir.GetHorizontalOffset();
420  int ay = cell.Y + dir.GetVerticalOffset();
421  int hx = ax - hp.X - borderSize;
422  int hy = ay - hp.Y - borderSize;
423 
424  if (hx < 0 || hx >= hp.Width || hy < 0 || hy >= hp.Length)
425  continue;
426 
427  //only continue if height is unset
428  if (hp.IsSet(hy * hp.Width + hx))
429  continue;
430 
431  //get new span
432  int ai = compactField.Cells[ay * compactField.Width + ax].StartIndex + CompactSpan.GetConnection(ref cs, dir);
433  CompactSpan ds = compactField.Spans[ai];
434 
435  hp[hx, hy] = ds.Minimum;
436 
437  stack.Add(new CompactSpanReference(ax, ay, ai));
438  }
439  }
440  }

452  private void GetHeightDataSeedsFromVertices(CompactHeightfield compactField, PolyMesh.Polygon poly, int polyCount, PolyVertex[] verts, int borderSize, HeightPatch hp, List<CompactSpanReference> stack)
453  {
454  hp.SetAll(0);
455 
456  //use poly vertices as seed points
457  for (int j = 0; j < polyCount; j++)
458  {
459  var csr = new CompactSpanReference(0, 0, -1);
460  int dmin = int.MaxValue;
461 
462  var v = verts[poly.Vertices[j]];
463 
464  for (int k = 0; k < 9; k++)
465  {
466  //get vertices and offset x and z coordinates depending on current drection
467  int ax = v.X + VertexOffset[k * 2 + 0];
468  int ay = v.Y;
469  int az = v.Z + VertexOffset[k * 2 + 1];
470 
471  //skip if out of bounds
472  if (ax < hp.X || ax >= hp.X + hp.Width || az < hp.Y || az >= hp.Y + hp.Length)
473  continue;
474 
475  //get new cell
476  CompactCell c = compactField.Cells[(az + borderSize) * compactField.Width + (ax + borderSize)];
477 
478  //loop through all the spans
479  for (int i = c.StartIndex, end = c.StartIndex + c.Count; i < end; i++)
480  {
481  CompactSpan s = compactField.Spans[i];
482 
483  //find minimum y-distance
484  int d = Math.Abs(ay - s.Minimum);
485  if (d < dmin)
486  {
487  csr = new CompactSpanReference(ax, az, i);
488  dmin = d;
489  }
490  }
491  }
492 
493  //only add if something new found
494  if (csr.Index != -1)
495  {
496  stack.Add(csr);
497  }
498  }
499 
500  //find center of polygon using flood fill
501  int pcx = 0, pcz = 0;
502  for (int j = 0; j < polyCount; j++)
503  {
504  var v = verts[poly.Vertices[j]];
505  pcx += v.X;
506  pcz += v.Z;
507  }
508 
509  pcx /= polyCount;
510  pcz /= polyCount;
511 
512  //stack groups 3 elements as one part
513  foreach (var cell in stack)
514  {
515  int idx = (cell.Y - hp.Y) * hp.Width + (cell.X - hp.X);
516  hp[idx] = 1;
517  }
518 
519  //process the entire stack
520  while (stack.Count > 0)
521  {
522  var cell = stack[stack.Count - 1];
523  stack.RemoveAt(stack.Count - 1);
524 
525  //check if close to center of polygon
526  if (Math.Abs(cell.X - pcx) <= 1 && Math.Abs(cell.Y - pcz) <= 1)
527  {
528  //clear the stack and add a new group
529  stack.Clear();
530 
531  stack.Add(cell);
532  break;
533  }
534 
535  CompactSpan cs = compactField[cell];
536 
537  //check all four directions
538  for (var dir = Direction.West; dir <= Direction.South; dir++)
539  {
540  //skip if disconnected
541  if (!cs.IsConnected(dir))
542  continue;
543 
544  //get neighbor
545  int ax = cell.X + dir.GetHorizontalOffset();
546  int ay = cell.Y + dir.GetVerticalOffset();
547 
548  //skip if out of bounds
549  if (ax < hp.X || ax >= (hp.X + hp.Width) || ay < hp.Y || ay >= (hp.Y + hp.Length))
550  continue;
551 
552  if (hp[(ay - hp.Y) * hp.Width + (ax - hp.X)] != 0)
553  continue;
554 
555  //get the new index
556  int ai = compactField.Cells[(ay + borderSize) * compactField.Width + (ax + borderSize)].StartIndex + CompactSpan.GetConnection(ref cs, dir);
557 
558  //save data
559  int idx = (ay - hp.Y) * hp.Width + (ax - hp.X);
560  hp[idx] = 1;
561 
562  //push to stack
563  stack.Add(new CompactSpanReference(ax, ay, ai));
564  }
565  }
566 
567  //clear the heightpatch
568  hp.Clear();
569 
570  //mark start locations
571  for (int i = 0; i < stack.Count; i++)
572  {
573  var c = stack[i];
574 
575  //set new heightpatch data
576  int idx = (c.Y - hp.Y) * hp.Width + (c.X - hp.X);
577  CompactSpan cs = compactField.Spans[c.Index];
578  hp[idx] = cs.Minimum;
579 
580  stack[i] = new CompactSpanReference(c.X + borderSize, c.Y + borderSize, c.Index);
581  }
582  }

584  private static float PolyMinExtent(Vector3[] verts)
585  {
586  float minDist = float.MaxValue;
587  for (int i = 0; i < verts.Length; i++)
588  {
589  int ni = (i + 1) % verts.Length;
590  Vector3 p0 = verts[i];
591  Vector3 p1 = verts[ni];
592 
593  float maxEdgeDist = 0;
594  for (int j = 0; j < verts.Length; j++)
595  {
596  if (j == i || j == ni)
597  continue;
598 
599  float d = Distance.PointToSegment2DSquared(ref verts[j], ref p0, ref p1);
600  maxEdgeDist = Math.Max(maxEdgeDist, d);
601  }
602 
603  minDist = Math.Min(minDist, maxEdgeDist);
604  }
605 
606  return (float)Math.Sqrt(minDist);
607  }

622  private void BuildPolyDetail(Vector3[] polyMeshVerts, int numMeshVerts, float sampleDist, float sampleMaxError, CompactHeightfield compactField, HeightPatch hp, List<Vector3> verts, List<TriangleData> tris, List<EdgeInfo> edges, List<SamplingData> samples)
623  {
624  const int MAX_VERTS = 127;
625  const int MAX_TRIS = 255;
626  const int MAX_VERTS_PER_EDGE = 32;
627  Vector3[] edge = new Vector3[MAX_VERTS_PER_EDGE + 1];
628  List<int> hull = new List<int>(MAX_VERTS);
629 
630  //fill up vertex array
631  for (int i = 0; i < numMeshVerts; ++i)
632  verts.Add(polyMeshVerts[i]);
633 
634  float cs = compactField.CellSize;
635  float ics = 1.0f / cs;
636 
637  float minExtent = PolyMinExtent(polyMeshVerts);
638 
639  //tessellate outlines
640  if (sampleDist > 0)
641  {
642  for (int i = 0, j = verts.Count - 1; i < verts.Count; j = i++)
643  {
644  Vector3 vi = verts[i];
645  Vector3 vj = verts[j];
646  bool swapped = false;
647 
648  //make sure order is correct, otherwise swap data
649  if (Math.Abs(vj.X - vi.X) < 1E-06f)
650  {
651  if (vj.Z > vi.Z)
652  {
653  Vector3 temp = vj;
654  vj = vi;
655  vi = temp;
656  swapped = true;
657  }
658  }
659  else if (vj.X > vi.X)
660  {
661  Vector3 temp = vj;
662  vj = vi;
663  vi = temp;
664  swapped = true;
665  }
666 
667  //create samples along the edge
668  Vector3 dv;
669  Vector3.Subtract(ref vi, ref vj, out dv);
670  float d = (float)Math.Sqrt(dv.X * dv.X + dv.Z * dv.Z);
671  int nn = 1 + (int)Math.Floor(d / sampleDist);
672 
673  if (nn >= MAX_VERTS_PER_EDGE)
674  nn = MAX_VERTS_PER_EDGE - 1;
675 
676  if (verts.Count + nn >= MAX_VERTS)
677  nn = MAX_VERTS - 1 - verts.Count;
678 
679  for (int k = 0; k <= nn; k++)
680  {
681  float u = (float)k / (float)nn;
682  Vector3 pos;
683 
684  Vector3 tmp;
685  Vector3.Multiply(ref dv, u, out tmp);
686  Vector3.Add(ref vj, ref tmp, out pos);
687 
688  pos.Y = GetHeight(pos, ics, compactField.CellHeight, hp) * compactField.CellHeight;
689 
690  edge[k] = pos;
691  }
692 
693  //simplify samples
694  int[] idx = new int[MAX_VERTS_PER_EDGE];
695  idx[0] = 0;
696  idx[1] = nn;
697  int nidx = 2;
698 
699  for (int k = 0; k < nidx - 1;)
700  {
701  int a = idx[k];
702  int b = idx[k + 1];
703  Vector3 va = edge[a];
704  Vector3 vb = edge[b];
705 
706  //find maximum deviation along segment
707  float maxd = 0;
708  int maxi = -1;
709  for (int m = a + 1; m < b; m++)
710  {
711  float dev = Distance.PointToSegmentSquared(ref edge[m], ref va, ref vb);
712  if (dev > maxd)
713  {
714  maxd = dev;
715  maxi = m;
716  }
717  }
718 
719  if (maxi != -1 && maxd > (sampleMaxError * sampleMaxError))
720  {
721  //shift data to the right
722  for (int m = nidx; m > k; m--)
723  idx[m] = idx[m - 1];
724 
725  //set new value
726  idx[k + 1] = maxi;
727  nidx++;
728  }
729  else
730  {
731  k++;
732  }
733  }
734 
735  hull.Add(j);
736 
737  //add new vertices
738  if (swapped)
739  {
740  for (int k = nidx - 2; k > 0; k--)
741  {
742  hull.Add(verts.Count);
743  verts.Add(edge[idx[k]]);
744  }
745  }
746  else
747  {
748  for (int k = 1; k < nidx - 1; k++)
749  {
750  hull.Add(verts.Count);
751  verts.Add(edge[idx[k]]);
752  }
753  }
754  }
755  }
756 
757  //tesselate base mesh
758  edges.Clear();
759  tris.Clear();
760 
761  if (minExtent < sampleDist * 2)
762  {
763  TriangulateHull(verts, hull, tris);
764  return;
765  }
766 
767  TriangulateHull(verts, hull, tris);
768 
769  if (tris.Count == 0)
770  {
771  Console.WriteLine("Can't triangulate polygon, adding default data.");
772  return;
773  }
774 
775  if (sampleDist > 0)
776  {
777  //create sample locations
778  BBox3 bounds = new BBox3();
779  bounds.Min = polyMeshVerts[0];
780  bounds.Max = polyMeshVerts[0];
781 
782  for (int i = 1; i < numMeshVerts; i++)
783  {
784  Vector3Extensions.ComponentMin(ref bounds.Min, ref polyMeshVerts[i], out bounds.Min);
785  Vector3Extensions.ComponentMax(ref bounds.Max, ref polyMeshVerts[i], out bounds.Max);
786  }
787 
788  int x0 = (int)Math.Floor(bounds.Min.X / sampleDist);
789  int x1 = (int)Math.Ceiling(bounds.Max.X / sampleDist);
790  int z0 = (int)Math.Floor(bounds.Min.Z / sampleDist);
791  int z1 = (int)Math.Ceiling(bounds.Max.Z / sampleDist);
792 
793  samples.Clear();
794 
795  for (int z = z0; z < z1; z++)
796  {
797  for (int x = x0; x < x1; x++)
798  {
799  Vector3 pt = new Vector3(x * sampleDist, (bounds.Max.Y + bounds.Min.Y) * 0.5f, z * sampleDist);
800 
801  //make sure samples aren't too close to edge
802  if (Distance.PointToPolygonSquared(pt, polyMeshVerts, numMeshVerts) > -sampleDist * 0.5f)
803  continue;
804 
805  SamplingData sd = new SamplingData(x, GetHeight(pt, ics, compactField.CellHeight, hp), z, false);
806  samples.Add(sd);
807  }
808  }
809 
810  //added samples
811  for (int iter = 0; iter < samples.Count; iter++)
812  {
813  if (verts.Count >= MAX_VERTS)
814  break;
815 
816  //find sample with most error
817  Vector3 bestPt = Vector3.Zero;
818  float bestDistance = 0;
819  int bestIndex = -1;
820 
821  for (int i = 0; i < samples.Count; i++)
822  {
823  SamplingData sd = samples[i];
824  if (sd.IsSampled)
825  continue;
826 
827  //jitter sample location to remove effects of bad triangulation
828  Vector3 pt;
829  pt.X = sd.X * sampleDist + GetJitterX(i) * compactField.CellSize * 0.1f;
830  pt.Y = sd.Y * compactField.CellHeight;
