244  public static void GetBoundingBox(ref Vector3 a, ref Vector3 b, ref Vector3 c, out BBox3 bbox)
245  {
246  Vector3 min = a, max = a;
247 
248  if (b.X < min.X) min.X = b.X;
249  if (b.Y < min.Y) min.Y = b.Y;
250  if (b.Z < min.Z) min.Z = b.Z;
251  if (c.X < min.X) min.X = c.X;
252  if (c.Y < min.Y) min.Y = c.Y;
253  if (c.Z < min.Z) min.Z = c.Z;
254 
255  if (b.X > max.X) max.X = b.X;
256  if (b.Y > max.Y) max.Y = b.Y;
257  if (b.Z > max.Z) max.Z = b.Z;
258  if (c.X > max.X) max.X = c.X;
259  if (c.Y > max.Y) max.Y = c.Y;
260  if (c.Z > max.Z) max.Z = c.Z;
261
