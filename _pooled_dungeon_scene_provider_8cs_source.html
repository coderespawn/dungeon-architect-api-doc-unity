19  public override void OnDungeonBuildStart() {
20  base.OnDungeonBuildStart();
21  pooledObjects.Clear ();
22  var items = GameObject.FindObjectsOfType<DungeonSceneProviderData>();
23  foreach (var item in items) {
24  if (item == null || item.externallyManaged) continue;
25  if (item.dungeon != this.dungeon) continue;
26  if (item.NodeId == null) continue;
27 
28  if (!pooledObjects.ContainsKey(item.NodeId)) {
29  pooledObjects.Add(item.NodeId, new Queue<GameObject>());
30  }
31  pooledObjects[item.NodeId].Enqueue(item.gameObject);
32  }
33  }
34 
35  public override void OnDungeonBuildStop() {
36  // Destroy all unused objects from the pool
37  foreach (var objects in pooledObjects.Values) {
38  foreach (var obj in objects) {
39  if (Application.isPlaying) {
40  Destroy(obj);
41  } else {
42  DestroyImmediate(obj);
43  }
44  }
45  }
46 
47  pooledObjects.Clear ();
48  }

50  public override GameObject AddSprite(SpriteDungeonThemeItem spriteProp, Matrix4x4 transform, IDungeonSceneObjectInstantiator objectInstantiator) {
51  if (spriteProp == null) return null;
52  string NodeId = spriteProp.NodeId;
53 
54  if (spriteProp.sprite == null) {
55  return null;
56  }
57 
58  FlipSpriteTransform(ref transform, spriteProp.sprite);
59 
60  GameObject item = null;
61  // Try to reuse an object from the pool
62  if (pooledObjects.ContainsKey (NodeId) && pooledObjects [NodeId].Count > 0) {
63  item = pooledObjects [NodeId].Dequeue ();
64  SetTransform (item.transform, transform);
65  } else {
66  // Pool is exhausted for this object
67  item = BuildSpriteObject(spriteProp, transform, NodeId);
68  }
69 
70  SetStaticState(item, spriteProp.StaticState);
71 
72  return item;
73  }
