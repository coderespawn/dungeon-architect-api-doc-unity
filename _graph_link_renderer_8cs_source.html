55  private Vector2 GetPointOnNodeBounds(Vector2 position, GraphPin pin, float distanceBias)
56  {
57  var nodeRect = (pin.Node != null) ? pin.Node.Bounds : new Rect(pin.WorldPosition, Vector2.one);
58  var center = nodeRect.position + nodeRect.size * 0.5f;
59  var b = new Bounds(center, nodeRect.size);
60  var direction = (center - position).normalized;
61  var r = new Ray(position, direction);
62  float intersectDistance;
63  if (b.IntersectRay(r, out intersectDistance))
64  {
65  return position + direction * (intersectDistance + distanceBias);
66  }
67 
68  return pin.WorldPosition;
69  }
