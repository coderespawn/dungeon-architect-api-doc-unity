35  public override void OnPostDungeonBuild(Dungeon dungeon, DungeonModel model)
36  {

40  foreach (var instance in snapModel.modules)
41  {
42  instanceMap[instance.InstanceID] = instance;
43  }
44 
45  var moduleList = new List<SnapQueryModuleInfo>();
46  var dungeonItemList = FindObjectsOfType<DungeonSceneProviderData>();
47  foreach (var dungeonItem in dungeonItemList)
48  {
49  var instanceId = dungeonItem.NodeId;
50  if (!instanceMap.ContainsKey(instanceId)) continue;
51 
52  var moduleInfo = new SnapQueryModuleInfo();
53  moduleInfo.instanceInfo = instanceMap[instanceId];
54  moduleInfo.moduleGameObject = dungeonItem.gameObject;
55 
56  // Build the doors
57  var parent = dungeonItem.gameObject;
58  var numChildren = parent.transform.childCount;
59  var connections = new List<SnapQueryConnectionInfo>();
60  for (int i = 0; i < numChildren; i++)
61  {
62  var child = parent.transform.GetChild(i);
63  var connectionComponent = child.gameObject.GetComponent<SnapConnection>();
64  if (connectionComponent != null)
65  {
66  var connectionInfo = new SnapQueryConnectionInfo();
67  bool valid = true;
68  if (connectionComponent.doorObject.activeInHierarchy)
69  {
70  connectionInfo.connectionGameObject = connectionComponent.doorObject;
71  connectionInfo.isDoor = true;
72  }
73  else if (connectionComponent.wallObject.activeInHierarchy)
74  {
75  connectionInfo.connectionGameObject = connectionComponent.wallObject;
76  connectionInfo.isDoor = false;
77  }
78  else
79  {
80  valid = false;
81  }
82 
83  if (valid)
84  {
85  connections.Add(connectionInfo);
86  }
87  }
88  }
89  moduleInfo.connections = connections.ToArray();
90 
91  moduleList.Add(moduleInfo);
92  }
93  modules = moduleList.ToArray();
