98  int[] boxesForGauss(float sigma, int n) // standard deviation, number of boxes
99  {
100  var wIdeal = Mathf.Sqrt((12*sigma*sigma/n)+1); // Ideal averaging filter width
101  var wl = Mathf.FloorToInt(wIdeal); if(wl%2==0) wl--;
102  var wu = wl+2;
103 
104  var mIdeal = (12*sigma*sigma - n*wl*wl - 4*n*wl - 3*n)/(-4*wl - 4);
105  var m = Mathf.Round(mIdeal);
106  // var sigmaActual = Math.sqrt( (m*wl*wl + (n-m)*wu*wu - n)/12 );
107 
108  var sizes = new List<int>(); for(var i=0; i<n; i++) sizes.Add(i<m?wl:wu);
109  return sizes.ToArray();
