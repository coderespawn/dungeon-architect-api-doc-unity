138  [UnityEditor.Callbacks.OnOpenAsset(1)]
139  public static bool OnOpenAsset(int instanceID, int line)
140  {
141  Object activeObject = EditorUtility.InstanceIDToObject(instanceID);
142  if (activeObject is Graph)
143  {
144  var graph = Selection.activeObject as Graph;
145  ShowThemeEditor(graph);
146  return true; //catch open file
147  }
148  else if (activeObject is SnapFlowAsset)
149  {
150  var dungeonFlow = Selection.activeObject as SnapFlowAsset;
151  ShowDungeonFlowEditor(dungeonFlow);
152  return true;
153  }
154  else if (activeObject is GridFlowAsset)
155  {
156  var dungeonFlow = Selection.activeObject as GridFlowAsset;
157  ShowDungeonGridFlowEditor(dungeonFlow);
158  return true;
159  }
160  else if (activeObject is SnapGridFlowAsset)
161  {
162  var dungeonFlow = Selection.activeObject as SnapGridFlowAsset;
163  ShowSnapGridFlowEditor(dungeonFlow);
164  return true;
165  }
166  return false; // let unity open the file
167  }

283  public static string MakeFilenameUnique(string dir, string filename)
284  {
285  string fileNamePart = System.IO.Path.GetFileNameWithoutExtension(filename);
286  string fileExt = System.IO.Path.GetExtension(filename);
287  var indexedFileName = fileNamePart + fileExt;
288  string path = System.IO.Path.Combine(dir, indexedFileName);
289  for (int i = 1; ; ++i)
290  {
291  if (!System.IO.File.Exists(path))
292  return indexedFileName;
293 
294  indexedFileName = fileNamePart + " " + i + fileExt;
295  path = System.IO.Path.Combine(dir, indexedFileName);
296  }
297  }

310  public static void DestroySpatialConstraintAsset(SpatialConstraintAsset spatialConstraint)
311  {
312  if (spatialConstraint == null)
313  {
314  return;
315  }
316  if (spatialConstraint.hostThemeNode == null)
317  {
318  return;
319  }
320 
321  var asset = spatialConstraint.hostThemeNode.Graph;
322  Undo.RegisterCompleteObjectUndo(asset, "Delete Node Spatial Constraint");
323 
324  var objectsToDestroy = new List<ScriptableObject>();
325  if (spatialConstraint != null && spatialConstraint.Graph != null)
326  {
327  foreach (var node in spatialConstraint.Graph.Nodes)
328  {
329  if (node is SCRuleNode)
330  {
331  var ruleNode = node as SCRuleNode;
332  foreach (var constraint in ruleNode.constraints)
333  {
334  objectsToDestroy.Add(constraint);
335  }
336  }
337  objectsToDestroy.Add(node);
338  }
339  objectsToDestroy.Add(spatialConstraint.Graph);
340  objectsToDestroy.Add(spatialConstraint);
341  }
342 
343  foreach (var objectToDestroy in objectsToDestroy)
344  {
345  if (objectToDestroy != null)
346  {
347  Undo.DestroyObjectImmediate(objectToDestroy);
348  }
349  }
350  }

410  static void CreateDefaultMarkerNodes(Graph graph)
411  {
412  if (graph == null)
413  {
414  Debug.LogWarning("Cannot create default marker nodes. graph is null");
415  return;
416  }
417  var markerNames = new string[] {
418  DungeonConstants.ST_GROUND,
419  DungeonConstants.ST_WALL,
420  DungeonConstants.ST_WALLSEPARATOR,
421  DungeonConstants.ST_FENCE,
422  DungeonConstants.ST_FENCESEPARATOR,
423  DungeonConstants.ST_DOOR,
424  DungeonConstants.ST_STAIR,
425  DungeonConstants.ST_STAIR2X,
426  DungeonConstants.ST_WALLHALF,
427  DungeonConstants.ST_WALLHALFSEPARATOR
428  };
429 
430  // Make sure we don't have any nodes in the graph
431  if (graph.Nodes.Count > 0)
432  {
433  return;
434  }
435 
436  const int INTER_NODE_X = 200;
437  const int INTER_NODE_Y = 300;
438  int itemsPerRow = markerNames.Length / 2;
439  for (int i = 0; i < markerNames.Length; i++)
440  {
441  int ix = i % itemsPerRow;
442  int iy = i / itemsPerRow;
443  int x = ix * INTER_NODE_X;
444  int y = iy * INTER_NODE_Y;
445  var node = GraphOperations.CreateNode<MarkerNode>(graph, null);
446  GraphEditorUtils.AddToAsset(new UnityEditorUIPlatform(), graph, node);
447  node.Position = new Vector2(x, y);
448  node.Caption = markerNames[i];
449  }
450  }

456  public static void CreateEditorTag(string tag)
457  {
458  SerializedObject tagManager = new SerializedObject(AssetDatabase.LoadAllAssetsAtPath("ProjectSettings/TagManager.asset")[0]);
459  SerializedProperty tagsProp = tagManager.FindProperty("tags");
460 
461  // Check if the tag is already present
462  for (int i = 0; i < tagsProp.arraySize; i++)
463  {
464  SerializedProperty t = tagsProp.GetArrayElementAtIndex(i);
465  if (t.stringValue.Equals(tag))
466  {
467  // Tag already exists. do not add a duplicate
468  return;
469  }
470  }
471 
472  tagsProp.InsertArrayElementAtIndex(0);
473  SerializedProperty n = tagsProp.GetArrayElementAtIndex(0);
474  n.stringValue = tag;
475 
476  tagManager.ApplyModifiedProperties();
477  }
