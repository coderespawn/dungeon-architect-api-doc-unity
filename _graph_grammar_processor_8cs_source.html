201  void ApplyMatch(RGraph dataGraph, RGraph lhs, RGraph rhs, Dictionary<RNode, RNode> LHSToDataNode)
202  {
203  // Break the links
204  {
205  var matchedDataNodes = LHSToDataNode.Values.ToArray();
206  foreach (var dataNode in matchedDataNodes)
207  {
208  var outgoingNodes = new List<RNode>(dataNode.Outgoing);
209  foreach (var outgoingNode in outgoingNodes)
210  {
211  if (matchedDataNodes.Contains(outgoingNode))
212  {
213  // Break this link
214  dataNode.BreakLinkTo(outgoingNode);
215  }
216  }
217  }
218  }
219 
220  var LHSNodesByIndex = CreateNodeByIndexMap(lhs);
221  var RHSNodesByIndex = CreateNodeByIndexMap(rhs);
222 
223  var RHStoDataNode = new Dictionary<RNode, RNode>();
224 
225  // Delete unused nodes
226  {
227  var dataNodeDeletionList = new List<RNode>(LHSToDataNode.Values);
228  foreach (var entry in RHSNodesByIndex)
229  {
230  var indexToRetain = entry.Key;
231  var rhsNode = entry.Value;
232  if (LHSNodesByIndex.ContainsKey(indexToRetain))
233  {
234  var lhsNodeToRetain = LHSNodesByIndex[indexToRetain];
235  if (LHSToDataNode.ContainsKey(lhsNodeToRetain))
236  {
237  var dataNodeToRetain = LHSToDataNode[lhsNodeToRetain];
238  dataNodeDeletionList.Remove(dataNodeToRetain);
239 
240  // Update the caption if this is a wildcard node
241  if (!rhsNode.Payload.nodeType.wildcard)
242  {
243  dataNodeToRetain.Payload.nodeType = rhsNode.Payload.nodeType;
244  }
245 
246  RHStoDataNode.Add(rhsNode, dataNodeToRetain);
247  }
248  }
249  }
250 
251  // Now delete the unused nodes
252  foreach (var dataNodeToDelete in dataNodeDeletionList)
253  {
254  dataGraph.RemoveNode(dataNodeToDelete);
255  }
256  }
257 
258  // Add new nodes
259  {
260  var newRhsIndices = new List<int>(RHSNodesByIndex.Keys);
261  foreach (var lhsEntry in LHSNodesByIndex)
262  {
263  newRhsIndices.Remove(lhsEntry.Key);
264  }
265 
266  foreach (var rhsIndexToCreate in newRhsIndices)
267  {
268  var rhsNode = RHSNodesByIndex[rhsIndexToCreate];
269  var dataNode = new RNode(dataGraph);
270 
271  var payload = new GrammarRuntimeGraphNodeData();
272  payload.nodeType = rhsNode.Payload.nodeType;
273 
274  dataNode.Payload = payload;
275  dataGraph.Nodes.Add(dataNode);
276 
277  RHStoDataNode.Add(rhsNode, dataNode);
278  }
279  }
280 
281  // Make the links
282  {
283  foreach (var entry in RHSNodesByIndex)
284  {
285  var rhsNode = entry.Value;
286  foreach (var outgoingRHS in rhsNode.Outgoing)
287  {
288  var dataNodeStart = RHStoDataNode[rhsNode];
289  var dataNodeEnd = RHStoDataNode[outgoingRHS];
290  dataNodeStart.MakeLinkTo(dataNodeEnd);

356  HashSet<RNode> unmatchedNodes, GraphPatternMatch match, HashSet<RNode> visited)
357  {
358  if (patternChildNodes.Count == 0)

393  private static bool MatchRecursive(RNode dataNode, RNode patternNode, HashSet<RNode> unmatchedNodes, GraphPatternMatch match, HashSet<RNode> visited)
394  {
395  if (dataNode == null || patternNode == null || !IsNodeDataEqual(dataNode, patternNode) || !unmatchedNodes.Contains(dataNode))
396  {
397  return false;
398  }
399 
400  visited.Add(dataNode);
401  visited.Add(patternNode);
402 
403  match.PatternToDataNode.Add(patternNode, dataNode);
404 
405  bool success = TraverseChildren(dataNode.Incoming, patternNode.Incoming, unmatchedNodes, match, visited)
406  && TraverseChildren(dataNode.Outgoing, patternNode.Outgoing, unmatchedNodes, match, visited);
407 
408  if (!success)
409  {
410  visited.Remove(dataNode);
411  visited.Remove(patternNode);
412  match.PatternToDataNode.Remove(patternNode);
413  }
414 
415  return success;
416  }
