126  public override T CreateLink<T>(Graph graph, GraphPin output, GraphPin input, UISystem uiSystem)
127  {
128  if (input != null && input.Node != null)
129  {
130  input = input.Node.InputPin;
131  }
132 
133  // If we have a link in the opposite direction, then break that link
134  var sourceNode = output.Node;
135  var destNode = input.Node;
136  if (sourceNode != null && destNode != null)
137  {
138  var links = destNode.OutputPin.GetConntectedLinks();
139  foreach (var link in links)
140  {
141  if (link.Output.Node == input.Node && link.Input.Node == output.Node)
142  {
143  GraphOperations.DestroyLink(link, uiSystem.Undo);
144  }
145  }
146  }
147 
148  if (input.Node != null)
149  {
150  input = input.Node.InputPin;
151  if (input != null)
152  {
153  return base.CreateLink<T>(graph, output, input, uiSystem);
154  }
155  }
156  return null;
157  }

226  int FindNextAvailableIndex()
227  {
228  if (graph.Nodes.Count == 0)
229  {
230  return 0;
231  }
232 
233  // Find an appropriate execution index for this node
234  var usedIndices = new HashSet<int>();
235  foreach (var graphNode in graph.Nodes)

242  }
243

245  {
246  if (!usedIndices.Contains(i))

248  return i;
