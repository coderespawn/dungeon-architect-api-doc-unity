147  public override T CreateLink<T>(Graph graph, GraphPin output, GraphPin input, UISystem uiSystem)
148  {
149  if (input != null && input.Node != null)
150  {
151  input = input.Node.InputPin;
152  }
153 
154  // If we have a link in the opposite direction, then break that link
155  var sourceNode = output.Node;
156  var destNode = input.Node;
157  if (sourceNode != null && destNode != null)
158  {
159  var links = destNode.OutputPin.GetConntectedLinks();
160  foreach (var link in links)
161  {
162  if (link.Output.Node == input.Node && link.Input.Node == output.Node)
163  {
164  GraphOperations.DestroyLink(link, uiSystem.Undo);
165  }
166  }
167  }
168 
169  if (input.Node != null)
170  {
171  input = input.Node.InputPin;
172  if (input != null)
173  {
174  return base.CreateLink<T>(graph, output, input, uiSystem);
175  }
176  }
177  return null;

216  public void RefreshGraph(Graph graph, GrammarRuntimeGraph runtimeGraph, UISystem uiSystem)
217  {
218  SetGraph(graph);
219 
220  PerformLayout(runtimeGraph);
221 
222  ClearActiveGraph(uiSystem);
223 
224  var map = new Dictionary<RuntimeGraphNode<GrammarRuntimeGraphNodeData>, GraphNode>();
225  // Add nodes
226  foreach (var runtimeNode in runtimeGraph.Nodes)
227  {
228  var screenCoord = camera.WorldToScreen(runtimeNode.Position);
229  var graphNode = CreateNode<GrammarTaskNode>(screenCoord, uiSystem);
230  graphNode.NodeType = runtimeNode.Payload.nodeType;
231  graphNode.DisplayExecutionIndex = false;
232 
233  map.Add(runtimeNode, graphNode);
234  }
235 
236  // Add links
237  {
238  foreach (var runtimeNode in runtimeGraph.Nodes)
239  {
240  foreach (var outgoingRuntimeNode in runtimeNode.Outgoing)
241  {
242  GraphNode srcGraphNode = map.ContainsKey(runtimeNode) ? map[runtimeNode] : null;
243  GraphNode dstGraphNode = map.ContainsKey(outgoingRuntimeNode) ? map[outgoingRuntimeNode] : null;
244  if (srcGraphNode == null || dstGraphNode == null || srcGraphNode.OutputPin == null || dstGraphNode.InputPin == null)
245  {
246  Debug.LogWarning("Cannot create link in result graph due to invalid node state");
247  continue;
248  }
249  CreateLinkBetweenPins(srcGraphNode.OutputPin, dstGraphNode.InputPin, uiSystem);
250  }
251  }
252  }
253 
254  FocusCameraOnBestFit();
255  }
