36  {

38  {
39  groupWeights[chunkSize] = 0;
40  }
41 
42  groupWeights[chunkSize] += module.SelectionWeight;
43 
44  if (!groupCounts.ContainsKey(chunkSize))
45  {
46  groupCounts[chunkSize] = 0;
47  }
48 
49  groupCounts[chunkSize] = groupCounts[chunkSize] + 1;
50  }
51  }
52 
53  // Average out the weights
54  var keys = groupWeights.Keys.ToArray();
55  foreach (var key in keys)
56  {
57  var value = groupWeights[key];
58  var count = groupCounts[key];
59  groupWeights[key] = value / count;
60  }
61  }

64  foreach (var entry in groupWeights) {
65

101  public override FlowLayoutPathNodeGroup[] Generate(FlowLayoutGraphQuery graphQuery, FlowLayoutGraphNode currentNode, System.Random random, HashSet<DungeonUID> visited)
102  {
103  var Node = currentNode;
104  if (Node == null) {
105  return new FlowLayoutPathNodeGroup[0];
106  }
107 
108  if (groupSettings.Length == 0) {
109  var nullGenerator = new NullFlowLayoutNodeGroupGenerator();
110  return nullGenerator.Generate(graphQuery, Node, random, visited);
111  }
112 
113  var weightedGroupSettings = new List<NodeGroupSettings>(groupSettings);
114  for (var i = 0; i < weightedGroupSettings.Count; i++)
115  {
116  var setting = weightedGroupSettings[i];
117  setting.Weight += random.NextFloat() * 0.0001f;
118  weightedGroupSettings[i] = setting;
119  }
120 
121  var result = new List<FlowLayoutPathNodeGroup>();
122  foreach (var groupSetting in weightedGroupSettings)
123  {
124  Vector3Int[] LocalSurfaceCoords;
125  Vector3Int[] LocalVolumeCoords;
126  coordBuilder.GetCoords(groupSetting.GroupSize, out LocalVolumeCoords, out LocalSurfaceCoords);
127 
128  foreach (var localSurfaceCoord in LocalSurfaceCoords) {
129  bool valid = true;
130  var baseCoord = MathUtils.RoundToVector3Int(Node.coord) - localSurfaceCoord;
131  foreach (var localVolumeCoord in LocalVolumeCoords) {
132  var groupNodeCoord = baseCoord + localVolumeCoord;
133 
134  var testNodeId = graphQuery.GetNodeAtCoord(MathUtils.ToVector3(groupNodeCoord));
135  var testNode = graphQuery.GetNode(testNodeId);
136  if (testNode == null || visited.Contains(testNode.nodeId) || testNode.active) {
137  valid = false;
138  break;
139  }
140  }
141 
142  if (valid) {
143  // Add this group
144  var newGroup = new FlowLayoutPathNodeGroup();
145  newGroup.IsGroup = true;
146  newGroup.Weight = groupSetting.Weight;
147  foreach (var localVolumeCoord in LocalVolumeCoords) {
148  var nodeCoord = baseCoord + localVolumeCoord;
149  var groupNodeId = graphQuery.GetNodeAtCoord(nodeCoord);
150  var groupNode = graphQuery.GetNode(groupNodeId);
151  if (groupNode != null)
152  {
153  newGroup.GroupNodes.Add(groupNode.nodeId);
154  }
155  }
156  foreach (var surfCoord in LocalSurfaceCoords) {
157  var nodeCoord = baseCoord + surfCoord;
158  var groupNodeId = graphQuery.GetNodeAtCoord(nodeCoord);
159  var groupNode = graphQuery.GetNode(groupNodeId);
160  newGroup.GroupEdgeNodes.Add(groupNode.nodeId);
161  }
162  result.Add(newGroup);
163  }
164  }
165  }
166 
167  MathUtils.Shuffle(result, random);
168  return result.ToArray();
169  }

178  if (VolumeCoordsMap.ContainsKey(groupSize) && SurfaceCoordsMap.ContainsKey(groupSize))
179  {
180  outVolumeCoords = VolumeCoordsMap[groupSize].ToArray();
181  outSurfaceCoords = SurfaceCoordsMap[groupSize].ToArray();
182  return;

184 
185  var volumeCoords = new List<Vector3Int>();
186  var surfaceCoords = new List<Vector3Int>();
187 
188  for (int dz = 0; dz < groupSize.z; dz++) {
189  for (int dy = 0; dy < groupSize.y; dy++) {
190  for (int dx = 0; dx < groupSize.x; dx++) {
191  volumeCoords.Add(new Vector3Int(dx, dy, dz));
192  if (dx == 0 || dx == groupSize.x - 1 ||
193  dy == 0 || dy == groupSize.y - 1 ||
194  dz == 0 || dz == groupSize.z - 1) {
195  surfaceCoords.Add(new Vector3Int(dx, dy, dz));
196  }
197  }
198  }
199  }
200 
201  VolumeCoordsMap.Add(groupSize, volumeCoords);
202  SurfaceCoordsMap.Add(groupSize, surfaceCoords);
203 
204  outVolumeCoords = volumeCoords.ToArray();
205  outSurfaceCoords = surfaceCoords.ToArray();
206  }

296  public static void BuildNodeGroup(FlowLayoutGraphQuery graphQuery, FlowLayoutGraphNode node, FlowLayoutGraphNode[] incomingNodes,
297  out FlowLayoutPathNodeGroup outGroup, out FFAGConstraintsLink[] outConstraintLinks)
298  {
299  var graph = graphQuery.Graph;
300  outGroup = new FlowLayoutPathNodeGroup();
301  if (graph == null)
302  {
303  outGroup = null;
304  outConstraintLinks = new FFAGConstraintsLink[0];
305  return;
306  }
307 
308 
309  Vector3Int minCoord;
310  Vector3Int maxCoord;
311  if (node.MergedCompositeNodes.Count <= 1) {
312  outGroup.IsGroup = false;
313  outGroup.GroupNodes.Add(node.nodeId);
314  outGroup.GroupEdgeNodes.Add(node.nodeId);
315  minCoord = maxCoord = MathUtils.RoundToVector3Int(node.coord);
316  }
317  else {
318  outGroup.IsGroup = true;
319  var minCoordF = node.MergedCompositeNodes[0].coord;
320  var maxCoordF = minCoordF;
321 
322  foreach (var subNode in node.MergedCompositeNodes) {
323  minCoordF = MathUtils.ComponentMin(minCoordF, subNode.coord);
324  maxCoordF = MathUtils.ComponentMax(maxCoordF, subNode.coord);
325  outGroup.GroupNodes.Add(subNode.nodeId);
326  }
327  minCoord = MathUtils.RoundToVector3Int(minCoordF);
328  maxCoord = MathUtils.RoundToVector3Int(maxCoordF);
329 
330  foreach (var subNode in node.MergedCompositeNodes) {
331  var coord = MathUtils.RoundToVector3Int(subNode.coord);
332  if (coord.x == minCoord.x || coord.y == minCoord.y || coord.z == minCoord.z ||
333  coord.x == maxCoord.x || coord.y == maxCoord.y || coord.z == maxCoord.z) {
334  outGroup.GroupEdgeNodes.Add(subNode.nodeId);
335  }
336  }
337  }
338 
339  var constraintLinkList = new List<FFAGConstraintsLink>();
340 
341  foreach (var link in graph.Links) {
342  if (link.state.type == FlowLayoutGraphLinkType.Unconnected) continue;
343 
344  var source = link.sourceSubNode.IsValid() ? link.sourceSubNode : link.source;
345  var destination = link.destinationSubNode.IsValid() ? link.destinationSubNode : link.destination;
346 
347  var bHostsSource = outGroup.GroupNodes.Contains(source);
348  var bHostsDest = outGroup.GroupNodes.Contains(destination);
349  if (!bHostsSource && !bHostsDest) continue;
350  if (bHostsSource && bHostsDest) continue;
351 
352  if (bHostsSource) {
353  if (outGroup.GroupEdgeNodes.Contains(source)) {
354  var sourceNode = graphQuery.GetNode(source);
355  if (sourceNode == null) sourceNode = graphQuery.GetSubNode(source);
356  var destinationNode = graphQuery.GetNode(destination);
357  if (destinationNode == null) destinationNode = graphQuery.GetSubNode(destination);
358  if (sourceNode != null && destinationNode != null) {
359  constraintLinkList.Add(new FFAGConstraintsLink(sourceNode, destinationNode));
360  }
361  }
362  }
363  else if (bHostsDest) {
364  if (outGroup.GroupEdgeNodes.Contains(destination)) {
365  var sourceNode = graphQuery.GetNode(source);
366  if (sourceNode == null) sourceNode = graphQuery.GetSubNode(source);
367  var destinationNode = graphQuery.GetNode(destination);
368  if (destinationNode == null) destinationNode = graphQuery.GetSubNode(destination);
369  if (sourceNode != null && destinationNode != null) {
370  constraintLinkList.Add(new FFAGConstraintsLink(destinationNode, sourceNode));
371  }
372  }
373  }
374  }
375 
376  var nodeByCoords = new Dictionary<Vector3Int, FlowLayoutGraphNode>();
377  foreach (var graphNode in graph.Nodes) {
378  if (graphNode.MergedCompositeNodes.Count > 0) {
379  foreach (var subNode in graphNode.MergedCompositeNodes) {
380  var coord = MathUtils.RoundToVector3Int(subNode.coord);
381  nodeByCoords[coord] = subNode;
382  }
383  }
384  else {
385  var coord = MathUtils.RoundToVector3Int(graphNode.coord);
386  nodeByCoords[coord] = graphNode;
387  }
388  }
389 
390  foreach (var incomingNode in incomingNodes) {
391  if (incomingNode == null) continue;
392  var innerCoord = MathUtils.RoundToVector3Int(incomingNode.coord);
393  innerCoord.x = Mathf.Clamp(innerCoord.x, minCoord.x, maxCoord.x);
394  innerCoord.y = Mathf.Clamp(innerCoord.y, minCoord.y, maxCoord.y);
395  innerCoord.z = Mathf.Clamp(innerCoord.z, minCoord.z, maxCoord.z);
396  if (nodeByCoords.ContainsKey(innerCoord))
397  {
398  var innerNode = nodeByCoords[innerCoord];
399  constraintLinkList.Add(new FFAGConstraintsLink(innerNode, incomingNode));
400  }
401  }
402 
403  outConstraintLinks = constraintLinkList.ToArray();
404  }
