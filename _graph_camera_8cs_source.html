80  public void HandleInput(Event e)
81  {
82  // Handle zooming
83  if (e.type == EventType.ScrollWheel) {
84  // Grab the original position under the mouse so we can restore it after the zoom
85  var originalGraphPosition = ScreenToWorld(e.mousePosition);
86 
87  float zoomMultiplier = 0.1f;
88  zoomMultiplier *= Mathf.Sign(e.delta.y);
89  zoomLevel = Mathf.Clamp(zoomLevel * (1 + zoomMultiplier), 1, maxAllowedZoom);
90 
91  var newGraphPosition = ScreenToWorld (e.mousePosition);
92  position += originalGraphPosition - newGraphPosition;
93  }

95  // Handle pan

162  public void FocusOnBestFit(Graph graph, Rect editorBounds)
163  {
164  if (graph == null) return;
165  if (graph.Nodes.Count > 0)
166  {
167  var graphVisibleSize = editorBounds.size * zoomLevel;
168  Vector2 average = Vector2.zero;
169  foreach (var node in graph.Nodes)
170  {
171  if (node == null) continue;
172  average += node.Bounds.center;
173  }
174  average /= graph.Nodes.Count;
175  position = average - graphVisibleSize / 2.0f;
176  }
