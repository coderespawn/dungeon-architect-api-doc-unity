326  public static Bounds TransformBounds(Matrix4x4 transform, Bounds bounds)
327  {
328  var vertices = new Vector3[8];
329  var center = bounds.center;
330  var extents = bounds.extents;
331 
332  vertices[0] = center + new Vector3(extents.x, extents.y, extents.z);
333  vertices[1] = center + new Vector3(extents.x, extents.y, -extents.z);
334  vertices[2] = center + new Vector3(extents.x, -extents.y, extents.z);
335  vertices[3] = center + new Vector3(extents.x, -extents.y, -extents.z);
336  vertices[4] = center + new Vector3(-extents.x, extents.y, extents.z);
337  vertices[5] = center + new Vector3(-extents.x, extents.y, -extents.z);
338  vertices[6] = center + new Vector3(-extents.x, -extents.y, extents.z);
339  vertices[7] = center + new Vector3(-extents.x, -extents.y, -extents.z);
340 
341  for (int i = 0; i < 8; i++)
342  {
343  vertices[i] = transform.MultiplyPoint3x4(vertices[i]);
344  }
345 
346  var newBounds = new Bounds(vertices[0], Vector3.zero);
347  for (int i = 1; i < 8; i++) {
348  newBounds.Encapsulate(vertices[i]);
349  }
350 
351  return newBounds;
352  }

354  public static Bounds TransformBoundsX(Matrix4x4 transform, Bounds localBounds)
355  {
356 
357  var center = transform * localBounds.center;
358 
359  // transform the local extents' axes
360  var extents = localBounds.extents;
361  var axisX = transform * new Vector3(extents.x, 0, 0);
362  var axisY = transform * new Vector3(0, extents.y, 0);
363  var axisZ = transform * new Vector3(0, 0, extents.z);
364 
365  // sum their absolute value to get the world extents
366  extents.x = Mathf.Abs(axisX.x) + Mathf.Abs(axisY.x) + Mathf.Abs(axisZ.x);
367  extents.y = Mathf.Abs(axisX.y) + Mathf.Abs(axisY.y) + Mathf.Abs(axisZ.y);
368  extents.z = Mathf.Abs(axisX.z) + Mathf.Abs(axisY.z) + Mathf.Abs(axisZ.z);
369 
370  return new Bounds { center = center, extents = extents };
371  }
