179  private void SpawnItems(SgfModuleNode[] modules, DungeonSceneProvider sceneProvider, IDungeonSceneObjectInstantiator objectInstantiator)
180  {
181  var levelMarkers = new LevelMarkerList();
182  foreach (var module in modules)
183  {
184  if (module == null || module.SpawnedModule == null) continue;
185 
186  var markers = new List<PlaceableMarker>(module.SpawnedModule.GetComponentsInChildren<PlaceableMarker>());
187  MathUtils.Shuffle(markers, random);
188 
189  foreach (var item in module.LayoutNode.items)
190  {
191  if (item == null) continue;
192 
193  PlaceableMarker bestMarker = null;
194  foreach (var markerInfo in markers)
195  {
196  if (markerInfo.supportedMarkers == null) continue;
197 
198  var supportedMarkers = new List<string>(markerInfo.supportedMarkers);
199  if (supportedMarkers.Contains(item.markerName))
200  {
201  bestMarker = markerInfo;
202  break;
203  }
204  }
205 
206  if (bestMarker != null)
207  {
208  markers.Remove(bestMarker);
209 
210  var flowItemMetadata = new FlowItemMetadata();
211  flowItemMetadata.itemId = item.itemId;
212  flowItemMetadata.itemType = item.type;
213  flowItemMetadata.referencedItems = item.referencedItemIds.ToArray();
214 
215  var themeMarkerEntry = new PropSocket();
216  themeMarkerEntry.SocketType = item.markerName;
217  themeMarkerEntry.Transform = bestMarker.transform.localToWorldMatrix;
218  themeMarkerEntry.metadata = flowItemMetadata;
219  levelMarkers.Add(themeMarkerEntry);
220  }
221  else
222  {
223  Debug.LogWarning(string.Format("Cannot spawn item: {0}. Make sure you have a placeable marker in the module prefab", item.markerName));
224  }
225  }
226  }
227 
228  // Run the theme engine
229  if (levelMarkers.Count > 0)
230  {
231  var dungeon = GetComponent<Dungeon>();
232  if (dungeon != null)
233  {
234  var context = new DungeonThemeExecutionContext();
235  context.builder = this;
236  context.config = config;
237  context.model = model;
238  context.spatialConstraintProcessor = null;
239  context.themeOverrideVolumes = new ThemeOverrideVolume[0];
240  context.sceneProvider = sceneProvider;
241  context.objectSpawner = new SyncDungeonSceneObjectSpawner();
242  context.objectInstantiator = objectInstantiator;
243  context.spawnListeners = new DungeonItemSpawnListener[]
244  {
245  GetComponent<FlowItemMetadataHandler>()
246  };
247 
248  var themeEngine = new DungeonThemeEngine(context);
249  themeEngine.ApplyTheme(levelMarkers, dungeon.GetThemeAssets());
250  }
251  else
252  {
253  Debug.LogError("Invalid dungeon reference");
254  }

280  private void FixupDoorStates(SgfModuleNode[] snapModules, FlowLayoutGraph layoutGraph)
281  {
282  var graphQuery = new FlowLayoutGraphQuery(layoutGraph);
283 
284  foreach (var moduleInfo in snapModules)
285  {
286  var moduleComponent = moduleInfo.SpawnedModule;
287  if (moduleComponent == null) continue;
288  var connectionComponents = moduleComponent.gameObject.GetComponentsInChildren<SnapConnection>();
289 
290  for (var doorIdx = 0; doorIdx < moduleInfo.Doors.Length; doorIdx++)
291  {
292  var doorInfo = moduleInfo.Doors[doorIdx];
293  doorInfo.SpawnedDoor = connectionComponents[doorIdx];
294 
295  bool foundDoor = false;
296  GameObject spawnedObject = null;
297 
298  bool containsLock = false;
299  FlowItem lockItem = null;
300 
301  if (doorInfo.CellInfo.connectionIdx != -1)
302  {
303  var link = graphQuery.GetLink(doorInfo.CellInfo.linkId);
304  if (link != null)
305  {
306  if (link.state.type != FlowLayoutGraphLinkType.Unconnected)
307  {
308  if (link.source == moduleInfo.ModuleInstanceId)
309  {
310  // Check if we have a lock here
311  if (link.state.items != null)
312  {
313  foreach (var item in link.state.items)
314  {
315  if (item.type == FlowGraphItemType.Lock)
316  {
317  containsLock = true;
318  lockItem = item;
319  // TODO: Setup Key-Lock metadata
320  }
321  }
322  }
323 
324  if (containsLock)
325  {
326  spawnedObject = doorInfo.SpawnedDoor.UpdateDoorState(SnapConnectionState.DoorLocked, lockItem.markerName);
327  }
328  else if (link.state.type == FlowLayoutGraphLinkType.OneWay)
329  {
330  spawnedObject = doorInfo.SpawnedDoor.UpdateDoorState(SnapConnectionState.DoorOneWay);
331  }
332  else
333  {
334  spawnedObject = doorInfo.SpawnedDoor.UpdateDoorState(SnapConnectionState.Door);
335  }
336  }
337  else
338  {
339  // Hide the other door so we don't have duplicates.
340  spawnedObject = doorInfo.SpawnedDoor.UpdateDoorState(SnapConnectionState.None);
341  }
342 
343  foundDoor = true;
344  }
345  }
346  }
347 
348  if (!foundDoor)
349  {
350  spawnedObject = doorInfo.SpawnedDoor.UpdateDoorState(SnapConnectionState.Wall);
351  }
352 
353  if (spawnedObject != null)
354  {
355  if (containsLock && lockItem != null)
356  {
357  var metaDataComponent = spawnedObject.GetComponent<FlowItemMetadataComponent>();
358  if (metaDataComponent == null)
359  {
360  metaDataComponent = spawnedObject.AddComponent<FlowItemMetadataComponent>();
361  }
362 
363  metaDataComponent.itemType = FlowGraphItemType.Lock;
364  metaDataComponent.itemId = lockItem.itemId.ToString();
365  var referencesIds = new List<string>();
366  foreach (var lockRefId in lockItem.referencedItemIds)
367  {
368  referencesIds.Add(lockRefId.ToString());
369  }
370  metaDataComponent.referencedItemIds = referencesIds.ToArray();
371  }
372  else
373  {
374  var metaDataComponent = spawnedObject.GetComponent<FlowItemMetadataComponent>();
375  if (metaDataComponent != null)
376  {
377  DungeonUtils.DestroyObject(metaDataComponent);
378  }
379  }
380  }
381  }
382  }
383  }
