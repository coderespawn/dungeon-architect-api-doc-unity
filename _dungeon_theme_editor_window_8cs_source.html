56  static ThemeEditorToolFunctionInfo[] FetchToolFunctions()
57  {
58  var functions = new List<ThemeEditorToolFunctionInfo>();
59  var assembly = System.Reflection.Assembly.GetAssembly(typeof(ThemeEditorToolFunctionDelegate));
60  var methods = assembly.GetTypes()
61  .SelectMany(t => t.GetMethods())
62  .Where(m => m.GetCustomAttributes(typeof(ThemeEditorToolAttribute), false).Length > 0)
63  .ToArray();
64  foreach (var method in methods)
65  {
66  if (method.IsStatic)
67  {
68  var function = Delegate.CreateDelegate(typeof(ThemeEditorToolFunctionDelegate), method) as ThemeEditorToolFunctionDelegate;
69  if (function != null)
70  {
71  var functionInfo = new ThemeEditorToolFunctionInfo();
72  functionInfo.function = function;
73  functionInfo.attribute = method.GetCustomAttributes(typeof(ThemeEditorToolAttribute), false)[0] as ThemeEditorToolAttribute;
74  functions.Add(functionInfo);
75  }
76  }
77  }
78 
79  // Sort based no priority
80  return functions.OrderBy(o => o.attribute.Priority).ToArray();
81  }

231  void OnGUI()
232  {
233  if (uiSystem == null || renderer == null)

238  var guiState = new GUIState(renderer);
239 
240  var bounds = new Rect(Vector2.zero, position.size);
241  graphEditor.UpdateWidget(uiSystem, bounds);
242  graphEditor.Draw(uiSystem, renderer);
243 
244  Event e = Event.current;
245  switch (e.type)
246  {
247  case EventType.DragUpdated:
248  case EventType.DragPerform:
249  DragAndDrop.visualMode = DragAndDropVisualMode.Copy;
250  if (e.type == EventType.DragPerform)
251  {
252  DragAndDrop.AcceptDrag();
253 
254  var gameObjects = new List<GameObject>();
255  var sprites = new List<Sprite>();
256 
257  foreach (var draggedObject in DragAndDrop.objectReferences) {
258  if (draggedObject is GameObject) {
259  gameObjects.Add(draggedObject as GameObject);
260  }
261  else if (draggedObject is Sprite) {
262  sprites.Add(draggedObject as Sprite);
263  }
264  }
265 
266  // Build the sprite nodes
267  foreach (var sprite in sprites) {
268  var node = graphEditor.CreateNode<SpriteNode>(e.mousePosition, uiSystem);
269  node.sprite = sprite;
270  graphEditor.SelectNode(node, uiSystem);
271  }
272 
273  // Build the game object nodes
274  if (gameObjects.Count > 0) {
275  if (gameObjects.Count == 1) {
276  // Build a game object node
277  var node = graphEditor.CreateNode<GameObjectNode>(e.mousePosition, uiSystem);
278  node.Template = gameObjects[0];
279 
280  var originalTransform = node.Template.transform;
281  node.offset = Matrix4x4.TRS(Vector3.zero, originalTransform.rotation, originalTransform.localScale);
282 
283  graphEditor.SelectNode(node, uiSystem);
284  }
285  else {
286  // Build a game object array node
287  var node = graphEditor.CreateNode<GameObjectArrayNode>(e.mousePosition, uiSystem);
288  node.Templates = gameObjects.ToArray();
289  graphEditor.SelectNode(node, uiSystem);
290  }
291  }
292  }
293  break;
294  }
295 
296  guiState.Restore();
297  DrawToolbar();
298 
299  HandleInput(Event.current);
300  }
