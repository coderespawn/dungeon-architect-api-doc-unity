100  private void TilemapDomainEditor_OnTileDoubleClicked(FlowTilemap tilemap, int tileX, int tileY)
101  {
102  if (tilemap != null)
103  {
104  bool valid = false;
105  var bounds = new Bounds();
106  var cellId = tileY * tilemap.Width + tileX;
107  var items = GameObject.FindObjectsOfType<DungeonSceneProviderData>();
108  GameObject selectedObject = null;
109  foreach (var item in items)
110  {
111  if (item.userData == cellId)
112  {
113  var renderer = item.gameObject.GetComponent<Renderer>();
114  if (renderer != null)
115  {
116  if (!valid)
117  {
118  valid = true;
119  bounds = renderer.bounds;
120  }
121  else
122  {
123  bounds.Encapsulate(renderer.bounds);
124  }
125  }
126 
127  selectedObject = item.gameObject;
128  }
129  }
130 
131  if (valid && SceneView.lastActiveSceneView != null)
132  {
133  bounds.center += new Vector3(0, bounds.extents.y, 0);
134  bounds.extents = new Vector3(bounds.extents.x, 1, bounds.extents.z);
135 
136  SceneView.lastActiveSceneView.Frame(bounds, false);
137  }
138 
139  if (selectedObject != null)
140  {
141  Selection.activeGameObject = selectedObject;
142  EditorApplication.RepaintHierarchyWindow();
143  }
144  }
