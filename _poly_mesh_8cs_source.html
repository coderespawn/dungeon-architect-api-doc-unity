60  public PolyMesh(ContourSet contSet, float cellSize, float cellHeight, int borderSize, int numVertsPerPoly)
61  {
62  //copy contour data
63  this.bounds = contSet.Bounds;
64  this.cellSize = cellSize;
65  this.cellHeight = cellHeight;
66  this.borderSize = borderSize;
67 
68  //get maximum limits
69  //TODO move to ContourSet?
70  int maxVertices = 0;
71  int maxTris = 0;
72  int maxVertsPerCont = 0;
73  foreach (var cont in contSet)
74  {
75  int vertCount = cont.Vertices.Length;
76 
77  //skip null contours
78  if (vertCount < 3)
79  continue;
80 
81  maxVertices += vertCount;
82  maxTris += vertCount - 2;
83  maxVertsPerCont = Math.Max(maxVertsPerCont, vertCount);
84  }
85 
86  //initialize the mesh members
87  var verts = new List<PolyVertex>(maxVertices);
88  var polys = new List<Polygon>(maxTris);
89 
90  Queue<int> vertRemoveQueue = new Queue<int>(maxVertices);
91 
92  this.numVertsPerPoly = numVertsPerPoly;
93 
94  var vertDict = new Dictionary<PolyVertex, int>(new PolyVertex.RoughYEqualityComparer(2));
95 
96  int[] indices = new int[maxVertsPerCont]; //keep track of vertex hash codes
97  Triangle[] tris = new Triangle[maxVertsPerCont];
98  List<Polygon> contPolys = new List<Polygon>(maxVertsPerCont + 1);
99 
100  //extract contour data
101  foreach (Contour cont in contSet)
102  {
103  //skip null contours
104  if (cont.IsNull)
105  continue;
106 
107  PolyVertex[] vertices = new PolyVertex[cont.Vertices.Length];
108 
109  //triangulate contours
110  for (int i = 0; i < cont.Vertices.Length; i++)
111  {
112  var cv = cont.Vertices[i];
113  vertices[i] = new PolyVertex(cv.X, cv.Y, cv.Z);
114  indices[i] = i;
115  }
116 
117  //Form triangles inside the area bounded by the contours
118  int ntris = Triangulate(vertices, indices, tris);
119  if (ntris <= 0) //TODO notify user when this happens. Logging?
120  ntris = -ntris;
121 
122  //add and merge vertices
123  for (int i = 0; i < cont.Vertices.Length; i++)
124  {
125  var cv = cont.Vertices[i];
126  var pv = vertices[i];
127 
128  //save the hash code for each vertex
129  indices[i] = AddVertex(vertDict, pv, verts);
130 
131  if (RegionId.HasFlags(cv.RegionId, RegionFlags.VertexBorder))
132  {
133  //the vertex should be removed
134  vertRemoveQueue.Enqueue(indices[i]);
135  }
136  }
137 
138  contPolys.Clear();
139 
140  //iterate through all the triangles
141  for (int i = 0; i < ntris; i++)
142  {
143  Triangle ti = tris[i];
144 
145  //make sure there are three distinct vertices. anything less can't be a polygon.
146  if (ti.Index0 == ti.Index1
147  || ti.Index0 == ti.Index2
148  || ti.Index1 == ti.Index2)
149  continue;
150 
151  //each polygon has numVertsPerPoly
152  //index 0, 1, 2 store triangle vertices
153  //other polygon indexes (3 to numVertsPerPoly - 1) should be used for storing extra vertices when two polygons merge together
154  Polygon p = new Polygon(numVertsPerPoly, Area.Null, RegionId.Null, 0);
155  p.Vertices[0] = RemoveDiagonalFlag(indices[ti.Index0]);
156  p.Vertices[1] = RemoveDiagonalFlag(indices[ti.Index1]);
157  p.Vertices[2] = RemoveDiagonalFlag(indices[ti.Index2]);
158  contPolys.Add(p);
159  }
160 
161  //no polygons generated, so skip
162  if (contPolys.Count == 0)
163  continue;
164 
165  //merge polygons
166  if (numVertsPerPoly > 3)
167  {
168  while (true)
169  {
170  //find best polygons
171  int bestMergeVal = 0;
172  int bestPolyA = 0, bestPolyB = 0, bestEdgeA = 0, bestEdgeB = 0;
173 
174  for (int i = 0; i < contPolys.Count - 1; i++)
175  {
176  int pj = i;
177 
178  for (int j = i + 1; j < contPolys.Count; j++)
179  {
180  int pk = j;
181  int ea = 0, eb = 0;
182  int v = GetPolyMergeValue(contPolys, pj, pk, verts, out ea, out eb);
183  if (v > bestMergeVal)
184  {
185  bestMergeVal = v;
186  bestPolyA = i;
187  bestPolyB = j;
188  bestEdgeA = ea;
189  bestEdgeB = eb;
190  }
191  }
192  }
193 
194  if (bestMergeVal > 0)
195  {
196  int pa = bestPolyA;
197  int pb = bestPolyB;
198  MergePolys(contPolys, pa, pb, bestEdgeA, bestEdgeB);
199  contPolys[pb] = contPolys[contPolys.Count - 1];
200  contPolys.RemoveAt(contPolys.Count - 1);
201  }
202  else
203  {
204  //no more merging
205  break;
206  }
207  }
208  }
209 
210  //store polygons
211  for (int i = 0; i < contPolys.Count; i++)
212  {
213  Polygon p = contPolys[i];
214  Polygon p2 = new Polygon(numVertsPerPoly, cont.Area, cont.RegionId, 0);
215 
216  Buffer.BlockCopy(p.Vertices, 0, p2.Vertices, 0, numVertsPerPoly * sizeof(int));
217 
218  polys.Add(p2);
219  }
220  }
221 
222  //remove edge vertices
223  while (vertRemoveQueue.Count > 0)
224  {
225  int i = vertRemoveQueue.Dequeue();
226 
227  if (CanRemoveVertex(polys, i))
228  RemoveVertex(verts, polys, i);
229  }
230 
231  //calculate adjacency (edges)
232  BuildMeshAdjacency(verts, polys, numVertsPerPoly);
233 
234  //find portal edges
235  if (this.borderSize > 0)
236  {
237  //iterate through all the polygons
238  for (int i = 0; i < polys.Count; i++)
239  {
240  Polygon p = polys[i];
241 
242  //iterate through all the vertices
243  for (int j = 0; j < numVertsPerPoly; j++)
244  {
245  if (p.Vertices[j] == NullId)
246  break;
247 
248  //skip connected edges
249  if (p.NeighborEdges[j] != NullId)
250  continue;
251 
252  int nj = j + 1;
253  if (nj >= numVertsPerPoly || p.Vertices[nj] == NullId)
254  nj = 0;
255 
256  //grab two consecutive vertices
257  int va = p.Vertices[j];
258  int vb = p.Vertices[nj];
259 
260  //set some flags
261  if (verts[va].X == 0 && verts[vb].X == 0)
262  p.NeighborEdges[j] = NeighborEdgeFlag | 0;
263  else if (verts[va].Z == contSet.Height && verts[vb].Z == contSet.Height)
264  p.NeighborEdges[j] = NeighborEdgeFlag | 1;
265  else if (verts[va].X == contSet.Width && verts[vb].X == contSet.Width)
266  p.NeighborEdges[j] = NeighborEdgeFlag | 2;
267  else if (verts[va].Z == 0 && verts[vb].Z == 0)
268  p.NeighborEdges[j] = NeighborEdgeFlag | 3;
269  }
270  }
271  }
272 
273  this.vertices = verts.ToArray();
274  this.polygons = polys.ToArray();
275  }

429  public static bool InCone(int i, int j, PolyVertex[] verts, int[] indices)
430  {
431  int pi = RemoveDiagonalFlag(indices[i]);
432  int pj = RemoveDiagonalFlag(indices[j]);
433  int pi1 = RemoveDiagonalFlag(indices[Next(i, verts.Length)]);
434  int pin1 = RemoveDiagonalFlag(indices[Prev(i, verts.Length)]);
435 
436  //if P[i] is convex vertex (i + 1 left or on (i - 1, i))
437  if (PolyVertex.IsLeftOn(ref verts[pin1], ref verts[pi], ref verts[pi1]))
438  return PolyVertex.IsLeft(ref verts[pi], ref verts[pj], ref verts[pin1]) && PolyVertex.IsLeft(ref verts[pj], ref verts[pi], ref verts[pi1]);
439 
440  //assume (i - 1, i, i + 1) not collinear
441  return !(PolyVertex.IsLeftOn(ref verts[pi], ref verts[pj], ref verts[pi1]) && PolyVertex.IsLeftOn(ref verts[pj], ref verts[pi], ref verts[pin1]));
442  }

453  public static bool Diagonalie(int i, int j, PolyVertex[] verts, int[] indices)
454  {
455  int d0 = RemoveDiagonalFlag(indices[i]);
456  int d1 = RemoveDiagonalFlag(indices[j]);
457 
458  //for each edge (k, k + 1)
459  for (int k = 0; k < verts.Length; k++)
460  {
461  int k1 = Next(k, verts.Length);
462 
463  //skip edges incident to i or j
464  if (!((k == i) || (k1 == i) || (k == j) || (k1 == j)))
465  {
466  int p0 = RemoveDiagonalFlag(indices[k]);
467  int p1 = RemoveDiagonalFlag(indices[k1]);
468 
469  if (PolyVertex.Equal2D(ref verts[d0], ref verts[p0]) ||
470  PolyVertex.Equal2D(ref verts[d1], ref verts[p0]) ||
471  PolyVertex.Equal2D(ref verts[d0], ref verts[p1]) ||
472  PolyVertex.Equal2D(ref verts[d1], ref verts[p1]))
473  continue;
474 
475  if (PolyVertex.Intersect(ref verts[d0], ref verts[d1], ref verts[p0], ref verts[p1]))
476  return false;
477  }
478  }
479 
480  return true;
481  }

554  private static int Triangulate(PolyVertex[] verts, int[] indices, Triangle[] tris)
555  {
556  int ntris = 0;
557  int n = verts.Length;
558 
559  //last bit of index determines whether vertex can be removed
560  for (int i = 0; i < n; i++)
561  {
562  int i1 = Next(i, n);
563  int i2 = Next(i1, n);
564  if (Diagonal(i, i2, verts, indices))
565  {
566  SetDiagonalFlag(ref indices[i1]);
567  }
568  }
569 
570  //need 3 verts minimum for a polygon
571  while (n > 3)
572  {
573  //find the minimum distance betwee two vertices.
574  //also, save their index
575  int minLen = -1;
576  int minIndex = -1;
577  for (int i = 0; i < n; i++)
578  {
579  int i1 = Next(i, n);
580 
581  if (HasDiagonalFlag(indices[i1]))
582  {
583  int p0 = RemoveDiagonalFlag(indices[i]);
584  int p2 = RemoveDiagonalFlag(indices[Next(i1, n)]);
585 
586  int dx = verts[p2].X - verts[p0].X;
587  int dy = verts[p2].Z - verts[p0].Z;
588  int len = dx * dx + dy * dy;
589 
590  if (minLen < 0 || len < minLen)
591  {
592  minLen = len;
593  minIndex = i;
594  }
595  }
596  }
597 
598  if (minIndex == -1)
599  {
600  minLen = -1;
601  minIndex = -1;
602 
603 
604  for (int i = 0; i < n; i++)
605  {
606  int i1 = Next(i, n);
607  int i2 = Next(i1, n);
608 
609  if (IsDiagonalLoose(i, i2, verts, indices))
610  {
611  var p0 = verts[RemoveDiagonalFlag(indices[i])];
612  var p2 = verts[RemoveDiagonalFlag(indices[Next(i2, n)])];
613  var dx = p2.X - p0.X;
614  var dy = p2.Z - p0.Z;
615  var len = dx * dx + dy * dy;
616  if (minLen < 0 || len < minLen)
617  {
618  minLen = len;
619  minIndex = i;
620  }
621  }
622  }
623 
624 
625  if (minIndex == -1)
626  {
627  return -ntris;
628  }
629  }
630 
631 
632  int mi = minIndex;
633  int mi1 = Next(mi, n);
634  int mi2 = Next(mi1, n);
635 
636  tris[ntris] = new Triangle();
637  tris[ntris].Index0 = RemoveDiagonalFlag(indices[mi]);
638  tris[ntris].Index1 = RemoveDiagonalFlag(indices[mi1]);
639  tris[ntris].Index2 = RemoveDiagonalFlag(indices[mi2]);
640  ntris++;
641 
642  //remove P[i1]
643  n--;
644  for (int k = mi1; k < n; k++)
645  indices[k] = indices[k + 1];
646 
647  if (mi1 >= n) mi1 = 0;
648  mi = Prev(mi1, n);
649 
650  //update diagonal flags
651  if (Diagonal(Prev(mi, n), mi1, verts, indices))
652  {
653  SetDiagonalFlag(ref indices[mi]);
654  }
655  else
656  {
657  RemoveDiagonalFlag(ref indices[mi]);
658  }
659 
660  if (Diagonal(mi, Next(mi1, n), verts, indices))
661  {
662  SetDiagonalFlag(ref indices[mi1]);
663  }
664  else
665  {
666  RemoveDiagonalFlag(ref indices[mi1]);
667  }
668  }
669 
670  //append remaining triangle
671  tris[ntris] = new Triangle();
672  tris[ntris].Index0 = RemoveDiagonalFlag(indices[0]);
673  tris[ntris].Index1 = RemoveDiagonalFlag(indices[1]);
674  tris[ntris].Index2 = RemoveDiagonalFlag(indices[2]);
675  ntris++;
676 
677  return ntris;
678  }

680  static bool IsInConeLoose(int i, int j, PolyVertex[] verts, int[] indices)
681  {
682  var n = verts.Length;
683  // int p2 = RemoveDiagonalFlag(indices[Next(i1, n)]);
684 
685  //const int* pi = &verts[(indices[i] & 0x0fffffff) * 4];
686  //const int* pj = &verts[(indices[j] & 0x0fffffff) * 4];
687  //const int* pi1 = &verts[(indices[next(i, n)] & 0x0fffffff) * 4];
688  //const int* pin1 = &verts[(indices[prev(i, n)] & 0x0fffffff) * 4];
689  var pi = verts[RemoveDiagonalFlag(indices[i])];
690  var pj = verts[RemoveDiagonalFlag(indices[j])];
691  var pi1 = verts[RemoveDiagonalFlag(indices[Next(i, n)])];
692  var pin1 = verts[RemoveDiagonalFlag(indices[Prev(i, n)])];
693 
694  // If P[i] is a convex vertex [ i+1 left or on (i-1,i) ].
695 
696  if (PolyVertex.IsLeftOn(ref pin1, ref pi, ref pi1))
697  return PolyVertex.IsLeftOn(ref pi, ref pj, ref pin1) && PolyVertex.IsLeftOn(ref pj, ref pi, ref pi1);
698  // Assume (i-1,i,i+1) not collinear.
699  // else P[i] is reflex.
700  return !(PolyVertex.IsLeftOn(ref pi, ref pj, ref pi1) && PolyVertex.IsLeftOn(ref pj, ref pi, ref pin1));
701  }

709  static bool IsDiagonalieLoose(int i, int j, PolyVertex[] verts, int[] indices)
710  {
711  //const int* d0 = &verts[(indices[i] & 0x0fffffff) * 4];
712  //const int* d1 = &verts[(indices[j] & 0x0fffffff) * 4];
713  var n = verts.Length;
714  var d0 = verts[RemoveDiagonalFlag(indices[i])];
715  var d1 = verts[RemoveDiagonalFlag(indices[j])];
716 
717  // For each edge (k,k+1) of P
718  for (int k = 0; k < n; k++)
719  {
720  int k1 = Next(k, n);
721  // Skip edges incident to i or j
722  if (!((k == i) || (k1 == i) || (k == j) || (k1 == j)))
723  {
724  //const int* p0 = &verts[(indices[k] & 0x0fffffff) * 4];
725  //const int* p1 = &verts[(indices[k1] & 0x0fffffff) * 4];
726  var p0 = verts[RemoveDiagonalFlag(indices[k])];
727  var p1 = verts[RemoveDiagonalFlag(indices[k1])];
728 
729 
730  if (PolyVertex.Equal2D(ref d0, ref p0) ||
731  PolyVertex.Equal2D(ref d1, ref p0) ||
732  PolyVertex.Equal2D(ref d0, ref p1) ||
733  PolyVertex.Equal2D(ref d1, ref p1))
734  continue;
735 
736  if (IsIntersectProp(ref d0, ref d1, ref p0, ref p1))
737  return false;
738  }
739  }
740  return true;
741  }

743  static bool xorb(bool x, bool y)
744  {

748  // Returns true iff ab properly intersects cd: they share
749  // a point interior to both segments. The properness of the
750  // intersection is ensured by using strict leftness.
751  static bool IsIntersectProp(ref PolyVertex a, ref PolyVertex b, ref PolyVertex c, ref PolyVertex d)
752  {
753  // Eliminate improper cases.
754  if (PolyVertex.IsCollinear(ref a, ref b, ref c) || PolyVertex.IsCollinear(ref a, ref b, ref d) ||
755  PolyVertex.IsCollinear(ref c, ref d, ref a) || PolyVertex.IsCollinear(ref c, ref d, ref b))
756  return false;
757 
758 
759  return xorb(PolyVertex.IsLeft(ref a, ref b, ref c), PolyVertex.IsLeft(ref a, ref b, ref d)) &&
760  xorb(PolyVertex.IsLeft(ref c, ref d, ref a), PolyVertex.IsLeft(ref c, ref d, ref b));
761  }

794  private static int GetPolyMergeValue(List<Polygon> polys, int polyA, int polyB, List<PolyVertex> verts, out int edgeA, out int edgeB)
795  {
796  int numVertsA = polys[polyA].VertexCount;
797  int numVertsB = polys[polyB].VertexCount;
798 
799  //check if polygons share an edge
800  edgeA = -1;
801  edgeB = -1;
802 
803  //don't merge if result is too big
804  if (numVertsA + numVertsB - 2 > polys[polyA].Vertices.Length)
805  return -1;
806 
807  //iterate through all the vertices of polygonA
808  for (int i = 0; i < numVertsA; i++)
809  {
810  //take two nearby vertices
811  int va0 = polys[polyA].Vertices[i];
812  int va1 = polys[polyA].Vertices[(i + 1) % numVertsA];
813 
814  //make sure va0 < va1
815  if (va0 > va1)
816  {
817  int temp = va0;
818  va0 = va1;
819  va1 = temp;
820  }
821 
822  //iterate through all the vertices of polygon B
823  for (int j = 0; j < numVertsB; j++)
824  {
825  //take two nearby vertices
826  int vb0 = polys[polyB].Vertices[j];
827  int vb1 = polys[polyB].Vertices[(j + 1) % numVertsB];
828 
829  //make sure vb0 < vb1
830  if (vb0 > vb1)
831  {
832  int temp = vb0;
833  vb0 = vb1;
834  vb1 = temp;
835  }
836 
837  //edge shared, since vertices are equal
838  if (va0 == vb0 && va1 == vb1)
839  {
840  edgeA = i;
841  edgeB = j;
842  break;
843  }
844  }
845  }
846 
847  //no common edge
848  if (edgeA == -1 || edgeB == -1)
849  return -1;
850 
851  //check if merged polygon would be convex
852  int vertA, vertB, vertC;
853 
854  vertA = polys[polyA].Vertices[(edgeA + numVertsA - 1) % numVertsA];
855  vertB = polys[polyA].Vertices[edgeA];
856  vertC = polys[polyB].Vertices[(edgeB + 2) % numVertsB];
