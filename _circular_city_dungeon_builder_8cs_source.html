104  void GenerateCityLayout()
105  {
106  var roadGraphSettings = new RoadGraphBuilderSettings();
107  roadGraphSettings.interNodeDistance = cityConfig.interNodeDistance;
108  var roadGraphBuilder = new RoadGraphBuilder(roadGraphSettings);
109 
110  float startRadius = cityConfig.startRadius;
111  float endRadius = cityConfig.endRadius;
112  var center = Vector3.zero;
113 
114  float mainRoadStrength = cityConfig.mainRoadStrength;
115  float sideRoadStrength = cityConfig.sideRoadStrength;
116  float interRingDistance = (endRadius - startRadius) / (cityConfig.numRings - 1);
117 
118  // Draw the city rings
119  {
120  float ringRadius = startRadius;
121  for (int r = 0; r < cityConfig.numRings; r++)
122  {
123  roadGraphBuilder.CreateCircle(center, ringRadius, mainRoadStrength);
124  if (r < cityConfig.numRings - 1)
125  {
126  float ringLaneRadius = ringRadius + interRingDistance / 2.0f;
127  roadGraphBuilder.CreateCircle(center, ringLaneRadius, sideRoadStrength);
128  }
129  ringRadius += interRingDistance;
130  }
131  }
132 
133  // Draw the ray lanes ejecting out from the center
134  float interRayAngle = Mathf.PI * 2 / cityConfig.numRays;
135  for (int i = 0; i < cityConfig.numRays; i++)
136  {
137  float ringRadius = startRadius;
138  int mainSegmentToRemove = random.Next() % (cityConfig.numRings - 1);
139  for (int r = 0; r < cityConfig.numRings - 1; r++)
140  {
141  float nextRingRadius = ringRadius + interRingDistance;
142 
143  bool bDrawMainSegment = (r != mainSegmentToRemove);
144 
145  if (bDrawMainSegment)
146  {
147  float angle = i * interRayAngle;
148  var direction = new Vector3(Mathf.Cos(angle), 0, Mathf.Sin(angle));
149  float midRingRadius = ringRadius + (nextRingRadius - ringRadius) / 2.0f;
150 
151  var start = center + direction * ringRadius;
152  var end = center + direction * midRingRadius;
153  roadGraphBuilder.CreateLine(start, end, mainRoadStrength);
154 
155  start = center + direction * midRingRadius;
156  end = center + direction * nextRingRadius;
157  roadGraphBuilder.CreateLine(start, end, mainRoadStrength);
158  }
159 
160  // Draw the side segments
161  {
162  float interSideRayAngle = interRayAngle / (r + 2);
163  float sideRayAngle = i * interRayAngle;
164  float interSideRingDistance = interRingDistance / 2.0f;
165  for (int si = 0; si <= r; si++)
166  {
167  sideRayAngle += interSideRayAngle;
168  for (int t = 0; t < 2; t++)
169  {
170  bool bRemoveSideSegment = (random.NextFloat() < cityConfig.sideRoadRemovalProbability);
171  if (!bRemoveSideSegment)
172  {
173  float perimeter = Mathf.PI * 2 * ringRadius;
174  float angleRandomization = cityConfig.randomSideLaneOffsetAngle * Mathf.Deg2Rad / perimeter;
175  float randomValue = random.NextFloat() * 2 - 1;
176  float randomizedAngle = sideRayAngle + angleRandomization * randomValue;
177  float laneStartRadius = ringRadius + interSideRingDistance * t;
178  float laneEndRadius = ringRadius + interSideRingDistance * (t + 1);
179 
180  var direction = new Vector3(Mathf.Cos(randomizedAngle), 0, Mathf.Sin(randomizedAngle));
181  var start = center + direction * laneStartRadius;
182  var end = center + direction * laneEndRadius;
183  roadGraphBuilder.CreateLine(start, end, sideRoadStrength);
184  }
185  }
186  }
187  }
188 
189  ringRadius = nextRingRadius;
190  }
191  }

334  void EmitCityMarkers()
335  {
336  var basePosition = transform.position;
337 
338  float occupancySearchDistance = cityConfig.buildingSize * 1.0f;
339  int cacheGridResolution = Mathf.RoundToInt(occupancySearchDistance * 2.0f);
340  var spatialOccupancy = new SpatialPartitionCache(cacheGridResolution);
341  foreach (var node in cityModel.roadGraph.nodes)
342  {
343  spatialOccupancy.RegisterAsOccupied(node.position);
344  }
345  foreach (var node in cityModel.layoutGraph.nodes)
346  {
347  spatialOccupancy.RegisterAsOccupied(node.position);
348  }
349 
350  for (float r = cityConfig.startRadius; r < cityConfig.endRadius; r += cityConfig.buildingSize)
351  {
352  float circumference = 2 * Mathf.PI * r;
353  int numBuildings = Mathf.RoundToInt(circumference / cityConfig.buildingSize);
354  for (int i = 0; i < numBuildings; i++)
355  {
356  float angle = 2 * Mathf.PI * i / (float)numBuildings;
357  var offset = Vector3.zero;
358  offset.x = Mathf.Cos(angle) * r;
359  offset.z = Mathf.Sin(angle) * r;
360  var position = basePosition + offset;
361  if (!spatialOccupancy.IsFree(offset, occupancySearchDistance))
362  {
363  continue;
364  }
365  var rotation = Quaternion.Euler(0, -angle * Mathf.Rad2Deg, 0);
366  EmitMarkerAt(CircularCityDungeonConstants.House, position, rotation);
