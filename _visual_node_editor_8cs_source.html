93  void DrawRule<T>(string caption, ref bool ruleEnabled, ref string ruleClassName) where T : ScriptableObject
94  {
95  GUI.enabled = true;
96  EditorGUILayout.BeginHorizontal();
97  GUILayout.Space(CATEGORY_SPACING);
98  ruleEnabled = EditorGUILayout.ToggleLeft(caption, ruleEnabled);
99  GUI.enabled = ruleEnabled;
100  MonoScript script = null;
101  if (ruleClassName != null)
102  {
103  var rule = instanceCache.GetInstance(ruleClassName) as ScriptableObject;
104  if (rule != null)
105  {
106  script = MonoScript.FromScriptableObject(rule);
107  }
108  }
109  var oldScript = script;
110  script = EditorGUILayout.ObjectField(script, typeof(MonoScript), false) as MonoScript;
111  if (oldScript != script && script != null)
112  {
113  var ruleType = script.GetClass();
114  if (ruleType.IsSubclassOf(typeof(T)))
115  {
116  ruleClassName = script.GetClass().AssemblyQualifiedName;
117  }
118  else
119  {
120  ruleClassName = null;
121  }
122  }
123  else if (script == null)
124  {
125  ruleClassName = null;
126  }
127 
128  EditorGUILayout.EndHorizontal();
129  GUI.enabled = true;
130  }
