187  void DrawNodeDragOverlay(UIRenderer renderer, Rect bounds)
188  {
189  var cursorWorld = camera.ScreenToWorld(lastMousePosition);
190 
191  var ringTexture = renderer.GetResource<Texture2D>(UIResourceLookup.TEXTURE_CURSOR_RING_SOLID) as Texture2D;
192 
193  int sx = Mathf.RoundToInt(cursorWorld.x / SCBaseDomainNode.TileSize);
194  int sy = Mathf.RoundToInt(cursorWorld.y / SCBaseDomainNode.TileSize);
195  int drawNodeRangeRadius = 4;
196  float drawPixelRangeRadius = drawNodeRangeRadius * SCBaseDomainNode.TileSize;
197  float maxCursorSize = 20;
198 
199 
200  for (int x = sx - drawNodeRangeRadius; x <= sx + drawNodeRangeRadius; x++)
201  {
202  for (int y = sy - drawNodeRangeRadius; y <= sy + drawNodeRangeRadius; y++)
203  {
204  for (int ox = 0; ox < 2; ox++)
205  {
206  for (int oy = 0; oy < 2; oy++)
207  {
208  var offset = new Vector2(ox, oy) * 0.5f;
209  var pos = (new Vector2(x, y) + offset) * SCBaseDomainNode.TileSize;
210 
211  var distanceToCursor = (pos - cursorWorld).magnitude;
212  var cursorScale = Mathf.Clamp(1 - distanceToCursor / drawPixelRangeRadius, 0.0f, 1.0f);
213 
214  // Ease the size
215  {
216  cursorScale = Mathf.Pow(cursorScale, 3);
217  }
218 
219  float cursorSize = maxCursorSize * cursorScale;
220  var size = new Vector2(cursorSize, cursorSize);
221  pos -= size / 2.0f;
222  pos = camera.WorldToScreen(pos);
223 
224  size /= camera.ZoomLevel;
225  var ringBounds = new Rect(pos, size);
226  var color = new Color(1, 0.3f, 0, cursorScale / 2.0f);
227  renderer.DrawTexture(ringBounds, ringTexture, ScaleMode.ScaleToFit, true, color);
228  }
229  }
230  }
231  }
