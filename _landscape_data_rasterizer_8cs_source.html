145  public static void WorldToTerrainCoord(Terrain terrain, float x, float y, out int gx, out int gy, RasterizerTextureSpace textureSpace)
146  {
147  var terrainSize = terrain.terrainData.size;
148  var data = terrain.terrainData;
149 
150  var terrainWidth = 0;
151  var terrainHeight = 0;
152 
153  if (textureSpace == RasterizerTextureSpace.HeightMap)
154  {
155  terrainWidth = data.heightmapResolution;
156  terrainHeight = data.heightmapResolution;
157  }
158  else if (textureSpace == RasterizerTextureSpace.AlphaMap)
159  {
160  terrainWidth = data.alphamapWidth;
161  terrainHeight = data.alphamapHeight;
162  }
163  else if (textureSpace == RasterizerTextureSpace.DetailMap)
164  {
165  terrainWidth = data.detailWidth;
166  terrainHeight = data.detailHeight;
167  }
168 
169  var multiplierX = (terrainWidth - 1) / terrainSize.x;
170  var multiplierZ = (terrainHeight - 1) / terrainSize.z;
171 
172  var offset = new Vector2();
173  offset.x = -terrain.transform.position.x;
174  offset.y = -terrain.transform.position.z;
175  var xf = (x + offset.x) * multiplierX;
176  var yf = (y + offset.y) * multiplierZ;
177 
178  gx = Mathf.RoundToInt(xf);
179  gy = Mathf.RoundToInt(yf);
180  }

254  public void DrawCell(float x, float y, float w, float h, float elevation, float flatten)
255  {
256  var data = terrain.terrainData;
257  var terrainWidth = data.heightmapResolution;
258  var terrainHeight = data.heightmapResolution;
259 
260  var fx = heightmapFrameStart.x;
261  var fy = heightmapFrameStart.y;
262  var fw = heightmapFrameSize.x;
263  var fh = heightmapFrameSize.y;
264 
265  flatten = Mathf.Clamp01(flatten);
266 
267  int x1, y1, x2, y2;
268  WorldToTerrainCoord(terrain, x, y, out x1, out y1);
269  WorldToTerrainCoord(terrain, x + w, y + h, out x2, out y2);
270 
271  for (int ix = x1; ix <= x2; ix++)
272  {
273  for (int iy = y1; iy <= y2; iy++)
274  {
275  if (ix < 0 || ix >= terrainWidth || iy < 0 || iy >= terrainHeight)
276  {
277  // Out of terrain boundaries. Ignore
278  continue;
279  }
280  int tx = Mathf.Clamp(ix - fx, 0, fw - 1);
281  int ty = Mathf.Clamp(iy - fy, 0, fh - 1);
282  var normalizedElevation = GetElevation(elevation);
283  float e0 = heights[ty, tx];
284  float e1 = normalizedElevation;
285  float e = Mathf.Lerp(e0, e1, flatten);
286  heights[ty, tx] = e;
287  lockedCells[ty, tx] = true;
288  }
289  }
290  }

307  public void SmoothCell(float x, float y, float w, float h, float elevation, int smoothingDistance, AnimationCurve smoothingCurve, float flatten)
308  {
309  var fx = heightmapFrameStart.x;
310  var fy = heightmapFrameStart.y;
311  var fw = heightmapFrameSize.x;
312  var fh = heightmapFrameSize.y;
313 
314  flatten = Mathf.Clamp01(flatten);
315 
316  int x1, y1, x2, y2;
317  WorldToTerrainCoord(terrain, x, y, out x1, out y1);
318  WorldToTerrainCoord(terrain, x + w, y + h, out x2, out y2);
319  var bounds = new Rectangle(x1, y1, x2 - x1, y2 - y1);
320  var startElevation = GetElevation(elevation);
321  for (int i = 1; i <= smoothingDistance; i++)
322  {
323  bounds = Rectangle.ExpandBounds(bounds, 1);
324  var borderPoints = bounds.GetBorderPoints();
325  foreach (var borderPoint in borderPoints)
326  {
327  var ix = borderPoint.x;
328  var iy = borderPoint.z;
329  int tx = Mathf.Clamp(ix - fx, 0, fw - 1);
330  int ty = Mathf.Clamp(iy - fy, 0, fh - 1);
331 
332  if (lockedCells[ty, tx])
333  {
334  continue;
335  }
336 
337  var endElevation = heights[ty, tx];
338  var frameFlatten = (float)(smoothingDistance - i) / (smoothingDistance);
339  //frameSmooth = Mathf.Clamp01(smoothingCurve.Evaluate(frameSmooth));
340  frameFlatten = Remap(frameFlatten, 0, 1, 0, flatten, true);
341  var cellElevation = endElevation + (startElevation - endElevation) * frameFlatten;
342  heights[ty, tx] = cellElevation;
343  }
344  }
345  }

347  float Remap(float value, float oldMin, float oldMax, float newMin, float newMax, bool clamp)
348  {
349  var ratio = (value - oldMin) / (oldMax - oldMin);
350  if (clamp)
351  {
352  ratio = Mathf.Clamp01(ratio);
353  }
354  return newMin + (newMax - newMin) * ratio;
355  }
