20  void AddTerrainTriangles(List<Triangle3> triangles, Terrain terrain) {
21  var data = terrain.terrainData;
22  var terrainSize = data.size;
23  var terrainWidth = data.heightmapResolution;
24  var terrainHeight = data.heightmapResolution;
25  var terrainScale = data.heightmapScale;
26 
27  int incrementX = Mathf.Max (1, Mathf.RoundToInt(terrainCellSize / terrainScale.x));
28  int incrementZ = Mathf.Max (1, Mathf.RoundToInt(terrainCellSize / terrainScale.z));
29 
30  var multiplierX = terrainSize.x / (terrainWidth - 1);
31  var multiplierZ = terrainSize.z / (terrainHeight - 1);
32  var heights = data.GetHeights(0, 0, terrainWidth, terrainHeight);
33 
34  var optimizedWidth = Mathf.Floor(terrainWidth / incrementX);
35  var optimizedHeight = Mathf.Floor(terrainHeight / incrementZ);
36 
37  var worldHeights = new SVector3[terrainWidth, terrainHeight];
38  var resolution = terrain.terrainData.size.y;
39  int ox = 0;
40  for (var hx = 0; hx < terrainWidth; hx += incrementX) {
41  int oz = 0;
42  for (var hz = 0; hz < terrainHeight; hz += incrementZ) {
43  var x = hx * multiplierX + terrain.transform.position.x;
44  var z = hz * multiplierZ + terrain.transform.position.z;
45  var y = heights[hx, hz] * resolution + terrain.transform.position.y;
46  worldHeights[ox, oz] = new SVector3(z, y, x);
47  oz++;
48  }
49  ox++;
50  }
51 
52  var vertices = new SVector3[4];
53  for (var hx = 0; hx < optimizedWidth - 1; hx++) {
54  for (var hz = 0; hz < optimizedHeight - 1; hz++) {
55  vertices[0] = worldHeights[hx, hz];
56  vertices[1] = worldHeights[hx, hz + 1];
57  vertices[2] = worldHeights[hx + 1, hz + 1];
58  vertices[3] = worldHeights[hx + 1, hz];
59 
60  triangles.Add (new Triangle3(
61  vertices[0],
62  vertices[1],
63  vertices[2]
64  ));
65 
66  triangles.Add (new Triangle3(
67  vertices[2],
68  vertices[3],
69  vertices[0]
70  ));
71  }
72  }
73  }
74  }
