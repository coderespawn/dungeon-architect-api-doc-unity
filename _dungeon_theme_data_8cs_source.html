15  public void BuildFromGraph(Graph graph) {
16  Props.Clear();
17  if (graph == null) {
18  return;
19  }
20  var nodes = graph.Nodes.ToArray();
21  Array.Sort (nodes, new LeftToRightNodeComparer ());
22 
23  foreach (var node in nodes) {
24  if (node is VisualNode) {
25  var visualNode = node as VisualNode;
26 
27  foreach (var meshParentNode in visualNode.GetParentNodes()) {
28  if (meshParentNode is MarkerNode) {
29  var markerNode = meshParentNode as MarkerNode;
30 
31  DungeonThemeItem item = null;
32  if (visualNode is GameObjectNode) {
33  var meshItem = new GameObjectDungeonThemeItem ();
34  var goNode = visualNode as GameObjectNode;
35 
36  meshItem.Template = goNode.Template;
37 
38  item = meshItem;
39  }
40  else if (visualNode is GameObjectArrayNode) {
41  var arrayPropData = new GameObjectArrayDungeonThemeItem ();
42  var arrayNode = visualNode as GameObjectArrayNode;
43 
44  if (arrayNode == null || arrayNode.Templates == null) {
45  arrayPropData.Templates = new GameObject[0];
46  } else {
47  var count = arrayNode.Templates.Length;
48  arrayPropData.Templates = new GameObject[count];
49  System.Array.Copy (arrayNode.Templates, arrayPropData.Templates, count);
50  }
51  item = arrayPropData;
52  }
53  else if (visualNode is SpriteNode) {
54  var spriteItem = new SpriteDungeonThemeItem();
55  var spriteNode = visualNode as SpriteNode;
56 
57  spriteItem.sprite = spriteNode.sprite;
58  spriteItem.color = spriteNode.color;
59  spriteItem.materialOverride = spriteNode.materialOverride;
60  spriteItem.sortingLayerName = spriteNode.sortingLayerName;
61  spriteItem.orderInLayer = spriteNode.orderInLayer;
62 
63  spriteItem.collisionType = spriteNode.collisionType;
64  spriteItem.physicsMaterial = spriteNode.physicsMaterial;
65  spriteItem.physicsOffset = spriteNode.physicsOffset;
66  spriteItem.physicsSize = spriteNode.physicsSize;
67  spriteItem.physicsRadius = spriteNode.physicsRadius;
68 
69  item = spriteItem;
70  }
71  else {
72  // Unsupported visual node type
73  continue;
74  }
75 
76  // Set the common settings
77  item.NodeId = visualNode.Id.ToString();
78  item.AttachToSocket = markerNode.Caption;
79  item.Affinity = visualNode.attachmentProbability;
80  item.ConsumeOnAttach = visualNode.consumeOnAttach;
81  item.Offset = visualNode.offset;
82  item.StaticState = visualNode.IsStatic ? DungeonThemeItemStaticMode.ForceStatic : DungeonThemeItemStaticMode.Unchanged;
83  item.affectsNavigation = visualNode.affectsNavigation;
84  item.UseSelectionRule = visualNode.selectionRuleEnabled;
85  item.SelectorRuleClassName = visualNode.selectionRuleClassName;
86  item.UseTransformRule = visualNode.transformRuleEnabled;
87  item.TransformRuleClassName = visualNode.transformRuleClassName;
88  item.useSpatialConstraint = visualNode.useSpatialConstraint;
89  item.spatialConstraint = visualNode.spatialConstraint;
90 
91  var emitterNodes = visualNode.GetChildNodes();
92  foreach (var childNode in emitterNodes) {
93  if (childNode is MarkerEmitterNode) {
94  var emitterNode = childNode as MarkerEmitterNode;
95  if (emitterNode.Marker != null) {
96  PropChildSocketData childData = new PropChildSocketData();
97  childData.Offset = emitterNode.offset;
98  childData.SocketType = emitterNode.Marker.Caption;
99  item.ChildSockets.Add (childData);
100  }
101  }
102  }
103  Props.Add(item);
104  }
105  }
106  }
107  }
108  }
