144  {
145  //TODO we're object-oriented, can prevent this state from ever happening.
146  if (nav == null)
147  throw new InvalidOperationException("TODO prevent this state from ever occuring");
148 
149  //randomly pick one tile
150  //assume all tiles cover roughly the same area
151  MeshTile tile = null;
152  float tsum = 0.0f;
153 
154  for (int i = 0; i < nav.TileCount; i++)
155  {
156  MeshTile t = nav[i];
157 
158  if (t == null || t.Header == null)
159  continue;
160 
161  //choose random tile using reservoir sampling
162  float area = 1.0f;
163  tsum += area;
164  float u = (float)rand.NextDouble();
165  if (u * tsum <= area)
166  tile = t;
167  }
168 
169  //TODO why?
170  if (tile == null)
171  throw new InvalidOperationException("No tiles?");
172 
173  //randomly pick one polygon weighted by polygon area
174  Poly poly = null;
175  int polyRef = 0;
176  int polyBase = nav.GetPolyRefBase(tile);
177 
178  float areaSum = 0.0f;
179  for (int i = 0; i < tile.Header.PolyCount; i++)
180  {
181  Poly p = tile.Polys[i];
182 
183  //don't return off-mesh connection polygons
184  if (p.PolyType != PolygonType.Ground)
185  continue;
186 
187  int reference = polyBase | i;
188 
189  //calculate area of polygon
190  float polyArea = 0.0f;

192  for (int j = 2; j < p.VertCount; j++)

198  //choose random polygon weighted by area, usig resevoir sampling

208  //TODO why?

238  public void FindRandomPointAroundCircle(NavPoint center, float radius, out NavPoint randomPoint)
239  {
240  //TODO fix state
241  if (nav == null || nodePool == null || openList == null)
242  throw new InvalidOperationException("Something null");
243 
244  //validate input
245  if (center.Polygon == 0)
246  throw new ArgumentOutOfRangeException("startRef", "Null poly reference");
247 
248  if (!nav.IsValidPolyRef(center.Polygon))
249  throw new ArgumentException("startRef", "Poly reference is not valid for this navmesh");
250 
251  MeshTile startTile;
252  Poly startPoly;
253  nav.TryGetTileAndPolyByRefUnsafe(center.Polygon, out startTile, out startPoly);
254 
255  nodePool.Clear();
256  openList.Clear();
257 
258  Node startNode = nodePool.GetNode(center.Polygon);
259  startNode.Pos = center.Position;
260  startNode.ParentIdx = 0;
261  startNode.cost = 0;
262  startNode.total = 0;
263  startNode.Id = center.Polygon;
264  startNode.Flags = NodeFlags.Open;
265  openList.Push(startNode);
266 
267  float radiusSqr = radius * radius;
268  float areaSum = 0.0f;
269 
270  MeshTile randomTile = null;
271  Poly randomPoly = null;
272  int randomPolyRef = 0;
273 
274  while (openList.Count > 0)
275  {
276  Node bestNode = openList.Pop();
277  SetNodeFlagClosed(ref bestNode);
278 
279  //get poly and tile
280  int bestRef = bestNode.Id;
281  MeshTile bestTile;
282  Poly bestPoly;
283  nav.TryGetTileAndPolyByRefUnsafe(bestRef, out bestTile, out bestPoly);
284 
285  //place random locations on ground
286  if (bestPoly.PolyType == PolygonType.Ground)
287  {
288  //calculate area of polygon
289  float polyArea = 0.0f;
290  float area;
291  for (int j = 2; j < bestPoly.VertCount; j++)
292  {
293  Triangle3.Area2D(ref bestTile.Verts[bestPoly.Verts[0]], ref bestTile.Verts[bestPoly.Verts[j - 1]], ref bestTile.Verts[bestPoly.Verts[j]], out area);
294  polyArea += area;
295  }
296 
297  //choose random polygon weighted by area using resevoir sampling
298  areaSum += polyArea;
299  float u = (float)rand.NextDouble();
300  if (u * areaSum <= polyArea)
301  {
302  randomTile = bestTile;
303  randomPoly = bestPoly;
304  randomPolyRef = bestRef;
305  }
306  }
307 
308  //get parent poly and tile
309  int parentRef = 0;
310  MeshTile parentTile;
311  Poly parentPoly;
312  if (bestNode.ParentIdx != 0)
313  parentRef = nodePool.GetNodeAtIdx(bestNode.ParentIdx).Id;
314  if (parentRef != 0)
315  nav.TryGetTileAndPolyByRefUnsafe(parentRef, out parentTile, out parentPoly);
316 
317  for (int i = bestPoly.FirstLink; i != Link.Null; i = bestTile.Links[i].Next)
318  {
319  Link link = bestTile.Links[i];
320  int neighbourRef = link.Reference;
321 
322  //skip invalid neighbours and do not follor back to parent
323  if (neighbourRef == 0 || neighbourRef == parentRef)
324  continue;
325 
326  //expand to neighbour
327  MeshTile neighbourTile;
328  Poly neighbourPoly;
329  nav.TryGetTileAndPolyByRefUnsafe(neighbourRef, out neighbourTile, out neighbourPoly);
330 
331  //find edge and calculate distance to edge
332  Vector3 va = new Vector3();
333  Vector3 vb = new Vector3();
334  if (!GetPortalPoints(bestRef, bestPoly, bestTile, neighbourRef, neighbourPoly, neighbourTile, ref va, ref vb))
335  continue;
336 
337  //if circle isn't touching next polygon, skip it
338  float tseg;
339  float distSqr = Distance.PointToSegment2DSquared(ref center.Position, ref va, ref vb, out tseg);
340  if (distSqr > radiusSqr)
341  continue;
342 
343  Node neighbourNode = nodePool.GetNode(neighbourRef);
344  if (neighbourNode == null)
345  continue;
346 
347  if (IsInClosedList(neighbourNode))
348  continue;
349 
350  //cost
351  if (neighbourNode.Flags == 0)
352  neighbourNode.Pos = Vector3.Lerp(va, vb, 0.5f);
353 
354  float total = bestNode.total + (bestNode.Pos - neighbourNode.Pos).Length();
355 
356  //node is already in open list and new result is worse, so skip
357  if (IsInOpenList(neighbourNode) && total >= neighbourNode.total)
358  continue;
359 
360  neighbourNode.Id = neighbourRef;
361  neighbourNode.Flags = RemoveNodeFlagClosed(neighbourNode);
362  neighbourNode.ParentIdx = nodePool.GetNodeIdx(bestNode);
363  neighbourNode.total = total;
364 
365  if (IsInOpenList(neighbourNode))
366  {
367  openList.Modify(neighbourNode);
368  }
369  else
370  {
371  neighbourNode.Flags = NodeFlags.Open;
372  openList.Push(neighbourNode);
373  }
374  }
375  }
376 
377  //TODO invalid state.
378  if (randomPoly == null)
379  throw new InvalidOperationException("Poly null?");
380 
381  Vector3 randomPt;
382  FindRandomPointOnPoly(randomTile, randomPoly, randomPolyRef, out randomPt);
383 
384  randomPoint = new NavPoint(randomPolyRef, randomPt);
385  }

397  public bool FindPath(ref NavPoint startPt, ref NavPoint endPt, List<int> path)
398  {
399  //reset path of polygons
400  path.Clear();
401 
402  int startRef = startPt.Polygon;
403  Vector3 startPos = startPt.Position;
404  int endRef = endPt.Polygon;
405  Vector3 endPos = endPt.Position;
406 
407  if (startRef == 0 || endRef == 0)
408  return false;
409 
410  //path can't store any elements
411  if (path.Capacity == 0)
412  return false;
413 
414  //validate input
415  if (!nav.IsValidPolyRef(startRef) || !nav.IsValidPolyRef(endRef))
416  return false;
417 
418  //special case: both start and end are in the same polygon
419  if (startRef == endRef)
420  {
421  path.Add(startRef);
422  return true;
423  }
424 
425  nodePool.Clear();
426  openList.Clear();
427 
428  //initial node is located at the starting position
429  Node startNode = nodePool.GetNode(startRef);
430  startNode.Pos = startPos;
431  startNode.ParentIdx = 0;
432  startNode.cost = 0;
433  startNode.total = (startPos - endPos).Length() * H_SCALE;
434  startNode.Id = startRef;
435  startNode.Flags = NodeFlags.Open;
436  openList.Push(startNode);
437 
438  Node lastBestNode = startNode;
439  float lastBestTotalCost = startNode.total;
440 
441  while (openList.Count > 0)
442  {
443  //remove node from open list and put it in closed list
444  Node bestNode = openList.Pop();
445  SetNodeFlagClosed(ref bestNode);
446 
447  //reached the goal. stop searching
448  if (bestNode.Id == endRef)
449  {
450  lastBestNode = bestNode;
451  break;
452  }
453 
454  //get current poly and tile
455  int bestRef = bestNode.Id;
456  MeshTile bestTile;
457  Poly bestPoly;
458  nav.TryGetTileAndPolyByRefUnsafe(bestRef, out bestTile, out bestPoly);
459 
460  //get parent poly and tile
461  int parentRef = 0;
462  MeshTile parentTile;
463  Poly parentPoly;
464  if (bestNode.ParentIdx != 0)
465  parentRef = nodePool.GetNodeAtIdx(bestNode.ParentIdx).Id;
466  if (parentRef != 0)
467  nav.TryGetTileAndPolyByRefUnsafe(parentRef, out parentTile, out parentPoly);
468 
469  //examine neighbors
470  for (int i = bestPoly.FirstLink; i != Link.Null; i = bestTile.Links[i].Next)
471  {
472  int neighbourRef = bestTile.Links[i].Reference;
473 
474  //skip invalid ids and do not expand back to where we came from
475  if (neighbourRef == 0 || neighbourRef == parentRef)
476  continue;
477 
478  //get neighbour poly and tile
479  MeshTile neighbourTile;
480  Poly neighbourPoly;
481  nav.TryGetTileAndPolyByRefUnsafe(neighbourRef, out neighbourTile, out neighbourPoly);
482 
483  Node neighbourNode = nodePool.GetNode(neighbourRef);
484  if (neighbourNode == null)
485  continue;
486 
487  //if node is visited the first time, calculate node position
488  if (neighbourNode.Flags == 0)
489  {
490  GetEdgeMidPoint(bestRef, bestPoly, bestTile, neighbourRef, neighbourPoly, neighbourTile, ref neighbourNode.Pos);
491  }
492 
493  //calculate cost and heuristic
494  float cost = 0;
495  float heuristic = 0;
496 
497  //special case for last node
498  if (neighbourRef == endRef)
499  {
500  //cost
501  float curCost = GetCost(bestNode.Pos, neighbourNode.Pos, bestPoly);
502  float endCost = GetCost(neighbourNode.Pos, endPos, neighbourPoly);
503 
504  cost = bestNode.cost + curCost + endCost;
505  heuristic = 0;
506  }
507  else
508  {
509  //cost
510  float curCost = GetCost(bestNode.Pos, neighbourNode.Pos, bestPoly);
511 
512  cost = bestNode.cost + curCost;
513  heuristic = (neighbourNode.Pos - endPos).Length() * H_SCALE;
514  }
515 
516  float total = cost + heuristic;
517 
518  //the node is already in open list and new result is worse, skip
519  if (IsInOpenList(neighbourNode) && total >= neighbourNode.total)
520  continue;
521 
522  //the node is already visited and processesd, and the new result is worse, skip
523  if (IsInClosedList(neighbourNode) && total >= neighbourNode.total)
524  continue;
525 
526  //add or update the node
527  neighbourNode.ParentIdx = nodePool.GetNodeIdx(bestNode);
528  neighbourNode.Id = neighbourRef;
529  neighbourNode.Flags = RemoveNodeFlagClosed(neighbourNode);
530  neighbourNode.cost = cost;
531  neighbourNode.total = total;
532 
533  if (IsInOpenList(neighbourNode))
534  {
535  //already in open, update node location
536  openList.Modify(neighbourNode);
537  }
538  else
539  {
540  //put the node in the open list
541  SetNodeFlagOpen(ref neighbourNode);
542  openList.Push(neighbourNode);
543  }
544 
545  //update nearest node to target so far
546  if (heuristic < lastBestTotalCost)
547  {
548  lastBestTotalCost = heuristic;
549  lastBestNode = neighbourNode;
550  }
551  }
552  }
553 
554  //save path
555  Node node = lastBestNode;
556  do
557  {
558  path.Add(node.Id);
559  if (path.Count >= path.Capacity)
560  break;
561 
562  node = nodePool.GetNodeAtIdx(node.ParentIdx);
563  }
564  while (node != null);
565 
566  //reverse the path since it's backwards
567  path.Reverse();
568 
569  return true;
570  }

586  public bool FindStraightPath(Vector3 startPos, Vector3 endPos, int[] path, int pathSize, Vector3[] straightPath, int[] straightPathFlags, int[] straightPathRefs, ref int straightPathCount, int maxStraightPath, PathBuildFlags options)
587  {
588  straightPathCount = 0;
589 
590  if (path.Length == 0)
591  return false;
592 
593  bool stat = false;
594 
595  Vector3 closestStartPos = new Vector3();
596  ClosestPointOnPolyBoundary(path[0], startPos, ref closestStartPos);
597 
598  Vector3 closestEndPos = new Vector3();
599  ClosestPointOnPolyBoundary(path[pathSize - 1], endPos, ref closestEndPos);
600 
601  stat = AppendVertex(closestStartPos, PathfindingCommon.STRAIGHTPATH_START, path[0], straightPath, straightPathFlags, straightPathRefs, ref straightPathCount, maxStraightPath);
602 
603  if (!stat)
604  return true;
605 
606  if (pathSize > 1)
607  {
608  Vector3 portalApex = closestStartPos;
609  Vector3 portalLeft = portalApex;
610  Vector3 portalRight = portalApex;
611  int apexIndex = 0;
612  int leftIndex = 0;
613  int rightIndex = 0;
614 
615  PolygonType leftPolyType = 0;
616  PolygonType rightPolyType = 0;
617 
618  int leftPolyRef = path[0];
619  int rightPolyRef = path[0];
620 
621  for (int i = 0; i < pathSize; i++)
622  {
623  Vector3 left = new Vector3();
624  Vector3 right = new Vector3();
625  PolygonType fromType = 0, toType = 0;
626 
627  if (i + 1 < pathSize)
628  {
629  //next portal
630  if (GetPortalPoints(path[i], path[i + 1], ref left, ref right, ref fromType, ref toType) == false)
631  {
632  //failed to get portal points means path[i + 1] is an invalid polygon
633  //clamp end point to path[i] and return path so far
634  if (ClosestPointOnPolyBoundary(path[i], endPos, ref closestEndPos) == false)
635  {
636  //first polygon is invalid
637  return false;
638  }
639 
640  if ((options & (PathBuildFlags.AreaCrossingVertices | PathBuildFlags.AllCrossingVertices)) != 0)
641  {
642  //append portals
643  stat = AppendPortals(apexIndex, i, closestEndPos, path, straightPath, straightPathFlags, straightPathRefs, ref straightPathCount, maxStraightPath, options);
644  }
645 
646  stat = AppendVertex(closestEndPos, 0, path[i], straightPath, straightPathFlags, straightPathRefs, ref straightPathCount, maxStraightPath);
647 
648  return true;
649  }
650 
651  //if starting really close to the portal, advance
652  if (i == 0)
653  {
654  float t;
655  if (Distance.PointToSegment2DSquared(ref portalApex, ref left, ref right, out t) < 0.001 * 0.001)
656  continue;
657  }
658  }
659  else
660  {
661  //end of the path
662  left = closestEndPos;
663  right = closestEndPos;
664 
665  fromType = toType = PolygonType.Ground;
666  }
667 
668  //right vertex
669  float triArea2D;
670  Triangle3.Area2D(ref portalApex, ref portalRight, ref right, out triArea2D);
671  if (triArea2D <= 0.0)
672  {
673  Triangle3.Area2D(ref portalApex, ref portalLeft, ref right, out triArea2D);
674  if (portalApex == portalRight || triArea2D > 0.0)
675  {
676  portalRight = right;
677  rightPolyRef = (i + 1 < pathSize) ? path[i + 1] : 0;
678  rightPolyType = toType;
679  rightIndex = i;
680  }
681  else
682  {
683  //append portals along current straight path segment
684  if ((options & (PathBuildFlags.AreaCrossingVertices | PathBuildFlags.AllCrossingVertices)) != 0)
685  {
686  stat = AppendPortals(apexIndex, leftIndex, portalLeft, path, straightPath, straightPathFlags, straightPathRefs, ref straightPathCount, maxStraightPath, options);
687 
688  if (stat != true)
689  return true;
690  }
691 
692  portalApex = portalLeft;
693  apexIndex = leftIndex;
694 
695  int flags = 0;
696  if (leftPolyRef == 0)
697  flags = PathfindingCommon.STRAIGHTPATH_END;
698  else if (leftPolyType == PolygonType.OffMeshConnection)
699  flags = PathfindingCommon.STRAIGHTPATH_OFFMESH_CONNECTION;
700 
701  int reference = leftPolyRef;
702 
703  //append or update vertex
704  stat = AppendVertex(portalApex, flags, reference, straightPath, straightPathFlags, straightPathRefs, ref straightPathCount, maxStraightPath);
705 
706  if (stat != true)
707  return true;
708 
709  portalLeft = portalApex;
710  portalRight = portalApex;
711  leftIndex = apexIndex;
712  rightIndex = apexIndex;
713 
714  //restart
715  i = apexIndex;
716 
717  continue;
718  }
719  }
720 
721  //left vertex
722  Triangle3.Area2D(ref portalApex, ref portalLeft, ref left, out triArea2D);
723  if (triArea2D >= 0.0)
724  {
725  Triangle3.Area2D(ref portalApex, ref portalRight, ref left, out triArea2D);
726  if (portalApex == portalLeft || triArea2D < 0.0f)
727  {
728  portalLeft = left;
729  leftPolyRef = (i + 1 < pathSize) ? path[i + 1] : 0;
730  leftPolyType = toType;
731  leftIndex = i;
732  }
733  else
734  {
735  if ((options & (PathBuildFlags.AreaCrossingVertices | PathBuildFlags.AllCrossingVertices)) != 0)
736  {
737  stat = AppendPortals(apexIndex, rightIndex, portalRight, path, straightPath, straightPathFlags, straightPathRefs, ref straightPathCount, maxStraightPath, options);
738 
739  if (stat != true)
740  return true;
741  }
742 
743  portalApex = portalRight;
744  apexIndex = rightIndex;
745 
746  int flags = 0;
747  if (rightPolyRef == 0)
748  flags = PathfindingCommon.STRAIGHTPATH_END;
749  else if (rightPolyType == PolygonType.OffMeshConnection)
750  flags = PathfindingCommon.STRAIGHTPATH_OFFMESH_CONNECTION;
751 
752  int reference = rightPolyRef;
753 
754  //append or update vertex
755  stat = AppendVertex(portalApex, flags, reference, straightPath, straightPathFlags, straightPathRefs, ref straightPathCount, maxStraightPath);
756 
757  if (stat != true)
758  return true;
759 
760  portalLeft = portalApex;
761  portalRight = portalApex;
762  leftIndex = apexIndex;
763  rightIndex = apexIndex;
764 
765  //restart
766  i = apexIndex;
767 
768  continue;
769  }
770  }
771  }
772 
773  //append portals along the current straight line segment
774  if ((options & (PathBuildFlags.AreaCrossingVertices | PathBuildFlags.AllCrossingVertices)) != 0)
775  {
776  stat = AppendPortals(apexIndex, pathSize - 1, closestEndPos, path, straightPath, straightPathFlags, straightPathRefs, ref straightPathCount, maxStraightPath, options);
777 
778  if (stat != true)
779  return true;
780  }
781  }
782 
783  stat = AppendVertex(closestEndPos, PathfindingCommon.STRAIGHTPATH_END, 0, straightPath, straightPathFlags, straightPathRefs, ref straightPathCount, maxStraightPath);
784 
785  return true;
786  }

797  public bool MoveAlongSurface(NavPoint startPoint, Vector3 endPos, ref Vector3 resultPos, List<int> visited)
798  {
799  if (nav == null)
800  return false;
801  if (tinyNodePool == null)
802  return false;
803 
804  visited.Clear();
805 
806  //validate input
807  if (startPoint.Polygon == 0)
808  return false;
809  if (!nav.IsValidPolyRef(startPoint.Polygon))
810  return false;
811 
812  int MAX_STACK = 48;
813  Queue<Node> nodeQueue = new Queue<Node>(MAX_STACK);
814 
815  tinyNodePool.Clear();
816 
817  Node startNode = tinyNodePool.GetNode(startPoint.Polygon);
818  startNode.ParentIdx = 0;
819  startNode.cost = 0;
820  startNode.total = 0;
821  startNode.Id = startPoint.Polygon;
822  startNode.Flags = NodeFlags.Closed;
823  nodeQueue.Enqueue(startNode);
824 
825  Vector3 bestPos = startPoint.Position;
826  float bestDist = float.MaxValue;
